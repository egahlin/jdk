 26 package jdk.jfr;
 27 


 28 import java.util.Set;
 29 


 30 /**
 31  * Base class to extend to create setting controls.
 32  * <p>
 33  * The following example shows a naive implementation of a setting control for
 34  * regular expressions:
 35  *
 36  * {@snippet class="Snippets" region="SettingControlOverview1"}
 37  *
 38  * The {@code setValue(String)}, {@code getValue()} and
 39  * {@code combine(Set<String>)} methods are invoked when a setting value
 40  * changes, which typically happens when a recording is started or stopped. The
 41  * {@code combine(Set<String>)} method is invoked to resolve what value to use
 42  * when multiple recordings are running at the same time.
 43  * <p>
 44  * The setting control must have a default constructor that can be invoked when
 45  * the event is registered.
 46  * <p>
 47  * To use a setting control with an event, add a method that returns a
 48  * {@code boolean} value and takes the setting control as a parameter. Annotate
 49  * the method with the {@code @SettingDefinition} annotation. By default, the
 50  * method name is used as the setting name, but the name can be set explicitly
 51  * by using the {@code @Name} annotation. If the method returns {@code true},
 52  * the event will be committed.
 53  * <p>
 54  * It is recommended that the {@code setValue(String)} method updates an
 55  * efficient data structure that can be quickly checked when the event is
 56  * committed.
 57  * <p>
 58  * The following example shows how to create an event that uses the
 59  * regular expression filter defined above.
 60  *
 61  * {@snippet class="Snippets" region="SettingControlOverview2"}
 62  *
 63  * <p>
 64  * The following example shows how an event can be filtered by assigning the
 65  * {@code "uriFilter"} setting with the specified regular expressions.
 66  *
 67  * {@snippet class="Snippets" region="SettingControlOverview3"}
 68  *
 69  * @see SettingDefinition
 70  *
 71  * @since 9
 72  */
 73 @MetadataDefinition
 74 public abstract class SettingControl {
 75 




 76     /**
 77      * Constructor for invocation by subclass constructors.
 78      */

 79     protected SettingControl() {















 80     }
 81 
 82     /**
 83      * Combines the setting values for all running recordings into one value when
 84      * multiple recordings are running at the same time,
 85      * <p>
 86      * The semantics of how setting values are combined depends on the setting
 87      * control that is implemented, but all recordings should get at least all the
 88      * events they request.
 89      * <p>
 90      * This method should have no side effects, because the caller might cache values.
 91      * This method should never return {@code null} or throw an exception. If a
 92      * value is not valid for this setting control, the value should be ignored.
 93      * <p>
 94      * Examples:
 95      * <p>
 96      * if the setting control represents a threshold and three recordings are
 97      * running at the same time with the setting values {@code "10 ms"},
 98      * {@code "8 s"}, and {@code  "1 ms"}, this method returns {@code "1 ms"}
 99      * because it means that all recordings get at least all the requested data.
100      * <p>
101      * If the setting control represents a set of names and two recordings are
102      * running at the same time with the setting values {@code "Smith, Jones"} and {@code "Jones,
103      * Williams"} the returned value is {@code "Smith, Jones, Williams"} because all names would be accepted.
104      * <p>
105      * If the setting control represents a boolean condition and four recordings are
106      * running at the same time with the following values {@code "true"}, {@code "false"}, {@code "false"}, and
107      * {@code "incorrect"}, this method returns {@code "true"}, because all
108      * recordings get at least all the requested data.
109      *
110      * @param settingValues the set of values, not {@code null}
111      *
112      * @return the value to use, not {@code null}
113      */
114     public abstract String combine(Set<String> settingValues);
115 
116     /**
117      * Sets the value for this setting.
118      * <p>
119      * If the setting value is not valid for this setting, this method
120      * does not throw an exception. Instead, the value is ignored.
121      *
122      * @param settingValue the string value, not {@code null}
123      */
124     public abstract void setValue(String settingValue);
125 
126     /**
127      * Returns the currently used value for this setting, not {@code null}.
128      * <p>
129      * The value returned by this method is valid as an argument to both
130      * the {@code setValue(String)} method and {@code combine(Set)} method.
131      * <p>
132      * This method is invoked when an event is registered to obtain the
133      * default value. It is therefore important that a valid value can be
134      * returned immediately after an instance of this class is created. It is
135      * not valid to return {@code null}.
136      *
137      * @return the setting value, not {@code null}
138      */
139     public abstract String getValue();
140 }
