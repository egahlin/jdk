176     /**
177      * Constructs a {@code FlightRecorderPermission} with the specified name.
178      *
179      * @param name the permission name, must be either
180      *        {@code "accessFlightRecorder"} or {@code "registerEvent"}, not
181      *        {@code null}
182      *
183      * @throws IllegalArgumentException if {@code name} is empty or not valid
184      */
185     public FlightRecorderPermission(String name) {
186         super(Objects.requireNonNull(name, "name"));
187         if (!name.equals(SecuritySupport.ACCESS_FLIGHT_RECORDER) && !name.equals(SecuritySupport.REGISTER_EVENT)) {
188             throw new IllegalArgumentException("name: " + name);
189         }
190     }
