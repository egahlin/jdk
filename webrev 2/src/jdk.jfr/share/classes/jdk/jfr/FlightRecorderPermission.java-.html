 26 package jdk.jfr;
 27 
 28 import java.security.AccessControlContext;
 29 import java.util.List;
 30 import java.util.Map;
 31 import java.util.Objects;
 32 
 33 import jdk.jfr.internal.PlatformEventType;
 34 import jdk.jfr.internal.PlatformRecorder;
 35 import jdk.jfr.internal.PlatformRecording;
 36 import jdk.jfr.internal.PrivateAccess;
 37 import jdk.jfr.internal.SecuritySupport;
 38 import jdk.jfr.internal.Type;
 39 import jdk.jfr.internal.management.EventSettingsModifier;
 40 
 41 /**
 42  * Permission for controlling access to Flight Recorder.
 43  *
 44  * @apiNote
 45  * This permission cannot be used for controlling access to resources
 46  * as the Security Manager is no longer supported.
 47  *
 48  * @since 9
 49  *
 50  * @see java.security.BasicPermission
 51  * @see java.security.Permission
 52  * @see java.security.Permissions
 53  * @see java.security.PermissionCollection
 54  * @see java.lang.SecurityManager
 55  *
 56  */
 57 @SuppressWarnings("serial")
 58 public final class FlightRecorderPermission extends java.security.BasicPermission {
 59 
 60     // Purpose of InternalAccess is to give classes in jdk.jfr.internal
 61     // access to package private methods in this package (jdk.jfr).
 62     //
 63     // The initialization could be done in any class in this package,
 64     // but this one was chosen because it is light weight and
 65     // lacks dependencies on other public classes.
 66     static {
 67         PrivateAccess.setPrivateAccess(new InternalAccess());
 68     }
 69 
 70     private static final class InternalAccess extends PrivateAccess {
 71 
 72         @Override
 73         public Type getType(Object o) {
 74             if (o instanceof AnnotationElement ae) {
 75                 return ae.getType();
 76             }
 77             if (o instanceof EventType et) {
 78                 return et.getType();
 79             }
 80             if (o instanceof ValueDescriptor vd) {
 81                 return vd.getType();
 82             }
 83             if (o instanceof SettingDescriptor sd) {
 84                 return sd.getType();
 85             }
 86             throw new Error("Unknown type " + o.getClass());
 87         }
 88 
 89         @Override
 90         public Configuration newConfiguration(String name, String label, String description, String provider, Map<String, String> settings, String contents) {
 91             return new Configuration(name, label, description, provider, settings, contents);
 92         }
 93 
 94         @Override
 95         public EventType newEventType(PlatformEventType platformEventType) {
 96             return new EventType(platformEventType);
 97         }
 98 
 99         @Override
100         public AnnotationElement newAnnotation(Type annotationType, List<Object> values, boolean boot) {
101             return new AnnotationElement(annotationType, values, boot);
102         }
103 
104         @Override
105         public ValueDescriptor newValueDescriptor(String name, Type fieldType, List<AnnotationElement> annos, int dimension, boolean constantPool, String fieldName) {
106             return new ValueDescriptor(fieldType, name, annos, dimension, constantPool, fieldName);
107         }
108 
109         @Override
110         public PlatformRecording getPlatformRecording(Recording r) {
111             return r.getInternal();
112         }
113 
114         @Override
115         public PlatformEventType getPlatformEventType(EventType eventType) {
116             return eventType.getPlatformEventType();
117         }
118 
119         @Override
120         public boolean isConstantPool(ValueDescriptor v) {
121             return v.isConstantPool();
122         }
123 
124         @Override
125         public void setAnnotations(ValueDescriptor v, List<AnnotationElement> a) {
126             v.setAnnotations(a);
127         }
128 
129         @Override
130         public void setAnnotations(SettingDescriptor s, List<AnnotationElement> a) {
131            s.setAnnotations(a);
132         }
133 
134         @Override
135         public String getFieldName(ValueDescriptor v) {
136             return v.getJavaFieldName();
137         }
138 
139         @Override
140         public ValueDescriptor newValueDescriptor(Class<?> type, String name) {
141             return new ValueDescriptor(type, name, List.of(), true);
142         }
143 
144         @Override
145         public SettingDescriptor newSettingDescriptor(Type type, String name, String defaultValue, List<AnnotationElement> annotations) {
146             return new SettingDescriptor(type, name, defaultValue, annotations);
147         }
148 
149         @Override
150         public boolean isUnsigned(ValueDescriptor v) {
151             return v.isUnsigned();
152         }
153 
154         @Override
155         public PlatformRecorder getPlatformRecorder() {
156             return FlightRecorder.getFlightRecorder().getInternal();
157         }
158 
159         @SuppressWarnings("removal")
160         @Override
161         public AccessControlContext getContext(SettingControl settingControl) {
162             return settingControl.getContext();
163         }
164 
165         @Override
166         public EventSettings newEventSettings(EventSettingsModifier esm) {
167             return new EventSettings.DelegatedEventSettings(esm);
168         }
169 
170         @Override
171         public boolean isVisible(EventType t) {
172             return t.isVisible();
173         }
174     }
175 
176     /**
177      * Constructs a {@code FlightRecorderPermission} with the specified name.
178      *
179      * @param name the permission name, must be either
180      *        {@code "accessFlightRecorder"} or {@code "registerEvent"}, not
181      *        {@code null}
182      *
183      * @throws IllegalArgumentException if {@code name} is empty or not valid
184      */
185     public FlightRecorderPermission(String name) {
186         super(Objects.requireNonNull(name, "name"));
187         if (!name.equals(SecuritySupport.ACCESS_FLIGHT_RECORDER) && !name.equals(SecuritySupport.REGISTER_EVENT)) {
188             throw new IllegalArgumentException("name: " + name);
189         }
190     }
191 }
