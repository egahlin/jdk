26 package jdk.jfr.internal;
27 
28 import java.io.BufferedWriter;
29 import java.io.FileNotFoundException;
30 import java.io.IOException;
31 import java.nio.file.Files;
32 import java.nio.file.Path;
33 
34 public final class WriteablePath {
35     private final Path path;
36     private final Path real;
37 
38     public WriteablePath(Path path) throws IOException {
39         // verify that the path is writeable
40         if (Files.exists(path) && !Files.isWritable(path)) {
41             // throw same type of exception as FileOutputStream
42             // constructor, if file can't be opened.
43             throw new FileNotFoundException("Could not write to file: " + path.toAbsolutePath());
44         }
45         // will throw if non-writeable
46         BufferedWriter fw = Files.newBufferedWriter(path);
47         fw.close();
48         this.path = path;
49         this.real = path.toRealPath();
50     }
51 
52     public Path getPath() {
53         return path;
54     }
55 
56     public Path getReal() {
57         return real;
58     }
59 
60     public String getRealPathText() {
61         return real.toString();
62     }
63 }
