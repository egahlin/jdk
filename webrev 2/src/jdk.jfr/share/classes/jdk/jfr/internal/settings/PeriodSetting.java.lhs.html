 26 package jdk.jfr.internal.settings;
 27 
 28 import java.util.Objects;
 29 import java.util.Set;
 30 
 31 import jdk.jfr.Description;
 32 import jdk.jfr.Label;
 33 import jdk.jfr.MetadataDefinition;
 34 import jdk.jfr.Name;

 35 import jdk.jfr.internal.PlatformEventType;
 36 import jdk.jfr.internal.Type;
 37 import jdk.jfr.internal.util.ValueParser;
 38 import static jdk.jfr.internal.util.ValueParser.MISSING;
 39 
 40 @MetadataDefinition
 41 @Label("Period")
 42 @Description("Record event at interval")
 43 @Name(Type.SETTINGS_PREFIX + "Period")
 44 public final class PeriodSetting extends JDKSettingControl {
 45     private static final long typeId = Type.getTypeId(PeriodSetting.class);
 46 
 47     public static final String EVERY_CHUNK = "everyChunk";
 48     public static final String BEGIN_CHUNK = "beginChunk";
 49     public static final String END_CHUNK = "endChunk";
 50     public static final String DEFAULT_VALUE = EVERY_CHUNK;
 51     public static final String NAME = "period";
 52     private final PlatformEventType eventType;
 53     private String value = EVERY_CHUNK;
 54 
 55     public PeriodSetting(PlatformEventType eventType) {
 56         this.eventType = Objects.requireNonNull(eventType);
 57     }
 58 
 59     @Override
 60     public String combine(Set<String> values) {
 61         boolean beginChunk = false;
 62         boolean endChunk = false;
 63         Long min = null;
 64         String text = null;
 65         for (String value : values) {
 66             switch (value) {
 67             case EVERY_CHUNK:
 68                 beginChunk = true;
 69                 endChunk = true;
 70                 break;
 71             case BEGIN_CHUNK:
 72                 beginChunk = true;
 73                 break;
 74             case END_CHUNK:
 75                 endChunk = true;
 76                 break;
 77             default:
 78                 long nanos = ValueParser.parseTimespanWithInfinity(value, MISSING);
 79                 if (nanos != MISSING) {
 80                     if (min == null || nanos < min) {
 81                         text = value;
 82                         min = nanos;
 83                     }
 84                 }
 85             }
 86         }
 87         // A specified interval trumps *_CHUNK
 88         if (min != null) {
 89             return text;
 90         }
 91         if (beginChunk && !endChunk) {
 92             return BEGIN_CHUNK;
 93         }
 94         if (!beginChunk && endChunk) {
 95             return END_CHUNK;
 96         }
 97         return DEFAULT_VALUE; // "everyChunk" is default
 98     }
 99 
100     @Override
101     public void setValue(String value) {
102         switch (value) {
103         case EVERY_CHUNK:
104             eventType.setPeriod(0, true, true);
105             break;
106         case BEGIN_CHUNK:
107             eventType.setPeriod(0, true, false);
108             break;
109         case END_CHUNK:
110             eventType.setPeriod(0, false, true);
111             break;
112         default:
113             long nanos = ValueParser.parseTimespanWithInfinity(value, MISSING);
114             if (nanos == MISSING) {
115                 return;
116             }
117             if (nanos == 0 || nanos == Long.MAX_VALUE) {
118                 eventType.setPeriod(nanos, false, false);
119             } else {
120                 eventType.setPeriod(Math.max(1, nanos / 1_000_000), false, false);
121             }
122         }
123         this.value = value;
124     }
125 
126     @Override
127     public String getValue() {
128         return value;
129     }
130 
131     public static boolean isType(long typeId) {
132         return PeriodSetting.typeId == typeId;
133     }
134 }
