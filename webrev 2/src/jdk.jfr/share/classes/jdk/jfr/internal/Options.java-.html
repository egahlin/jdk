 26 package jdk.jfr.internal;
 27 
 28 import java.io.IOException;
 29 
 30 import jdk.jfr.internal.SecuritySupport.SafePath;
 31 import jdk.internal.misc.Unsafe;
 32 
 33 import static java.nio.file.LinkOption.*;
 34 
 35 /**
 36  * Options that control Flight Recorder.
 37  *
 38  * Can be set using JFR.configure
 39  *
 40  */
 41 public final class Options {
 42 
 43     private static final long WAIT_INTERVAL = 1000; // ms;
 44 
 45     private static final long MIN_MAX_CHUNKSIZE = 1024 * 1024;
 46 
 47     private static final long DEFAULT_GLOBAL_BUFFER_COUNT = 20;
 48     private static final long DEFAULT_GLOBAL_BUFFER_SIZE = 524288;
 49     private static final long DEFAULT_MEMORY_SIZE = DEFAULT_GLOBAL_BUFFER_COUNT * DEFAULT_GLOBAL_BUFFER_SIZE;
 50     private static long DEFAULT_THREAD_BUFFER_SIZE;
 51     private static final int DEFAULT_STACK_DEPTH = 64;
 52     private static final long DEFAULT_MAX_CHUNK_SIZE = 12 * 1024 * 1024;
 53     private static final SafePath DEFAULT_DUMP_PATH = null;
 54     private static final boolean DEFAULT_PRESERVE_REPOSITORY = false;
 55 
 56     private static long memorySize;
 57     private static long globalBufferSize;
 58     private static long globalBufferCount;
 59     private static long threadBufferSize;
 60     private static int stackDepth;
 61     private static long maxChunkSize;
 62     private static boolean preserveRepository;
 63 
 64     static {
 65         final long pageSize = Unsafe.getUnsafe().pageSize();
 66         DEFAULT_THREAD_BUFFER_SIZE = pageSize > 8 * 1024 ? pageSize : 8 * 1024;
 67         reset();
 68     }
 69 
 70     public static synchronized void setMaxChunkSize(long max) {
 71         if (max < MIN_MAX_CHUNKSIZE) {
 72             throw new IllegalArgumentException("Max chunk size must be at least " + MIN_MAX_CHUNKSIZE);
 73         }
 74         JVM.setFileNotification(max);
 75         maxChunkSize = max;
 76     }
 77 
 78     public static synchronized long getMaxChunkSize() {
 79         return maxChunkSize;
 80     }
 81 
 82     public static synchronized void setMemorySize(long memSize) {
 83         JVM.setMemorySize(memSize);
 84         memorySize = memSize;
 85     }
 86 
 87     public static synchronized long getMemorySize() {
 88         return memorySize;
 89     }
 90 
 91     public static synchronized void setThreadBufferSize(long threadBufSize) {
 92         JVM.setThreadBufferSize(threadBufSize);
 93         threadBufferSize = threadBufSize;
 94     }
 95 
 96     public static synchronized long getThreadBufferSize() {
 97         return threadBufferSize;
 98     }
 99 
100     public static synchronized long getGlobalBufferSize() {
101         return globalBufferSize;
102     }
103 
104     public static synchronized void setGlobalBufferCount(long globalBufCount) {
105         JVM.setGlobalBufferCount(globalBufCount);
106         globalBufferCount = globalBufCount;
107     }
108 
109     public static synchronized long getGlobalBufferCount() {
110         return globalBufferCount;
111     }
112 
113     public static synchronized void setGlobalBufferSize(long globalBufsize) {
114         JVM.setGlobalBufferSize(globalBufsize);
115         globalBufferSize = globalBufsize;
116     }
117 
118     public static synchronized void setDumpPath(SafePath path) throws IOException {
119         if (path != null) {
120             if (SecuritySupport.isWritable(path)) {
121                 path = SecuritySupport.toRealPath(path, NOFOLLOW_LINKS);
122             } else {
123                 throw new IOException("Cannot write JFR emergency dump to " + path.toString());
124             }
125         }
126         JVM.setDumpPath(path == null ? null : path.toString());
127     }
128 
129     public static synchronized SafePath getDumpPath() {
130         return new SafePath(JVM.getDumpPath());
131     }
132 
133     public static synchronized void setStackDepth(Integer stackTraceDepth) {
134         JVM.setStackDepth(stackTraceDepth);
135         stackDepth = stackTraceDepth;
136     }
137 
138     public static synchronized int getStackDepth() {
139         return stackDepth;
140     }
141 
142     public static synchronized void setPreserveRepository(boolean preserve) {
143         preserveRepository = preserve;
144     }
145 
146     public static synchronized boolean getPreserveRepository() {
147         return preserveRepository;
148     }
149 
150     private static synchronized void reset() {
151         setMaxChunkSize(DEFAULT_MAX_CHUNK_SIZE);
152         setMemorySize(DEFAULT_MEMORY_SIZE);
153         setGlobalBufferSize(DEFAULT_GLOBAL_BUFFER_SIZE);
154         setGlobalBufferCount(DEFAULT_GLOBAL_BUFFER_COUNT);
155         try {
156             setDumpPath(DEFAULT_DUMP_PATH);
157         } catch (IOException e) {
158             // Ignore (depends on default value in JVM: it would be NULL)
159         }
160         setStackDepth(DEFAULT_STACK_DEPTH);
161         setThreadBufferSize(DEFAULT_THREAD_BUFFER_SIZE);
162         setPreserveRepository(DEFAULT_PRESERVE_REPOSITORY);
163     }
164 
165     static synchronized long getWaitInterval() {
166         return WAIT_INTERVAL;
167     }
168 
169     static void ensureInitialized() {
170         // trigger clinit which will setup JVM defaults.
171     }
172 
173 
174 }
