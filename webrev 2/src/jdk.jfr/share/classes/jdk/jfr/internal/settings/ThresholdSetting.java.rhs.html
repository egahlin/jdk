26 package jdk.jfr.internal.settings;
27 
28 import static jdk.jfr.internal.util.ValueParser.MISSING;
29 
30 import java.util.Objects;
31 import java.util.Set;
32 
33 import jdk.jfr.Description;
34 import jdk.jfr.Label;
35 import jdk.jfr.MetadataDefinition;
36 import jdk.jfr.Name;
37 import jdk.jfr.SettingControl;
38 import jdk.jfr.Timespan;
39 import jdk.jfr.internal.PlatformEventType;
40 import jdk.jfr.internal.Type;
41 import jdk.jfr.internal.util.ValueParser;
42 
43 @MetadataDefinition
44 @Label("Threshold")
45 @Name(Type.SETTINGS_PREFIX + "Threshold")
46 @Description("Record event with duration above or equal to threshold")
47 @Timespan
48 public final class ThresholdSetting extends SettingControl {
49     public static final String DEFAULT_VALUE = "0 ns";
50     private static final long typeId = Type.getTypeId(ThresholdSetting.class);
51     private String value = DEFAULT_VALUE;
52     private final PlatformEventType eventType;
53 
54     public ThresholdSetting(PlatformEventType eventType) {
55        this.eventType = Objects.requireNonNull(eventType);
56     }
57 
58     @Override
59     public String combine(Set<String> values) {
60         Long min = null;
61         String text = null;
62         for (String value : values) {
63             long nanos = ValueParser.parseTimespanWithInfinity(value, MISSING);
64             if (nanos != MISSING) {
65                 if (min == null || nanos < min) {
66                     text = value;
67                     min = nanos;
68                 }
69             }
70         }
71         return Objects.requireNonNullElse(text, DEFAULT_VALUE);
72     }
73 
74     @Override
75     public void setValue(String value) {
76         long nanos = ValueParser.parseTimespanWithInfinity(value, MISSING);
77         if (nanos != MISSING) {
78             eventType.setThreshold(nanos);
79             this.value = value;
80         }
81     }
82 
83     @Override
84     public String getValue() {
85         return value;
86     }
87 
88     public static boolean isType(long typeId) {
89         return ThresholdSetting.typeId == typeId;
90     }
91 }
