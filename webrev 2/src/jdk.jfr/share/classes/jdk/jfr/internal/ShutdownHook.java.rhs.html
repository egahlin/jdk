26 package jdk.jfr.internal;
27 





28 import jdk.jfr.RecordingState;
29 
30 /**
31  * Class responsible for dumping recordings on exit
32  *
33  */
34 final class ShutdownHook extends Thread {
35     private final PlatformRecorder recorder;
36     Object tlabDummyObject;
37 
38     ShutdownHook(PlatformRecorder recorder) {
39         super("JFR Shutdown Hook");
40         this.recorder = recorder;
41     }
42 
43     @Override
44     public void run() {
45         // this allocation is done in order to fetch a new TLAB before
46         // starting any "real" operations. In low memory situations,
47         // we would like to take an OOM as early as possible.
48         tlabDummyObject = new Object();
49         recorder.setInShutDown();
50         for (PlatformRecording recording : recorder.getRecordings()) {
51             if (recording.getDumpOnExit() && recording.getState() == RecordingState.RUNNING) {
52                 dump(recording);
53             }
54         }
55         recorder.destroy();
56     }
57 
58     private void dump(PlatformRecording recording) {
59         try {
60             WriteablePath dest = recording.getDestination();
61             if (dest == null) {
62                 dest = recording.makeDumpPath();
63                 recording.setDestination(dest);
64             }
65             if (dest != null) {
66                 recording.stop("Dump on exit");
67             }
68         } catch (Exception e) {
69             if (Logger.shouldLog(LogTag.JFR, LogLevel.DEBUG)) {
70                 Logger.log(LogTag.JFR, LogLevel.DEBUG, "Could not dump recording " + recording.getName() + " on exit.");
71             }
72         }
73     }
74 
75     static final class ExceptionHandler implements Thread.UncaughtExceptionHandler {
76         @Override
77         public void uncaughtException(Thread t, Throwable e) {
78             JVM.uncaughtException(t, e);
79         }
80     }
81 }
