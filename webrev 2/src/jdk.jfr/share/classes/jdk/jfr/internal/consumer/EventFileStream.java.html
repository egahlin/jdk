 26 package jdk.jfr.internal.consumer;
 27 
 28 import java.io.IOException;
 29 import java.nio.file.Path;
 30 import java.util.Arrays;
 31 import java.util.Collections;
 32 import java.util.Comparator;
 33 import jdk.jfr.consumer.RecordedEvent;
 34 
 35 /**
 36  * Implementation of an event stream that operates against a recording file.
 37  *
 38  */
 39 public final class EventFileStream extends AbstractEventStream {
 40     private static final Comparator<? super RecordedEvent> EVENT_COMPARATOR = JdkJfrConsumer.instance().eventComparator();
 41 
 42     private final RecordingInput input;
 43 
 44     private ChunkParser currentParser;
 45     private RecordedEvent[] cacheSorted;
 46 
 47     public EventFileStream(Path file) throws IOException {
 48         super(Collections.emptyList());
 49         this.input = new RecordingInput(file.toFile());
 50         this.input.setStreamed();
 51     }
 52 
 53     @Override
 54     public void start() {
 55         start(0);
 56     }
 57 
 58     @Override
 59     public void startAsync() {
 60         startAsync(0);
 61     }
 62 
 63     @Override
 64     public void close() {
 65         closeParser();
 66         dispatcher().runCloseActions();
 67         try {
 68             input.close();
 69         } catch (IOException e) {
 70             // ignore
 71         }
 72     }
 73 
 74     @Override
 75     protected boolean isRecordingStream() {
 76         return false;
 77     }
 78 
 79     @Override
 80     protected void process() throws IOException {
 81         Dispatcher disp = dispatcher();
 82         long start = 0;
 83         long end = Long.MAX_VALUE;
 84         if (disp.startTime != null) {
 85             start = disp.startNanos;
 86         }
 87         if (disp.endTime != null) {
 88             end = disp.endNanos;
 89         }
 90 
 91         currentParser = new ChunkParser(input, disp.parserConfiguration, parserState());
 92         while (!isClosed()) {
 93             onMetadata(currentParser);
 94             if (currentParser.getStartNanos() > end) {
 95                 close();
 96                 return;
 97             }
 98             disp = dispatcher();
 99             var ranged  = disp.parserConfiguration.withRange(start, end);
100             currentParser.updateConfiguration(ranged, true);
101             if (disp.parserConfiguration.ordered()) {
102                 processOrdered(disp);
103             } else {
104                 processUnordered(disp);
105             }
106             currentParser.resetCache();
107             if (isClosed() || currentParser.isLastChunk()) {
108                 return;
109             }
110             currentParser = currentParser.nextChunkParser();
111         }
112     }
113 
114     private void processOrdered(Dispatcher c) throws IOException {
115         if (cacheSorted == null) {
116             cacheSorted = new RecordedEvent[10_000];
117         }
118         RecordedEvent event;
119         int index = 0;
120         while (!currentParser.isChunkFinished()) {
121             while ((event = currentParser.readStreamingEvent()) != null) {
122                 if (index == cacheSorted.length) {
123                     RecordedEvent[] tmp = cacheSorted;
124                     cacheSorted = new RecordedEvent[2 * tmp.length];
125                     System.arraycopy(tmp, 0, cacheSorted, 0, tmp.length);
126                 }
127                 cacheSorted[index++] = event;
128             }
129             dispatchOrdered(c, index);
130             index = 0;
131         }
132     }
133 
134     private void dispatchOrdered(Dispatcher c, int index) {
135         onMetadata(currentParser);
136         Arrays.sort(cacheSorted, 0, index, EVENT_COMPARATOR);
137         for (int i = 0; i < index; i++) {
138             c.dispatch(cacheSorted[i]);
139         }
140         onFlush();
141     }
142 
143     private void processUnordered(Dispatcher c) throws IOException {
144         onMetadata(currentParser);
145         while (!isClosed()) {
146             RecordedEvent event = currentParser.readStreamingEvent();
147             if (event == null) {
148                 onFlush();
149                 if (currentParser.isChunkFinished()) {
150                     return;
151                 }
152                 continue;
153             }
154             onMetadata(currentParser);
155             c.dispatch(event);
156         }
157     }
158 }
