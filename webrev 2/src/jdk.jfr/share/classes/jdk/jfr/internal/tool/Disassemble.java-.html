 49 final class Disassemble extends Command {
 50 
 51     @Override
 52     public String getName() {
 53         return "disassemble";
 54     }
 55 
 56     @Override
 57     public List<String> getOptionSyntax() {
 58         List<String> list = new ArrayList<>();
 59         list.add("[--output <directory>]");
 60         list.add("[--max-chunks <chunks>]");
 61         list.add("[--max-size <size>]");
 62         list.add("<file>");
 63         return list;
 64     }
 65 
 66     @Override
 67     public void displayOptionUsage(PrintStream stream) {
 68         stream.println(" --output <directory>    The location to write the disassembled file,");
 69         stream.println("                         by default the current directory");
 70         stream.println("");
 71         stream.println(" --max-chunks <chunks>   Maximum number of chunks per disassembled file,");
 72         stream.println("                         by default 5. The chunk size varies, but is ");
 73         stream.println("                         typically around 15 MB.");
 74         stream.println("");
 75         stream.println(" --max-size <size>       Maximum number of bytes per file.");
 76         stream.println("");
 77         stream.println("  <file>                 Location of the recording file (.jfr)");
 78     }
 79 
 80     @Override
 81     public String getDescription() {
 82         return "Disassemble a recording file into smaller files/chunks";
 83     }
 84 
 85     @Override
 86     public void execute(Deque<String> options) throws UserSyntaxException, UserDataException {
 87         if (options.isEmpty()) {
 88             throw new UserSyntaxException("missing file");
 89         }
 90         Path file = getJFRInputFile(options);
 91         int maxChunks = Integer.MAX_VALUE;
 92         int maxsize = Integer.MAX_VALUE;
 93         String output = System.getProperty("user.dir");
 94         int optionCount = options.size();
 95         while (optionCount > 0) {
 96             if (acceptOption(options, "--output")) {
 97                 output = options.pop();
 98             }
 99             if (acceptOption(options, "--max-size")) {
100                 String value = options.pop();
101                 try {
102                     maxsize = Integer.parseInt(value);
103                     if (maxsize < 1) {
104                         throw new UserDataException("max size must be at least 1");
105                     }
106                 } catch (NumberFormatException nfe) {
107                     throw new UserDataException("not a valid value for --max-size.");
108                 }
109             }
110             if (acceptOption(options, "--max-chunks")) {
111                 String value = options.pop();
112                 try {
113                     maxChunks = Integer.parseInt(value);
114                     if (maxChunks < 1) {
115                         throw new UserDataException("max chunks must be at least 1.");
116                     }
117                 } catch (NumberFormatException nfe) {
118                     throw new UserDataException("not a valid value for --max-size.");
119                 }
120             }
121             if (optionCount == options.size()) {
122                 // No progress made
123                 throw new UserSyntaxException("unknown option " + options.peek());
124             }
125             optionCount = options.size();
126         }
127         Path outputPath = getDirectory(output);
128 
129         println();
130         println("Examining recording " + file + " ...");
131         List<Long> sizes;
132         if (maxsize != Integer.MAX_VALUE && maxChunks == Integer.MAX_VALUE) {
133             try {
134                 long fileSize = Files.size(file);
135                 if (maxsize >=fileSize) {
136                     println();
137                     println("File size (" + fileSize +") does not exceed max size (" + maxsize + ")");
138                     return;
139                 }
140             } catch (IOException e) {
141                 throw new UserDataException("unexpected i/o error when determining file size" + e.getMessage());
142             }
143         }
144         if (maxsize == Integer.MAX_VALUE && maxChunks == Integer.MAX_VALUE) {
145             maxChunks = 5;
146         }
147 
148         try {
149             sizes = findChunkSizes(file);
150         } catch (IOException e) {
151             throw new UserDataException("unexpected i/o error. " + e.getMessage());
152         }
153         if (maxsize == Integer.MAX_VALUE == sizes.size() <= maxChunks) {
154             throw new UserDataException("number of chunks in recording (" + sizes.size() + ") doesn't exceed max chunks (" + maxChunks + ")");
155         }
156         println();
157         if (sizes.size() > 0) {
158             List<Long> combinedSizes = combineChunkSizes(sizes, maxChunks, maxsize);
159             print("File consists of " + sizes.size() + " chunks. The recording will be split into ");
160             println(combinedSizes.size() + " files");
161             println();
162             splitFile(outputPath, file, combinedSizes);
163         } else {
164             throw new UserDataException("no JFR chunks found in file.");
165         }
166     }
167 
168     private List<Long> findChunkSizes(Path p) throws IOException {
169         try (RecordingInput input = new RecordingInput(p.toFile(), FileAccess.UNPRIVILEGED)) {
170             List<Long> sizes = new ArrayList<>();
171             ChunkHeader ch = new ChunkHeader(input);
172             sizes.add(ch.getSize());
173             while (!ch.isLastChunk()) {
174                 ch = ch.nextHeader();
175                 sizes.add(ch.getSize());
176             }
177             return sizes;
178         }
179     }
180 
181     private List<Long> combineChunkSizes(List<Long> sizes, int maxChunks, long maxSize) {
182         List<Long> reduced = new ArrayList<Long>();
183         int chunks = 1;
184         long fileSize = sizes.getFirst();
185         for (int i = 1; i < sizes.size(); i++) {
186             long size = sizes.get(i);
187             if (fileSize + size > maxSize || chunks == maxChunks) {
188                 reduced.add(fileSize);
189                 chunks = 1;
190                 fileSize = size;
191                 continue;
192             }
193             fileSize += size;
194             chunks++;
195         }
196         if (fileSize != 0) {
197             reduced.add(fileSize);
198         }
199         return reduced;
200     }
201 
202     private void splitFile(Path directory, Path file, List<Long> splitPositions) throws UserDataException {
203         int padAmountZeros = String.valueOf(splitPositions.size() - 1).length();
204         String fileName = file.getFileName().toString();
205         String fileFormatter = fileName.subSequence(0, fileName.length() - 4) + "_%0" + padAmountZeros + "d.jfr";
206         for (int i = 0; i < splitPositions.size(); i++) {
207             String formattedFilename = String.format(fileFormatter, i);
208             try {
209                 Path p = directory.resolve(formattedFilename);
210                 if (Files.exists(p)) {
211                     throw new UserDataException("can't create disassembled file " + p + ", a file with that name already exist");
212                 }
213             } catch (InvalidPathException ipe) {
214                 throw new UserDataException("can't construct path with filename" + formattedFilename);
215             }
216         }
217 
218         try (DataInputStream stream = new DataInputStream(new BufferedInputStream(new FileInputStream(file.toFile())))) {
219             for (int i = 0; i < splitPositions.size(); i++) {
220                 Long l = splitPositions.get(i);
221                 byte[] bytes = readBytes(stream, l.intValue());
222                 String formattedFilename = String.format(fileFormatter, i);
223                 Path p = directory.resolve(formattedFilename);
224                 File splittedFile = p.toFile();
225                 println("Writing " + splittedFile + " ... " + bytes.length);
226                 try (var fos = new FileOutputStream(splittedFile)) {
227                     fos.write(bytes);
228                 }
229             }
230         } catch (IOException ioe) {
231             throw new UserDataException("i/o error writing file " + file);
232         }
233     }
234 
235     private byte[] readBytes(InputStream stream, int count) throws UserDataException, IOException {
236         byte[] data = new byte[count];
237         int totalRead = 0;
238         while (totalRead < data.length) {
239             int read = stream.read(data, totalRead, data.length - totalRead);
240             if (read == -1) {
241                 throw new UserDataException("unexpected end of data");
242             }
243             totalRead += read;
244         }
245         return data;
246     }
