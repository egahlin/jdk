25 package jdk.jfr.internal.management;
26 
27 import java.nio.file.Paths;
28 import java.nio.file.Files;
29 import java.nio.file.Path;
30 import java.time.LocalDateTime;
31 import java.io.IOException;
32 
33 import jdk.jfr.internal.util.ValueFormatter;
34 
35 // Allows a remote streaming client to create chunk files
36 // with same naming scheme as the JVM.
37 public final class ChunkFilename {
38    private static final int MAX_CHUNK_NAMES = 100_000;
39    private static final String FILE_EXTENSION = ".jfr";
40 
41    private final Path directory;
42 
43    private Path lastPath;
44    private int counter;
45 
46    public ChunkFilename(Path directory) {
47        this.directory = directory;
48    }
49 
50    public String next(LocalDateTime time) throws IOException {
51        String filename = ValueFormatter.formatDateTime(time);
52        Path p = directory.resolve(filename + FILE_EXTENSION);
53 
54        // If less than one file per second (typically case)
55        if (lastPath == null || !p.equals(lastPath)) {
56            if (!Files.exists(p)) {
57                counter = 1; // reset counter
58                lastPath = p;
59                return p.toString();
60            }
61        }
62 
63        // If more than one file per second
64        while (counter < MAX_CHUNK_NAMES) {
65            String extendedName = makeExtendedName(filename, counter);
66            p = directory.resolve(extendedName);
67            counter++;
68            if (!Files.exists(p)) {
69                return p.toString();
70            }
71        }
72        throw new IOException("Unable to find unused filename after " + counter + " attempts");
73    }
74 
75    private String makeExtendedName(String filename, int counter) {
76        StringBuilder sb = new StringBuilder();
77        sb.append(filename);
78        sb.append('_');
79        if (counter < 10) { // chronological sorted
80            sb.append('0');
81        }
82        sb.append(counter);
83        sb.append(FILE_EXTENSION);
84        return sb.toString();
85    }
86 }
