 39 public final class OngoingStream extends EventByteStream {
 40 
 41     private static final byte[] EMPTY_ARRAY = new byte[0];
 42     private static final int HEADER_SIZE = (int)ChunkHeader.HEADER_SIZE;
 43     private static final int HEADER_FILE_STATE_POSITION = (int)ChunkHeader.FILE_STATE_POSITION;
 44     private static final byte MODIFYING_STATE = ChunkHeader.UPDATING_CHUNK_HEADER;
 45 
 46     private final RepositoryFiles repositoryFiles;
 47     private final Recording recording;
 48     private final HiddenWait threadSleeper = new HiddenWait();
 49     private final int blockSize;
 50     private final long endTimeNanos;
 51     private final byte[] headerBytes = new byte[HEADER_SIZE];
 52 
 53     private RecordingInput input;
 54     private ChunkHeader header;
 55     private long position;
 56     private long startTimeNanos;
 57     private Path path;
 58     private boolean first = true;
 59 
 60     public OngoingStream(Recording recording, int blockSize, long startTimeNanos, long endTimeNanos) {
 61         super();
 62         this.recording = recording;
 63         this.blockSize = blockSize;
 64         this.startTimeNanos = startTimeNanos;
 65         this.endTimeNanos = endTimeNanos;
 66         this.repositoryFiles = new RepositoryFiles(SecuritySupport.PRIVILEGED, null, false);
 67     }
 68 
 69     @Override
 70     public synchronized byte[] read() throws IOException {
 71         try {
 72             return readBytes();
 73         } catch (IOException ioe) {
 74             if (recording.getState() == RecordingState.CLOSED) {
 75                 // Recording closed, return null;
 76                 return null;
 77             }
 78             // Something unexpected has happened.
 79             throw ioe;
 80         }
 81     }
 82 
 83     private byte[] readBytes() throws IOException {
 84         touch();
 85         if (recording.getState() == RecordingState.NEW) {
 86            return EMPTY_ARRAY;

189                     // 48-55: chunk start ticks
190                     // 56-63: ticks per second
191                     // 64: file state
192                     buffer.put(64, (byte) 1);
193                     // 65-67: extension bit
194                     int left = bytes.length - HEADER_SIZE;
195                     input.readFully(bytes, HEADER_SIZE, left);
196                     position += bytes.length;
197                     return bytes;
198                 }
199             }
200             if (!threadSleeper.takeNap(10)) {
201                 throw new IOException("Read operation interrupted");
202             }
203         }
204         return EMPTY_ARRAY;
205     }
206 
207     private boolean ensureInput() throws IOException {
208         if (input == null) {
209             if (SecuritySupport.getFileSize(new SafePath(path)) < HEADER_SIZE) {
210                 return false;
211             }
212             input = new RecordingInput(path.toFile(), SecuritySupport.PRIVILEGED);
213             input.setStreamed();
214             header = new ChunkHeader(input);
215         }
216         return true;
217     }
218 
219     private boolean ensurePath() {
220         if (path == null) {
221             path = repositoryFiles.nextPath(startTimeNanos, false);
222         }
223         return path != null;
224     }
225 
226     @Override
227     public synchronized void close() throws IOException {
228         closeInput();
229         // Close recording if stream times out.
230         if (recording.getName().startsWith(EventByteStream.NAME)) {
231             recording.close();
232         }
