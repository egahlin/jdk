25 package jdk.jfr.internal.periodic;
26 
27 /**
28  * Lookup key that can be used in a {@code Map} in
29  * case hashCode and equals are incorrectly overridden.



30  * <p>
31  * {@code IdentityHashMap} can't be used as it will not preserve order.
32  */
33 final class LookupKey {
34     private final Object object;
35 
36     public LookupKey(Object object) {
37         this.object = object;
38     }
39 
40     public int hashCode() {
41         return System.identityHashCode(object);
42     }
43 
44     public boolean equals(Object that) {
45         if (that instanceof LookupKey lookupKey) {
46             return lookupKey.object == object;
47         }
48         return false;
49     }
50 }
