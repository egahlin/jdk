 26 package jdk.jfr.internal.settings;
 27 
 28 import java.util.Objects;
 29 import java.util.Set;
 30 
 31 import jdk.jfr.Description;
 32 import jdk.jfr.Label;
 33 import jdk.jfr.MetadataDefinition;
 34 import jdk.jfr.Name;
 35 import jdk.jfr.SettingControl;
 36 import jdk.jfr.internal.PlatformEventType;
 37 import jdk.jfr.internal.Type;
 38 import jdk.jfr.internal.util.ValueParser;
 39 import static jdk.jfr.internal.util.ValueParser.MISSING;
 40 
 41 @MetadataDefinition
 42 @Label("Period")
 43 @Description("Record event at interval")
 44 @Name(Type.SETTINGS_PREFIX + "Period")
 45 public final class PeriodSetting extends SettingControl {
 46     private static final long typeId = Type.getTypeId(PeriodSetting.class);
 47 
 48     public static final String EVERY_CHUNK = "everyChunk";
 49     public static final String BEGIN_CHUNK = "beginChunk";
 50     public static final String END_CHUNK = "endChunk";
 51     public static final String DEFAULT_VALUE = EVERY_CHUNK;
 52     public static final String NAME = "period";
 53     private final PlatformEventType eventType;
 54     private String value = EVERY_CHUNK;
 55 
 56     public PeriodSetting(PlatformEventType eventType) {
 57         this.eventType = Objects.requireNonNull(eventType);
 58     }
 59 
 60     @Override
 61     public String combine(Set<String> values) {
 62         boolean beginChunk = false;
 63         boolean endChunk = false;
 64         Long min = null;
 65         String text = null;
 66         for (String value : values) {
 67             switch (value) {
 68             case EVERY_CHUNK:
 69                 beginChunk = true;
 70                 endChunk = true;
 71                 break;
 72             case BEGIN_CHUNK:
 73                 beginChunk = true;
 74                 break;
 75             case END_CHUNK:
 76                 endChunk = true;
 77                 break;
 78             default:
 79                 long nanos = ValueParser.parseTimespanWithInfinity(value, MISSING);
 80                 if (nanos != MISSING) {
 81                     if (min == null || nanos < min) {
 82                         text = value;
 83                         min = nanos;
 84                     }
 85                 }
 86             }
 87         }
 88         // A specified interval trumps *_CHUNK
 89         if (min != null) {
 90             return text;
 91         }
 92         if (beginChunk && !endChunk) {
 93             return BEGIN_CHUNK;
 94         }
 95         if (!beginChunk && endChunk) {
 96             return END_CHUNK;
 97         }
 98         return DEFAULT_VALUE; // "everyChunk" is default
 99     }
100 
101     @Override
102     public void setValue(String value) {
103         switch (value) {
104         case EVERY_CHUNK:
105             eventType.setPeriod(0, true, true);
106             break;
107         case BEGIN_CHUNK:
108             eventType.setPeriod(0, true, false);
109             break;
110         case END_CHUNK:
111             eventType.setPeriod(0, false, true);
112             break;
113         default:
114             long nanos = ValueParser.parseTimespanWithInfinity(value, MISSING);
115             if (nanos == MISSING) {
116                 return;
117             }
118             if (nanos == 0 || nanos == Long.MAX_VALUE) {
119                 eventType.setPeriod(nanos, false, false);
120             } else {
121                 eventType.setPeriod(Math.max(1, nanos / 1_000_000), false, false);
122             }
123         }
124         this.value = value;
125     }
126 
127     @Override
128     public String getValue() {
129         return value;
130     }
131 
132     public static boolean isType(long typeId) {
133         return PeriodSetting.typeId == typeId;
134     }
135 }
