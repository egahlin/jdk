26 package jdk.jfr.internal.consumer;
27 
28 import java.io.File;
29 import java.io.IOException;
30 import java.io.RandomAccessFile;
31 import java.nio.file.DirectoryStream;
32 import java.nio.file.Files;
33 import java.nio.file.Path;
34 import java.nio.file.attribute.FileTime;
35 
36 // Protected by modular boundaries.
37 public abstract class FileAccess {
38     public static final FileAccess UNPRIVILEGED = new UnPrivileged();
39 
40     public abstract RandomAccessFile openRAF(File f, String mode) throws IOException;
41 
42     public abstract DirectoryStream<Path> newDirectoryStream(Path repository) throws IOException;
43 
44     public abstract String getAbsolutePath(File f) throws IOException;
45 
46     public abstract long length(File f) throws IOException;
47 
48     public abstract long fileSize(Path p) throws IOException;
49 
50     public abstract boolean exists(Path s) throws IOException;
51 
52     public abstract boolean isDirectory(Path p);
53 
54     public abstract FileTime getLastModified(Path p) throws IOException;
55 
56     private static class UnPrivileged extends FileAccess {
57         @Override
58         public RandomAccessFile openRAF(File f, String mode) throws IOException {
59             return new RandomAccessFile(f, mode);
60         }
61 
62         @Override
63         public DirectoryStream<Path> newDirectoryStream(Path dir) throws IOException {
64             return Files.newDirectoryStream(dir);
65         }
66 
67         @Override
68         public String getAbsolutePath(File f) throws IOException {
69             return f.getAbsolutePath();
70         }
71 
72         @Override
73         public long length(File f) throws IOException {
74             return f.length();
75         }
76 
77         @Override
78         public long fileSize(Path p) throws IOException {
79             return Files.size(p);
80         }
81 
82         @Override
83         public boolean exists(Path p) {
84             return Files.exists(p);
85         }
86 
87         @Override
88         public boolean isDirectory(Path p) {
89             return Files.isDirectory(p);
90         }
91 
92         @Override
93         public FileTime getLastModified(Path p) throws IOException {
94             return Files.getLastModifiedTime(p);
95         }
96     }
97 }
