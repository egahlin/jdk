 26 package jdk.jfr.internal;
 27 
 28 import java.io.IOException;
 29 import java.io.RandomAccessFile;
 30 import java.nio.channels.FileChannel;
 31 import java.nio.channels.ReadableByteChannel;
 32 import java.nio.file.Files;
 33 import java.nio.file.Path;
 34 import java.nio.file.StandardOpenOption;
 35 import java.time.Instant;
 36 import java.util.Comparator;
 37 


 38 public final class RepositoryChunk {
 39 
 40     static final Comparator<RepositoryChunk> END_TIME_COMPARATOR = new Comparator<RepositoryChunk>() {
 41         @Override
 42         public int compare(RepositoryChunk c1, RepositoryChunk c2) {
 43             return c1.endTime.compareTo(c2.endTime);
 44         }
 45     };
 46 
 47     private final Path chunkFile;
 48     private final RandomAccessFile unFinishedRAF;
 49 
 50     private Instant endTime = null; // unfinished
 51     private Instant startTime;
 52     private int refCount = 1;
 53     private long size;
 54 
 55     RepositoryChunk(Path path) throws Exception {
 56         this.chunkFile = path;
 57         this.unFinishedRAF = new RandomAccessFile(path.toFile(), "rw");
 58     }
 59 
 60     boolean finish(Instant endTime) {
 61         try {
 62             unFinishedRAF.close();
 63             size = Files.size(chunkFile);
 64             this.endTime = endTime;
 65             if (Logger.shouldLog(LogTag.JFR_SYSTEM, LogLevel.DEBUG)) {
 66                 Logger.log(LogTag.JFR_SYSTEM, LogLevel.DEBUG, "Chunk finished: " + chunkFile);
 67             }
 68             return true;
 69         } catch (IOException e) {
 70             final String reason;
 71             if (isMissingFile()) {
 72                 reason = "Chunkfile \""+ getFile() + "\" is missing. " +
 73                          "Data loss might occur from " + getStartTime() + " to " + endTime;
 74             } else {
 75                 reason = e.getClass().getName();
 76             }
 77             Logger.log(LogTag.JFR, LogLevel.ERROR, "Could not finish chunk. " + reason);
 78             return false;
 79         }
 80     }
 81 
 82     public Instant getStartTime() {
 83         return startTime;
 84     }
 85 
 86     public void setStartTime(Instant timestamp) {
 87         this.startTime = timestamp;
 88     }
 89 
 90     public Instant getEndTime() {
 91         return endTime;
 92     }
 93 
 94     private void delete(Path f) {
 95         try {
 96             Files.delete(f);
 97             if (Logger.shouldLog(LogTag.JFR, LogLevel.DEBUG)) {
 98                 Logger.log(LogTag.JFR, LogLevel.DEBUG, "Repository chunk " + f + " deleted");
 99             }
100         } catch (IOException e) {
101             // Probably happens because file is being streamed
102             // on Windows where files in use can't be removed.
103             if (Logger.shouldLog(LogTag.JFR, LogLevel.DEBUG)) {
104                 Logger.log(LogTag.JFR, LogLevel.DEBUG, "Repository chunk " + f + " could not be deleted: " + e.getMessage());
105             }
106             if (f != null) {
107                 FilePurger.add(f);
108             }
109         }
110     }
111 
112     private void destroy() {
113         try {
114             unFinishedRAF.close();
115         } catch (IOException e) {
116             if (Logger.shouldLog(LogTag.JFR, LogLevel.ERROR)) {
117                 Logger.log(LogTag.JFR, LogLevel.ERROR, "Could not close random access file: " + chunkFile.toString() + ". File will not be deleted due to: " + e.getMessage());
118             }
119         } finally {
120             delete(chunkFile);
121         }
122     }
123 
124     public synchronized void use() {
125         ++refCount;
126         if (Logger.shouldLog(LogTag.JFR_SYSTEM, LogLevel.DEBUG)) {
127             Logger.log(LogTag.JFR_SYSTEM, LogLevel.DEBUG, "Use chunk " + toString() + " ref count now " + refCount);
128         }
129     }
130 
131     public synchronized void release() {
132         --refCount;
133         if (Logger.shouldLog(LogTag.JFR_SYSTEM, LogLevel.DEBUG)) {
134             Logger.log(LogTag.JFR_SYSTEM, LogLevel.DEBUG, "Release chunk " + toString() + " ref count now " + refCount);
135         }
136         if (refCount == 0) {
137             destroy();
138         }
139     }
140 
141     public long getSize() {
142         return size;
143     }
144 
145     public boolean isFinished() {
146         return endTime != null;
147     }
148 
149     @Override
150     public String toString() {
151         return chunkFile.toString();
152     }
153 
154     ReadableByteChannel newChannel() throws IOException {
155         if (!isFinished()) {
156             throw new IOException("Chunk not finished");
157         }
158         return FileChannel.open(chunkFile, StandardOpenOption.READ);
159     }
160 
161     public boolean inInterval(Instant startTime, Instant endTime) {
162         if (startTime != null && getEndTime().isBefore(startTime)) {
163             return false;
164         }
165         if (endTime != null && getStartTime().isAfter(endTime)) {
166             return false;
167         }
168         return true;
169     }
170 
171     public Path getFile() {
172         return chunkFile;
173     }
174 
175     public long getCurrentFileSize() {
176         try {
177             return Files.size(chunkFile);
178         } catch (IOException e) {
179             return 0L;
180         }
181     }
182 
183     boolean isMissingFile() {
184         return !Files.exists(chunkFile);




185     }
186 }
