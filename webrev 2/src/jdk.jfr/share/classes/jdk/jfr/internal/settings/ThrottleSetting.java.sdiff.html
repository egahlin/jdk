 46 @MetadataDefinition
 47 @Label("Throttle")
 48 @Description("Throttles the emission rate for an event")
 49 @Name(Type.SETTINGS_PREFIX + "Throttle")
 50 public final class ThrottleSetting extends JDKSettingControl {
 51     public static final String DEFAULT_VALUE = Throttle.DEFAULT;
 52     private final PlatformEventType eventType;
 53     private String value = DEFAULT_VALUE;
 54 
 55     public ThrottleSetting(PlatformEventType eventType) {
 56        this.eventType = Objects.requireNonNull(eventType);
 57     }
 58 
 59     @Override
 60     public String combine(Set<String> values) {
 61         Rate max = null;
 62         String text = null;
 63         for (String value : values) {
 64             Rate rate = Rate.of(value);
 65             if (rate != null) {
 66                 if (max == null || rate.isHigher(max)) {
 67                     text = value;
 68                     max = rate;
 69                 }
 70             }
