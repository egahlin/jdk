39 @MetadataDefinition

42 public final class LevelSetting extends JDKSettingControl {
43     private final PlatformEventType eventType;
44     private final List<String> levels;
45     private String value;
46 
47     public LevelSetting(PlatformEventType eventType, String[] levels) {
48         this.eventType = Objects.requireNonNull(eventType);
49         this.levels = Arrays.asList(Objects.requireNonNull(levels));
50         this.value = levels[0];
51     }
52 
53     @Override
54     public String combine(Set<String> values) {
55         int maxIndex = 0; // index 0 contains the default value
56         for (String value : values) {
57             maxIndex = Math.max(maxIndex, levels.indexOf(value));
58         }
59         return levels.get(maxIndex);
60     }
