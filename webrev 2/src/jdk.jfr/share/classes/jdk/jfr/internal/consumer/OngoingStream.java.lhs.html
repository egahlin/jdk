 25 package jdk.jfr.internal.consumer;
 26 
 27 import java.io.IOException;
 28 import java.nio.ByteBuffer;

 29 import java.nio.file.Path;
 30 
 31 import jdk.jfr.Recording;
 32 import jdk.jfr.RecordingState;
 33 import jdk.jfr.internal.SecuritySupport;
 34 import jdk.jfr.internal.SecuritySupport.SafePath;
 35 import jdk.jfr.internal.management.EventByteStream;
 36 import jdk.jfr.internal.management.HiddenWait;
 37 import jdk.jfr.internal.management.ManagementSupport;
 38 
 39 public final class OngoingStream extends EventByteStream {
 40 
 41     private static final byte[] EMPTY_ARRAY = new byte[0];
 42     private static final int HEADER_SIZE = (int)ChunkHeader.HEADER_SIZE;
 43     private static final int HEADER_FILE_STATE_POSITION = (int)ChunkHeader.FILE_STATE_POSITION;
 44     private static final byte MODIFYING_STATE = ChunkHeader.UPDATING_CHUNK_HEADER;
 45 
 46     private final RepositoryFiles repositoryFiles;
 47     private final Recording recording;
 48     private final HiddenWait threadSleeper = new HiddenWait();
 49     private final int blockSize;
 50     private final long endTimeNanos;
 51     private final byte[] headerBytes = new byte[HEADER_SIZE];
 52 
 53     private RecordingInput input;
 54     private ChunkHeader header;
 55     private long position;
 56     private long startTimeNanos;
 57     private Path path;
 58     private boolean first = true;
 59 
 60     public OngoingStream(Recording recording, int blockSize, long startTimeNanos, long endTimeNanos) {
 61         super();
 62         this.recording = recording;
 63         this.blockSize = blockSize;
 64         this.startTimeNanos = startTimeNanos;
 65         this.endTimeNanos = endTimeNanos;
 66         this.repositoryFiles = new RepositoryFiles(SecuritySupport.PRIVILEGED, null, false);
 67     }
 68 
 69     @Override
 70     public synchronized byte[] read() throws IOException {
 71         try {
 72             return readBytes();
 73         } catch (IOException ioe) {
 74             if (recording.getState() == RecordingState.CLOSED) {
 75                 // Recording closed, return null;
 76                 return null;
 77             }
 78             // Something unexpected has happened.
 79             throw ioe;
 80         }
 81     }
 82 
 83     private byte[] readBytes() throws IOException {
 84         touch();
 85         if (recording.getState() == RecordingState.NEW) {
 86            return EMPTY_ARRAY;
 87         }
 88 
 89         if (recording.getState() == RecordingState.DELAYED) {
 90             return EMPTY_ARRAY;
 91         }
 92 
 93         if (first) {
 94             // In case stream starts before recording
 95             long s = ManagementSupport.getStartTimeNanos(recording);
 96             startTimeNanos = Math.max(s, startTimeNanos);
 97             first = false;
 98         }
 99 
100         while (true) {
101             if (startTimeNanos > endTimeNanos) {
102                 return null;
103             }
104             if (isRecordingClosed()) {
105                 closeInput();
106                 return null;
107             }
108             if (!ensurePath()) {
109                 return EMPTY_ARRAY;
110             }
111             if (!ensureInput()) {
112                 return EMPTY_ARRAY;
113             }
114             if (position < header.getChunkSize()) {
115                 long size = Math.min(header.getChunkSize() - position, blockSize);
116                 return readBytes((int) size);
117             }
118             if (header.isFinished()) {
119                 if (header.getDurationNanos() < 1) {
120                     throw new IOException("No progress");
121                 }
122                 startTimeNanos += header.getDurationNanos();
123                 ManagementSupport.removePath(recording, path);
124                 closeInput();
125             } else {
126                 header.refresh();
127                 if (position >= header.getChunkSize()) {
128                     return EMPTY_ARRAY;
129                 }
130             }
131         }
132     }
133 
134     private boolean isRecordingClosed() {
135         return recording != null && recording.getState() == RecordingState.CLOSED;
136     }
137 
138     private void closeInput() {
139         if (input != null) {
140             try {
141                 input.close();
142             } catch (IOException ioe) {
143                 // ignore
144             }
145             input = null;
146             position = 0;
147             path = null;
148         }
149     }
150 
151     private byte[] readBytes(int size) throws IOException {
152         if (position == 0) {
153             return readWithHeader(size);
154         } else {
155             return readNonHeader(size);
156         }
157     }
158 
159     private byte[] readNonHeader(int size) throws IOException {
160         byte[] result = new byte[size];
161         input.readFully(result);
162         position += size;
163         return result;
164     }
165 
166     private byte[] readWithHeader(int size) throws IOException {
167         byte[] bytes = new byte[Math.max(HEADER_SIZE, size)];
168         for (int attempts = 0; attempts < 25; attempts++) {
169             // read twice and check files state to avoid simultaneous change by JVM
170             input.position(0);
171             input.readFully(bytes, 0, HEADER_SIZE);
172             input.position(0);
173             input.readFully(headerBytes);
174             if (bytes[HEADER_FILE_STATE_POSITION] != MODIFYING_STATE) {
175                 if (bytes[HEADER_FILE_STATE_POSITION] == headerBytes[HEADER_FILE_STATE_POSITION]) {
176                     ByteBuffer buffer = ByteBuffer.wrap(bytes);
177                     // 0-3: magic
178                     // 4-5: major
179                     // 6-7: minor
180                     // 8-15: chunk size
181                     buffer.putLong(8, HEADER_SIZE);
182                     // 16-23: constant pool offset
183                     buffer.putLong(16, 0);
184                     // 24-31: metadata offset
185                     buffer.putLong(24, 0);
186                     // 32-39: chunk start nanos
187                     // 40-47 duration
188                     buffer.putLong(40, 0);
189                     // 48-55: chunk start ticks
190                     // 56-63: ticks per second
191                     // 64: file state
192                     buffer.put(64, (byte) 1);
193                     // 65-67: extension bit
194                     int left = bytes.length - HEADER_SIZE;
195                     input.readFully(bytes, HEADER_SIZE, left);
196                     position += bytes.length;
197                     return bytes;
198                 }
199             }
200             if (!threadSleeper.takeNap(10)) {
201                 throw new IOException("Read operation interrupted");
202             }
203         }
204         return EMPTY_ARRAY;
205     }
206 
207     private boolean ensureInput() throws IOException {
208         if (input == null) {
209             if (SecuritySupport.getFileSize(new SafePath(path)) < HEADER_SIZE) {
210                 return false;
211             }
212             input = new RecordingInput(path.toFile(), SecuritySupport.PRIVILEGED);
213             input.setStreamed();
214             header = new ChunkHeader(input);
215         }
216         return true;
217     }
218 
219     private boolean ensurePath() {
220         if (path == null) {
221             path = repositoryFiles.nextPath(startTimeNanos, false);
222         }
223         return path != null;
224     }
225 
226     @Override
227     public synchronized void close() throws IOException {
228         closeInput();
229         // Close recording if stream times out.
230         if (recording.getName().startsWith(EventByteStream.NAME)) {
231             recording.close();
232         }
233     }
234 }
