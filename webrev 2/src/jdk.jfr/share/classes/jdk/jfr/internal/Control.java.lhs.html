 26 package jdk.jfr.internal;
 27 
 28 import java.security.AccessControlContext;
 29 import java.security.AccessController;
 30 import java.security.PrivilegedAction;
 31 import java.util.Collections;
 32 import java.util.HashSet;
 33 import java.util.Objects;
 34 import java.util.Set;
 35 
 36 import jdk.jfr.SettingControl;
 37 import jdk.jfr.internal.settings.JDKSettingControl;
 38 import jdk.jfr.internal.settings.PeriodSetting;
 39 import jdk.jfr.internal.settings.StackTraceSetting;
 40 import jdk.jfr.internal.settings.ThresholdSetting;
 41 
 42 final class Control {
 43     @SuppressWarnings("removal")
 44     private final AccessControlContext context;
 45     private static final int CACHE_SIZE = 5;
 46     private final Set<?>[] cachedUnions = new HashSet<?>[CACHE_SIZE];
 47     private final String[] cachedValues = new String[CACHE_SIZE];
 48     private final SettingControl delegate;
 49     private String defaultValue;
 50     private String lastValue;
 51 
 52     // called by exposed subclass in external API
 53     public Control(SettingControl delegate, String defaultValue) {
 54         this.context = PrivateAccess.getInstance().getContext(delegate);
 55         this.delegate = delegate;
 56         this.defaultValue = defaultValue;
 57         if (this.context == null && !(delegate instanceof JDKSettingControl)) {
 58             throw new InternalError("Security context can only be null for trusted setting controls");
 59         }
 60     }
 61 
 62     boolean isType(Class<? extends SettingControl> clazz) {
 63         return delegate.getClass() == clazz;
 64     }
 65 
 66     final void apply(Set<String> values) {
 67         setValue(findCombine(values));
 68     }
 69 
 70     final void setDefault() {
 71         if (defaultValue == null) {
 72             defaultValue = getValue();
 73         }
 74         apply(defaultValue);
 75     }
 76 
 77     @SuppressWarnings("removal")
 78     public String getValue() {
 79         if (context == null) {
 80             // VM events requires no access control context
 81             return delegate.getValue();
 82         } else {
 83             return AccessController.doPrivileged(new PrivilegedAction<String>() {
 84                 @Override
 85                 public String run() {
 86                     try {
 87                         return delegate.getValue();
 88                     } catch (Throwable t) {
 89                         // Prevent malicious user to propagate exception callback in the wrong context
 90                         Logger.log(LogTag.JFR_SETTING, LogLevel.WARN, "Exception occurred when trying to get value for " + getClass());
 91                     }
 92                     return defaultValue != null ? defaultValue : ""; // Need to return something
 93                 }
 94             }, context);
 95         }
 96     }
 97 
 98     private void apply(String value) {
 99         if (lastValue != null && Objects.equals(value, lastValue)) {
100             return;
101         }
102         setValue(value);
103     }
104 
105     @SuppressWarnings("removal")
106     public void setValue(String value) {
107         if (context == null) {
108             // VM events requires no access control context
109             try {
110                 delegate.setValue(value);
111                 lastValue = delegate.getValue();
112             } catch (Throwable t) {
113                 Logger.log(LogTag.JFR_SETTING, LogLevel.WARN, "Exception occurred when setting value \"" + value + "\" for " + getClass());
114                 lastValue = null;
115             }
116         } else {
117             lastValue = AccessController.doPrivileged(new PrivilegedAction<String>() {
118                 @Override
119                 public String run() {
120                     try {
121                         delegate.setValue(value);
122                         return delegate.getValue();
123                     } catch (Throwable t) {
124                         // Prevent malicious user to propagate exception callback in the wrong context
125                         Logger.log(LogTag.JFR_SETTING, LogLevel.WARN, "Exception occurred when setting value \"" + value + "\" for " + getClass());
126                     }
127                     return null;
128                 }
129             }, context);
130         }
131     }
132 
133 
134     @SuppressWarnings("removal")
135     public String combine(Set<String> values) {
136         if (context == null) {
137             // VM events requires no access control context
138             return delegate.combine(values);
139         }
140         return AccessController.doPrivileged(new PrivilegedAction<String>() {
141             @Override
142             public String run() {
143                 try {
144                     return delegate.combine(Collections.unmodifiableSet(values));
145                 } catch (Throwable t) {
146                     // Prevent malicious user to propagate exception callback in the wrong context
147                     Logger.log(LogTag.JFR_SETTING, LogLevel.WARN, "Exception occurred when combining " + values + " for " + getClass());
148                 }
149                 return null;
150             }
151         }, context);
152     }
153 
154     private final String findCombine(Set<String> values) {
155         if (values.size() == 1) {
156             return values.iterator().next();
157         }
158         for (int i = 0; i < CACHE_SIZE; i++) {
159             if (Objects.equals(cachedUnions[i], values)) {
160                 return cachedValues[i];
161             }
162         }
163         String result = combine(values);
164         for (int i = 0; i < CACHE_SIZE - 1; i++) {
165             cachedUnions[i + 1] = cachedUnions[i];
166             cachedValues[i + 1] = cachedValues[i];
167         }
168         cachedValues[0] = result;
169         cachedUnions[0] = values;
170         return result;
171     }
172 
173     final String getDefaultValue() {
174         return defaultValue;
175     }
176 
177     final String getLastValue() {
178         return lastValue;
179     }
180 
181     final SettingControl getSettingControl() {
182         return delegate;
183     }
184 
185     boolean isVisible(boolean hasEventHook) {
186         if (isType(ThresholdSetting.class)) {
187             return !hasEventHook;
188         }
189         if (isType(PeriodSetting.class)) {
190             return hasEventHook;
191         }
192         if (isType(StackTraceSetting.class)) {
193             return !hasEventHook;
194         }
195         return true;
196     }
197 }
