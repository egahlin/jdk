 77     public static List<Annotation> getAnnotations(Class<?> clazz) {
 78         List<Annotation> annos = new ArrayList<>();
 79         for (Annotation a : clazz.getAnnotations()) {
 80             annos.addAll(getAnnotation(a));
 81         }
 82         return annos;
 83     }

 85     private static List<? extends Annotation> getAnnotation(Annotation a) {
 86         Class<?> annotated = a.annotationType();
 87         Method valueMethod = getValueMethod(annotated);
 88         if (valueMethod != null) {
 89             Class<?> returnType = valueMethod.getReturnType();
 90             if (returnType.isArray()) {
 91                 Class<?> candidate = returnType.getComponentType();
 92                 Repeatable r = candidate.getAnnotation(Repeatable.class);
 93                 if (r != null) {
 94                     Class<?> repeatClass = r.value();
 95                     if (annotated == repeatClass) {
 96                         return getAnnotationValues(a, valueMethod);
 97                     }
 98                 }
 99             }
100         }
101         return List.of(a);
102     }

121     private static List<Annotation> getAnnotationValues(Annotation a, Method valueMethod) {
122         try {
123             return Arrays.asList((Annotation[]) valueMethod.invoke(a, new Object[0]));
124         } catch (IllegalAccessException | IllegalArgumentException | InvocationTargetException e) {
125             return new ArrayList<>();
126         }
127     }

129     private static Method getValueMethod(Class<?> annotated) {
130         try {
131             return annotated.getMethod("value", new Class<?>[0]);
132         } catch (NoSuchMethodException e) {
133             return null;
134         }
135     }

142     public static Class<?> unboxType(Class<?> t) {
143         if (t == Integer.class) {
144             return int.class;
145         }
146         if (t == Long.class) {
147             return long.class;
148         }
149         if (t == Float.class) {
150             return float.class;
151         }
152         if (t == Double.class) {
153             return double.class;
154         }
155         if (t == Byte.class) {
156             return byte.class;
157         }
158         if (t == Short.class) {
159             return short.class;
160         }
161         if (t == Boolean.class) {
162             return boolean.class;
163         }
164         if (t == Character.class) {
165             return char.class;
166         }
167         return t;
168     }

170     public static Map<String, String> sanitizeNullFreeStringMap(Map<String, String> settings) {
171         HashMap<String, String> map = HashMap.newHashMap(settings.size());
172         for (Map.Entry<String, String> e : settings.entrySet()) {
173             String key = e.getKey();
174             if (key == null) {
175                 throw new NullPointerException("Null key is not allowed in map");
176             }
177             String value = e.getValue();
178             if (value == null) {
179                 throw new NullPointerException("Null value is not allowed in map");
180             }
181             map.put(key, value);
182         }
183         return map;
184     }

186     public static <T> boolean compareLists(List<T> a, List<T> b, Comparator<T> c) {
187         int size = a.size();
188         if (size != b.size()) {
189             return false;
190         }
191         for (int i = 0; i < size; i++) {
192             if (c.compare(a.get(i), b.get(i)) != 0) {
193                 return false;
194             }
195         }
196         return true;
197     }

199     public static <T> List<T> sanitizeNullFreeList(List<T> elements, Class<T> clazz) {
200         List<T> sanitized = new ArrayList<>(elements.size());
201         for (T element : elements) {
202             if (element == null) {
203                 throw new NullPointerException("Null is not an allowed element in list");
204             }
205             if (element.getClass() != clazz) {
206                 throw new ClassCastException();
207             }
208             sanitized.add(element);
209         }
210         return sanitized;
211     }

213     public static List<Field> getVisibleEventFields(Class<?> clazz) {
214         List<Field> fields = new ArrayList<>();
215         for (Class<?> c = clazz; !Utils.isEventBaseClass(c); c = c.getSuperclass()) {
216             for (Field field : c.getDeclaredFields()) {
217                 // skip private field in base classes
218                 if (c == clazz || !Modifier.isPrivate(field.getModifiers())) {
219                     fields.add(field);
220                 }
221             }
222         }
223         return fields;
224     }

249     public static Object makePrimitiveArray(String typeName, List<Object> values) {
250         Class<?> componentType = makePrimitiveType(typeName);
251         if (componentType == null) {
252             return null;
253         }
254         int length = values.size();
255         Object array =  Array.newInstance(componentType, length);
256         for (int index = 0; index < length; index++) {
257             Array.set(array, index, values.get(index));
258         }
259         return array;
260     }

262     private static Class<?> makePrimitiveType(String typeName) {
263         return switch(typeName) {
264             case "void" -> null;
265             case "java.lang.String" -> String.class;
266             default -> Class.forPrimitiveName(typeName);
267         };
268     }

283     public static Type getValidType(Class<?> type, String name) {
284         Objects.requireNonNull(type, "Null is not a valid type for value descriptor " + name);
285         if (type.isArray()) {
286             type = type.getComponentType();
287             if (type != String.class && !type.isPrimitive()) {
288                 throw new IllegalArgumentException("Only arrays of primitives and Strings are allowed");
289             }
290         }
291 
292         Type knownType = Type.getKnownType(type);
293         if (knownType == null || knownType == Type.STACK_TRACE) {
294             throw new IllegalArgumentException("Only primitive types, java.lang.Thread, java.lang.String and java.lang.Class are allowed for value descriptors. " + type.getName());
295         }
296         return knownType;
297     }

299     public static String upgradeLegacyJDKEvent(String eventName) {
300         if (eventName.length() <= LEGACY_EVENT_NAME_PREFIX.length()) {
301             return eventName;
302         }
303         if (eventName.startsWith(LEGACY_EVENT_NAME_PREFIX)) {
304             int index = eventName.lastIndexOf(".");
305             if (index == LEGACY_EVENT_NAME_PREFIX.length() - 1) {
306                 return Type.EVENT_NAME_PREFIX + eventName.substring(index + 1);
307             }
308         }
309         return eventName;
310     }

312     public static void verifyMirror(Class<? extends MirrorEvent> mirror, Class<?> real) {
313         Class<?> cMirror = Objects.requireNonNull(mirror);
314         Class<?> cReal = Objects.requireNonNull(real);
315 
316         Map<String, Field> mirrorFields = new HashMap<>();
317         while (cMirror != null) {
318             for (Field f : cMirror.getDeclaredFields()) {
319                 if (isSupportedType(f.getType())) {
320                     mirrorFields.put(f.getName(), f);
321                 }
322             }
323             cMirror = cMirror.getSuperclass();
324         }
325         while (cReal != null) {
326             for (Field realField : cReal.getDeclaredFields()) {
327                 if (isSupportedType(realField.getType()) && !realField.isSynthetic()) {
328                     String fieldName = realField.getName();
329                     Field mirrorField = mirrorFields.get(fieldName);
330                     if (mirrorField == null) {
331                         throw new InternalError("Missing mirror field for " + cReal.getName() + "#" + fieldName);
332                     }
333                     if (realField.getType() != mirrorField.getType()) {
334                         throw new InternalError("Incorrect type for mirror field " + fieldName);
335                     }
336                     if (realField.getModifiers() != mirrorField.getModifiers()) {
337                         throw new InternalError("Incorrect modifier for mirror field " + fieldName);
338                     }
339                     mirrorFields.remove(fieldName);
340                 }
341             }
342             cReal = cReal.getSuperclass();
343         }
344 
345         if (!mirrorFields.isEmpty()) {
346             throw new InternalError("Found additional fields in mirror class " + mirrorFields.keySet());
347         }
348     }

350     private static boolean isSupportedType(Class<?> type) {
351         if (Modifier.isTransient(type.getModifiers()) || Modifier.isStatic(type.getModifiers())) {
352             return false;
353         }
354         return Type.isValidJavaFieldType(type.getName());
355     }

407     public static String format(String template, Map<String, String> parameters) {
408         StringBuilder sb = new StringBuilder(3 * template.length() / 2);
409         List<String> keys = new ArrayList<>(parameters.keySet());
410         // Sort so longest keys are checked first in case keys overlap.
411         keys.sort((a, b) -> b.length() - a.length());
412         for (int i = 0; i < template.length(); i++) {
413             int index = i;
414             for (int j = 0; j < keys.size(); j++) {
415                 String key = keys.get(j);
416                 if (template.startsWith(key, i)) {
417                     sb.append(parameters.get(key));
418                     i += key.length() - 1;
419                     break;
420                 }
421             }
422             if (i == index) {
423                 sb.append(template.charAt(i));
424             }
425         }
426         return sb.toString();
427     }

454     public static Path getPathInProperty(String prop, String subPath) {
455         String path = System.getProperty(prop);
456         if (path == null) {
