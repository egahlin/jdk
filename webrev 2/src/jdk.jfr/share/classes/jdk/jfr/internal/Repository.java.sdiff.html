 26 package jdk.jfr.internal;
 27 
 28 import java.io.IOException;

 29 import java.nio.file.Path;
 30 import java.time.DateTimeException;
 31 import java.time.LocalDateTime;
 32 import java.time.ZoneOffset;
 33 import java.util.HashSet;
 34 import java.util.Set;
 35 
 36 import jdk.jfr.internal.SecuritySupport.SafePath;
 37 import jdk.jfr.internal.management.ChunkFilename;
 38 import jdk.jfr.internal.util.ValueFormatter;


 39 
 40 public final class Repository {
 41 

 42     private static final int MAX_REPO_CREATION_RETRIES = 1000;
 43     private static final Repository instance = new Repository();
 44 
 45     private static final String JFR_REPOSITORY_LOCATION_PROPERTY = "jdk.jfr.repository";
 46 
 47     private final Set<SafePath> cleanupDirectories = new HashSet<>();
 48     private SafePath baseLocation;
 49     private SafePath repository;
 50     private ChunkFilename chunkFilename;
 51 
 52     private Repository() {
 53     }
 54 
 55     public static Repository getRepository() {
 56         return instance;
 57     }
 58 
 59     public synchronized void setBasePath(SafePath baseLocation) throws IOException {
 60         if(baseLocation.equals(this.baseLocation)) {
 61             Logger.log(LogTag.JFR, LogLevel.INFO, "Same base repository path " + baseLocation.toString() + " is set");
 62             return;
 63         }
 64         // Probe to see if repository can be created, needed for fail fast
 65         // during JVM startup or JFR.configure
 66         this.repository = createRepository(baseLocation);
 67         this.chunkFilename = null;
 68         try {
 69             // Remove so we don't "leak" repositories, if JFR is never started
 70             // and shutdown hook not added.
 71             SecuritySupport.delete(repository);
 72         } catch (IOException ioe) {
 73             Logger.log(LogTag.JFR, LogLevel.INFO, "Could not delete disk repository " + repository);
 74         }
 75         this.baseLocation = baseLocation;
 76     }
 77 
 78     public synchronized void ensureRepository() throws IOException {
 79         if (baseLocation == null) {
 80             setBasePath(SecuritySupport.JAVA_IO_TMPDIR);
 81         }
 82     }
 83 
 84     synchronized RepositoryChunk newChunk() {
 85         LocalDateTime timestamp = timestamp();
 86         try {
 87             if (!SecuritySupport.existDirectory(repository)) {
 88                 this.repository = createRepository(baseLocation);
 89                 JVM.setRepositoryLocation(repository.toString());
 90                 SecuritySupport.setProperty(JFR_REPOSITORY_LOCATION_PROPERTY, repository.toString());
 91                 cleanupDirectories.add(repository);
 92                 chunkFilename = null;
 93             }
 94             if (chunkFilename == null) {
 95                 chunkFilename = ChunkFilename.newPriviliged(repository.toPath());
 96             }
 97             String filename = chunkFilename.next(timestamp);
 98             return new RepositoryChunk(new SafePath(filename));
 99         } catch (Exception e) {
100             String errorMsg = String.format("Could not create chunk in repository %s, %s: %s", repository, e.getClass(), e.getMessage());
101             Logger.log(LogTag.JFR, LogLevel.ERROR, errorMsg);
102             JVM.abort(errorMsg);
103             throw new InternalError("Could not abort after JFR disk creation error");
104         }
105     }
106 
107     private static LocalDateTime timestamp() {
108         try {
109             return LocalDateTime.now();
110         } catch (DateTimeException d) {
111             Logger.log(LogTag.JFR, LogLevel.INFO, "Could not create LocalDateTime with the default time zone. Using UTC time zone for chunk filename.");
112             return LocalDateTime.now(ZoneOffset.UTC);
113         }
114     }
115 
116     private static SafePath createRepository(SafePath basePath) throws IOException {
117         SafePath canonicalBaseRepositoryPath = createRealBasePath(basePath);
118         SafePath f = null;
119 
120         String basename = ValueFormatter.formatDateTime(timestamp()) + "_" + JVM.getPid();
121         String name = basename;
122 
123         int i = 0;
124         for (; i < MAX_REPO_CREATION_RETRIES; i++) {
125             f = new SafePath(canonicalBaseRepositoryPath.toPath().resolve(name));
126             if (tryToUseAsRepository(f)) {
127                 break;
128             }
129             name = basename + "_" + i;
130         }
131 
132         if (i == MAX_REPO_CREATION_RETRIES) {
133             throw new IOException("Unable to create JFR repository directory using base location (" + basePath + ")");
134         }
135         return SecuritySupport.toRealPath(f);
136     }
137 
138     private static SafePath createRealBasePath(SafePath safePath) throws IOException {
139         if (SecuritySupport.exists(safePath)) {
140             if (!SecuritySupport.isWritable(safePath)) {
141                 throw new IOException("JFR repository directory (" + safePath.toString() + ") exists, but isn't writable");
142             }
143             return SecuritySupport.toRealPath(safePath);
144         }
145         SafePath p = SecuritySupport.createDirectories(safePath);
146         return SecuritySupport.toRealPath(p);
147     }
148 
149     private static boolean tryToUseAsRepository(final SafePath path) {
150         Path parent = path.toPath().getParent();
151         if (parent == null) {
152             return false;
153         }
154         try {
155             try {
156                 SecuritySupport.createDirectories(path);
157             } catch (Exception e) {
158                 // file already existed or some other problem occurred
159             }
160             if (!SecuritySupport.exists(path)) {
161                 return false;
162             }
163             if (!SecuritySupport.isDirectory(path)) {
164                 return false;
165             }
166             return true;
167         } catch (IOException io) {
168             return false;
169         }

170     }
171 
172     synchronized void clear() {
173         if (Options.getPreserveRepository()) {
174             return;
175         }
176 
177         for (SafePath p : cleanupDirectories) {
178             try {
179                 SecuritySupport.clearDirectory(p);
180                 Logger.log(LogTag.JFR, LogLevel.INFO, "Removed repository " + p);
181             } catch (IOException e) {
182                 Logger.log(LogTag.JFR, LogLevel.INFO, "Repository " + p + " could not be removed at shutdown: " + e.getMessage());
183             }
184         }
185     }
186 
187     public synchronized SafePath getRepositoryPath() {
188         return repository;
189     }
190 
191     public synchronized SafePath getBaseLocation() {
192         return baseLocation;
193     }
194 
195 }

 44     private static final Path JAVA_IO_TMPDIR = Utils.getPathInProperty("java.io.tmpdir", null);
 45     private static final int MAX_REPO_CREATION_RETRIES = 1000;
 46     private static final Repository instance = new Repository();

 47     private static final String JFR_REPOSITORY_LOCATION_PROPERTY = "jdk.jfr.repository";
 48     private final Set<Path> cleanupDirectories = new HashSet<>();
 49     private Path baseLocation;
 50     private Path repository;

 51     private ChunkFilename chunkFilename;
 52 
 53     private Repository() {
 54     }
 55 
 56     public static Repository getRepository() {
 57         return instance;
 58     }

 60     public synchronized void setBasePath(Path baseLocation) throws IOException {
 61         if(baseLocation.equals(this.baseLocation)) {
 62             Logger.log(LogTag.JFR, LogLevel.INFO, "Same base repository path " + baseLocation.toString() + " is set");
 63             return;
 64         }
 65         // Probe to see if repository can be created, needed for fail fast
 66         // during JVM startup or JFR.configure
 67         this.repository = createRepository(baseLocation);
 68         this.chunkFilename = null;
 69         try {
 70             // Remove so we don't "leak" repositories, if JFR is never started
 71             // and shutdown hook not added.
 72             Files.delete(repository);
 73         } catch (IOException ioe) {
 74             Logger.log(LogTag.JFR, LogLevel.INFO, "Could not delete disk repository " + repository);
 75         }
 76         this.baseLocation = baseLocation;
 77     }
 78 
 79     public synchronized void ensureRepository() throws IOException {
 80         if (baseLocation == null) {
 81             setBasePath(JAVA_IO_TMPDIR);
 82         }
 83     }
 84 
 85     synchronized RepositoryChunk newChunk() {
 86         LocalDateTime timestamp = timestamp();
 87         try {
 88             if (!Files.exists(repository)) {
 89                 this.repository = createRepository(baseLocation);
 90                 JVM.setRepositoryLocation(repository.toString());
 91                 System.setProperty(JFR_REPOSITORY_LOCATION_PROPERTY, repository.toString());
 92                 cleanupDirectories.add(repository);
 93                 chunkFilename = null;
 94             }
 95             if (chunkFilename == null) {
 96                 chunkFilename = new ChunkFilename(repository);
 97             }
 98             String filename = chunkFilename.next(timestamp);
 99             return new RepositoryChunk(Path.of(filename));
100         } catch (Exception e) {
101             String errorMsg = String.format("Could not create chunk in repository %s, %s: %s", repository, e.getClass(), e.getMessage());
102             Logger.log(LogTag.JFR, LogLevel.ERROR, errorMsg);
103             JVM.abort(errorMsg);
104             throw new InternalError("Could not abort after JFR disk creation error");
105         }
106     }
107 
108     private static LocalDateTime timestamp() {
109         try {
110             return LocalDateTime.now();
111         } catch (DateTimeException d) {
112             Logger.log(LogTag.JFR, LogLevel.INFO, "Could not create LocalDateTime with the default time zone. Using UTC time zone for chunk filename.");
113             return LocalDateTime.now(ZoneOffset.UTC);
114         }
115     }
116 
117     private static Path createRepository(Path basePath) throws IOException {
118         Path canonicalBaseRepositoryPath = createRealBasePath(basePath);
119         Path f = null;
120 
121         String basename = ValueFormatter.formatDateTime(timestamp()) + "_" + JVM.getPid();
122         String name = basename;
123 
124         int i = 0;
125         for (; i < MAX_REPO_CREATION_RETRIES; i++) {
126             f = canonicalBaseRepositoryPath.resolve(name);
127             if (tryToUseAsRepository(f)) {
128                 break;
129             }
130             name = basename + "_" + i;
131         }
132 
133         if (i == MAX_REPO_CREATION_RETRIES) {
134             throw new IOException("Unable to create JFR repository directory using base location (" + basePath + ")");
135         }
136         return f.toRealPath();
137     }
138 
139     private static Path createRealBasePath(Path path) throws IOException {
140         if (Files.exists(path)) {
141             if (!Files.isWritable(path)) {
142                 throw new IOException("JFR repository directory (" + path.toString() + ") exists, but isn't writable");
143             }
144             return path.toRealPath();
145         }
146         return Files.createDirectories(path).toRealPath();

147     }
148 
149     private static boolean tryToUseAsRepository(Path path) {
150         Path parent = path.getParent();
151         if (parent == null) {
152             return false;
153         }
154         try {
155             Files.createDirectories(path);
156         } catch (Exception e) {
157             // file already existed or some other problem occurred
158         }
159         if (!Files.exists(path)) {
160             return false;
161         }
162         if (!Files.isDirectory(path)) {





163             return false;
164         }
165         return true;
166     }

168     synchronized void clear() {
169         if (Options.getPreserveRepository()) {
170             return;
171         }
172 
173         for (Path p : cleanupDirectories) {
174             try {
175                 DirectoryCleaner.clear(p);
176                 Logger.log(LogTag.JFR, LogLevel.INFO, "Removed repository " + p);
177             } catch (IOException e) {
178                 Logger.log(LogTag.JFR, LogLevel.INFO, "Repository " + p + " could not be removed at shutdown: " + e.getMessage());
179             }
180         }
181     }

183     public synchronized Path getRepositoryPath() {
184         return repository;
185     }

188         return baseLocation;
189     }
