 26 package jdk.jfr.internal;
 27 
 28 import java.io.IOException;
 29 import java.io.RandomAccessFile;

 30 import java.nio.channels.ReadableByteChannel;



 31 import java.time.Instant;
 32 import java.util.Comparator;
 33 
 34 import jdk.jfr.internal.SecuritySupport.SafePath;
 35 
 36 public final class RepositoryChunk {
 37 
 38     static final Comparator<RepositoryChunk> END_TIME_COMPARATOR = new Comparator<RepositoryChunk>() {
 39         @Override
 40         public int compare(RepositoryChunk c1, RepositoryChunk c2) {
 41             return c1.endTime.compareTo(c2.endTime);
 42         }
 43     };
 44 
 45     private final SafePath chunkFile;
 46     private final RandomAccessFile unFinishedRAF;
 47 
 48     private Instant endTime = null; // unfinished
 49     private Instant startTime;
 50     private int refCount = 1;
 51     private long size;
 52 
 53     RepositoryChunk(SafePath path) throws Exception {
 54         this.chunkFile = path;
 55         this.unFinishedRAF = SecuritySupport.createRandomAccessFile(chunkFile);
 56     }
 57 
 58     boolean finish(Instant endTime) {
 59         try {
 60             unFinishedRAF.close();
 61             size = SecuritySupport.getFileSize(chunkFile);
 62             this.endTime = endTime;
 63             if (Logger.shouldLog(LogTag.JFR_SYSTEM, LogLevel.DEBUG)) {
 64                 Logger.log(LogTag.JFR_SYSTEM, LogLevel.DEBUG, "Chunk finished: " + chunkFile);
 65             }
 66             return true;
 67         } catch (IOException e) {
 68             final String reason;
 69             if (isMissingFile()) {
 70                 reason = "Chunkfile \""+ getFile() + "\" is missing. " +
 71                          "Data loss might occur from " + getStartTime() + " to " + endTime;
 72             } else {
 73                 reason = e.getClass().getName();
 74             }
 75             Logger.log(LogTag.JFR, LogLevel.ERROR, "Could not finish chunk. " + reason);
 76             return false;
 77         }
 78     }
 79 
 80     public Instant getStartTime() {
 81         return startTime;
 82     }
 83 
 84     public void setStartTime(Instant timestamp) {
 85         this.startTime = timestamp;
 86     }
 87 
 88     public Instant getEndTime() {
 89         return endTime;
 90     }
 91 
 92     private void delete(SafePath f) {
 93         try {
 94             SecuritySupport.delete(f);
 95             if (Logger.shouldLog(LogTag.JFR, LogLevel.DEBUG)) {
 96                 Logger.log(LogTag.JFR, LogLevel.DEBUG, "Repository chunk " + f + " deleted");
 97             }
 98         } catch (IOException e) {
 99             // Probably happens because file is being streamed
100             // on Windows where files in use can't be removed.
101             if (Logger.shouldLog(LogTag.JFR, LogLevel.DEBUG)) {
102                 Logger.log(LogTag.JFR, LogLevel.DEBUG, "Repository chunk " + f + " could not be deleted: " + e.getMessage());
103             }
104             if (f != null) {
105                 FilePurger.add(f);
106             }
107         }
108     }
109 
110     private void destroy() {
111         try {
112             unFinishedRAF.close();
113         } catch (IOException e) {
114             if (Logger.shouldLog(LogTag.JFR, LogLevel.ERROR)) {
115                 Logger.log(LogTag.JFR, LogLevel.ERROR, "Could not close random access file: " + chunkFile.toString() + ". File will not be deleted due to: " + e.getMessage());
116             }
117         } finally {
118             delete(chunkFile);
119         }
120     }
121 
122     public synchronized void use() {
123         ++refCount;
124         if (Logger.shouldLog(LogTag.JFR_SYSTEM, LogLevel.DEBUG)) {
125             Logger.log(LogTag.JFR_SYSTEM, LogLevel.DEBUG, "Use chunk " + toString() + " ref count now " + refCount);
126         }
127     }
128 
129     public synchronized void release() {
130         --refCount;
131         if (Logger.shouldLog(LogTag.JFR_SYSTEM, LogLevel.DEBUG)) {
132             Logger.log(LogTag.JFR_SYSTEM, LogLevel.DEBUG, "Release chunk " + toString() + " ref count now " + refCount);
133         }
134         if (refCount == 0) {
135             destroy();
136         }
137     }
138 
139     public long getSize() {
140         return size;
141     }
142 
143     public boolean isFinished() {
144         return endTime != null;
145     }
146 
147     @Override
148     public String toString() {
149         return chunkFile.toString();
150     }
151 
152     ReadableByteChannel newChannel() throws IOException {
153         if (!isFinished()) {
154             throw new IOException("Chunk not finished");
155         }
156         return ((SecuritySupport.newFileChannelToRead(chunkFile)));
157     }
158 
159     public boolean inInterval(Instant startTime, Instant endTime) {
160         if (startTime != null && getEndTime().isBefore(startTime)) {
161             return false;
162         }
163         if (endTime != null && getStartTime().isAfter(endTime)) {
164             return false;
165         }
166         return true;
167     }
168 
169     public SafePath getFile() {
170         return chunkFile;
171     }
172 
173     public long getCurrentFileSize() {
174         try {
175             return SecuritySupport.getFileSize(chunkFile);
176         } catch (IOException e) {
177             return 0L;
178         }
179     }
180 
181     boolean isMissingFile() {
182         try {
183             return !SecuritySupport.exists(chunkFile);
184         } catch (IOException ioe) {
185             return true;
186         }
187     }
188 }
