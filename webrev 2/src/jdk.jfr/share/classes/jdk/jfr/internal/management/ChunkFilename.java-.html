25 package jdk.jfr.internal.management;
26 
27 import java.nio.file.Paths;
28 import java.nio.file.Path;
29 import java.time.LocalDateTime;
30 import java.io.IOException;
31 
32 import jdk.jfr.internal.SecuritySupport;
33 import jdk.jfr.internal.util.ValueFormatter;
34 import jdk.jfr.internal.consumer.FileAccess;
35 
36 // Allows a remote streaming client to create chunk files
37 // with same naming scheme as the JVM.
38 public final class ChunkFilename {
39    private static final int MAX_CHUNK_NAMES = 100_000;
40    private static final String FILE_EXTENSION = ".jfr";
41 
42    private final Path directory;
43    private final FileAccess fileAcess;
44 
45    private Path lastPath;
46    private int counter;
47 
48    public static ChunkFilename newUnpriviliged(Path directory) {
49        return new ChunkFilename(directory, FileAccess.UNPRIVILEGED);
50    }
51 
52    public static ChunkFilename newPriviliged(Path directory) {
53        return new ChunkFilename(directory, SecuritySupport.PRIVILEGED);
54    }
55 
56    private ChunkFilename(Path directory, FileAccess fileAccess) {
57        // Avoid malicious implementations of Path interface
58        this.directory = Paths.get(directory.toString());
59        this.fileAcess = fileAccess;
60    }
61 
62    public String next(LocalDateTime time) throws IOException {
63        String filename = ValueFormatter.formatDateTime(time);
64        Path p = directory.resolve(filename + FILE_EXTENSION);
65 
66        // If less than one file per second (typically case)
67        if (lastPath == null || !p.equals(lastPath)) {
68            if (!fileAcess.exists(p)) {
69                counter = 1; // reset counter
70                lastPath = p;
71                return p.toString();
72            }
73        }
74 
75        // If more than one file per second
76        while (counter < MAX_CHUNK_NAMES) {
77            String extendedName = makeExtendedName(filename, counter);
78            p = directory.resolve(extendedName);
79            counter++;
80            if (!fileAcess.exists(p)) {
81                return p.toString();
82            }
83        }
84        throw new IOException("Unable to find unused filename after " + counter + " attempts");
85    }
86 
87    private String makeExtendedName(String filename, int counter) {
88        StringBuilder sb = new StringBuilder();
89        sb.append(filename);
90        sb.append('_');
91        if (counter < 10) { // chronological sorted
92            sb.append('0');
93        }
94        sb.append(counter);
95        sb.append(FILE_EXTENSION);
96        return sb.toString();
97    }
98 }
