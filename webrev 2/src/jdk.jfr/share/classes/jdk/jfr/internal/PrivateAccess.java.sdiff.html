 43 /**
 44  * Provides access to package private function in jdk.jfr.
 45  * <p>
 46  * The static initializer in this class loads the Settings class, which will
 47  * call {@link #setPrivateAccess(PrivateAccess)} on this class, which can be
 48  * used call to package protected methods.
 49  *
 50  * This is similar to how java.lang accesses package private methods in
 51  * java.lang.reflect.
 52  */
 53 public abstract class PrivateAccess {
 54     private static volatile PrivateAccess instance;
 55 
 56     public static PrivateAccess getInstance() {
 57         // Can't be initialized in <clinit> because it may
 58         // deadlock with FlightRecorderPermission.<clinit>
 59         if (instance == null) {
 60             // Will trigger
 61             // FlightRecorderPermission.<clinit>
 62             // which will call PrivateAccess.setPrivateAccess
 63             new FlightRecorderPermission(SecuritySupport.REGISTER_EVENT);
 64         }
 65         return instance;
 66     }
 67 
 68     public static void setPrivateAccess(PrivateAccess pa) {
 69         instance = pa;
 70     }
 71

 74     public abstract Configuration newConfiguration(String name, String label, String description, String provider, Map<String,String> settings, String contents);
 75 
 76     public abstract EventType newEventType(PlatformEventType eventTypes);
 77 
 78     public abstract AnnotationElement newAnnotation(Type annotationType, List<Object> values, boolean boot);
 79 
 80     public abstract ValueDescriptor newValueDescriptor(String name, Type fieldType, List<AnnotationElement> annotations, int dimension, boolean constantPool, String fieldName);
 81 
 82     public abstract PlatformRecording getPlatformRecording(Recording r);
 83 
 84     public abstract PlatformEventType getPlatformEventType(EventType eventType);
 85 
 86     public abstract boolean isConstantPool(ValueDescriptor v);
 87 
 88     public abstract String getFieldName(ValueDescriptor v);
 89 
 90     public abstract ValueDescriptor newValueDescriptor(Class<?> type, String name);
 91 
 92     public abstract SettingDescriptor newSettingDescriptor(Type type, String name, String def, List<AnnotationElement> aes);
 93 
 94     public abstract void setAnnotations(ValueDescriptor v, List<AnnotationElement> a);
 95 
 96     public abstract void setAnnotations(SettingDescriptor s, List<AnnotationElement> a);
 97 
 98     public abstract boolean isUnsigned(ValueDescriptor v);
 99 
100     public abstract PlatformRecorder getPlatformRecorder();
101 
102     @SuppressWarnings("removal")
103     public abstract AccessControlContext getContext(SettingControl sc);
104 
105     public abstract EventSettings newEventSettings(EventSettingsModifier esm);
106 
107     public abstract boolean isVisible(EventType t);
108 }
