 26 package jdk.jfr.internal.consumer;
 27 
 28 import java.io.IOException;
 29 import java.nio.file.Path;
 30 import java.security.AccessControlContext;
 31 import java.util.Arrays;
 32 import java.util.Collections;
 33 import java.util.Comparator;
 34 import jdk.jfr.consumer.RecordedEvent;
 35 
 36 /**
 37  * Implementation of an event stream that operates against a recording file.
 38  *
 39  */
 40 public final class EventFileStream extends AbstractEventStream {
 41     private static final Comparator<? super RecordedEvent> EVENT_COMPARATOR = JdkJfrConsumer.instance().eventComparator();
 42 
 43     private final RecordingInput input;
 44 
 45     private ChunkParser currentParser;
 46     private RecordedEvent[] cacheSorted;
 47 
 48     public EventFileStream(@SuppressWarnings("removal") AccessControlContext acc, Path file) throws IOException {
 49         super(acc, Collections.emptyList());
 50         this.input = new RecordingInput(file.toFile(), FileAccess.UNPRIVILEGED);
 51         this.input.setStreamed();
 52     }
 53 
 54     @Override
 55     public void start() {
 56         start(0);
 57     }
 58 
 59     @Override
 60     public void startAsync() {
 61         startAsync(0);
 62     }
 63 
 64     @Override
 65     public void close() {
 66         closeParser();
 67         dispatcher().runCloseActions();
 68         try {
 69             input.close();
 70         } catch (IOException e) {
 71             // ignore
 72         }
 73     }
 74 
 75     @Override
 76     protected boolean isRecordingStream() {
 77         return false;
 78     }
 79 
 80     @Override
 81     protected void process() throws IOException {
 82         Dispatcher disp = dispatcher();
 83         long start = 0;
 84         long end = Long.MAX_VALUE;
 85         if (disp.startTime != null) {
 86             start = disp.startNanos;
 87         }
 88         if (disp.endTime != null) {
 89             end = disp.endNanos;
 90         }
 91 
 92         currentParser = new ChunkParser(input, disp.parserConfiguration, parserState());
 93         while (!isClosed()) {
 94             onMetadata(currentParser);
 95             if (currentParser.getStartNanos() > end) {
 96                 close();
 97                 return;
 98             }
 99             disp = dispatcher();
100             var ranged  = disp.parserConfiguration.withRange(start, end);
101             currentParser.updateConfiguration(ranged, true);
102             if (disp.parserConfiguration.ordered()) {
103                 processOrdered(disp);
104             } else {
105                 processUnordered(disp);
106             }
107             currentParser.resetCache();
108             if (isClosed() || currentParser.isLastChunk()) {
109                 return;
110             }
111             currentParser = currentParser.nextChunkParser();
112         }
113     }
114 
115     private void processOrdered(Dispatcher c) throws IOException {
116         if (cacheSorted == null) {
117             cacheSorted = new RecordedEvent[10_000];
118         }
119         RecordedEvent event;
120         int index = 0;
121         while (!currentParser.isChunkFinished()) {
122             while ((event = currentParser.readStreamingEvent()) != null) {
123                 if (index == cacheSorted.length) {
124                     RecordedEvent[] tmp = cacheSorted;
125                     cacheSorted = new RecordedEvent[2 * tmp.length];
126                     System.arraycopy(tmp, 0, cacheSorted, 0, tmp.length);
127                 }
128                 cacheSorted[index++] = event;
129             }
130             dispatchOrdered(c, index);
131             index = 0;
132         }
133     }
134 
135     private void dispatchOrdered(Dispatcher c, int index) {
136         onMetadata(currentParser);
137         Arrays.sort(cacheSorted, 0, index, EVENT_COMPARATOR);
138         for (int i = 0; i < index; i++) {
139             c.dispatch(cacheSorted[i]);
140         }
141         onFlush();
142     }
143 
144     private void processUnordered(Dispatcher c) throws IOException {
145         onMetadata(currentParser);
146         while (!isClosed()) {
147             RecordedEvent event = currentParser.readStreamingEvent();
148             if (event == null) {
149                 onFlush();
150                 if (currentParser.isChunkFinished()) {
151                     return;
152                 }
153                 continue;
154             }
155             onMetadata(currentParser);
156             c.dispatch(event);
157         }
158     }
159 }
