25 package jdk.jfr.internal.periodic;
26 
27 import jdk.internal.event.Event;
28 import jdk.jfr.internal.util.Utils;
29 
30 /**
31  * Periodic task that runs trusted code that doesn't require an access control
32  * context.
33  * <p>
34  * This class can be removed once the Security Manager is no longer supported.
35  */
36 final class JDKEventTask extends JavaEventTask {
37 
38     public JDKEventTask(Class<? extends Event> eventClass, Runnable runnable) {
39         super(eventClass, runnable);
40         if (!getEventType().isJDK()) {
41             throw new InternalError("Must be a JDK event");
42         }
43         if (!Utils.isJDKClass(eventClass)) {
44             throw new SecurityException("Periodic task can only be registered for event classes that belongs to the JDK");
45         }
46         if (!Utils.isJDKClass(runnable.getClass())) {
47             throw new SecurityException("Runnable class must belong to the JDK");
48         }
49     }
50 
51     @Override
52     public void execute(long timestamp, PeriodicType periodicType) {
53         getRunnable().run();
54     }
55 }
