26 package jdk.jfr.internal;
27 






28 import java.lang.invoke.MethodHandles;



























29 import jdk.internal.module.Modules;
30 import jdk.jfr.Event;





31 




32 public final class SecuritySupport {






33     private static final Module JFR_MODULE = Event.class.getModule();
34     private static final MethodHandles.Lookup LOOKUP = MethodHandles.lookup();


35 
36     static {
37         // ensure module java.base can read module jdk.jfr as early as possible
38         addReadEdge(Object.class);
39         addInternalEventExport(Object.class);
40         addEventsExport(Object.class);
41     }
42 

















































































































































































































43     static void makeVisibleToJFR(Class<?> clazz) {
44         Module classModule = clazz.getModule();
45         Modules.addReads(JFR_MODULE, classModule);
46         if (clazz.getPackage() != null) {
47             String packageName = clazz.getPackage().getName();
48             Modules.addExports(classModule, packageName, JFR_MODULE);
49             Modules.addOpens(classModule, packageName, JFR_MODULE);
50         }
51     }
52 
53     /**
54      * Adds a qualified export of the internal.jdk.jfr.internal.event package
55      * (for EventConfiguration and EventWriter)
56      */
57     static void addInternalEventExport(Class<?> clazz) {
58         Modules.addExports(JFR_MODULE, "jdk.jfr.internal.event", clazz.getModule());
59     }
60 
61     static void addEventsExport(Class<?> clazz) {
62         Modules.addExports(JFR_MODULE, "jdk.jfr.events", clazz.getModule());
63     }
64 
65     static void addReadEdge(Class<?> clazz) {
66         Modules.addReads(clazz.getModule(), JFR_MODULE);
67     }
68 











































































































69     public static void ensureClassIsInitialized(Class<?> clazz) {
70         try {
71             MethodHandles.Lookup lookup = MethodHandles.privateLookupIn(clazz, LOOKUP);










72             lookup.ensureInitialized(clazz);
73         } catch (IllegalAccessException e) {
74             throw new InternalError(e);


75         }
76     }
77 

78     static Class<?> defineClass(Class<?> lookupClass, byte[] bytes) {
79         try {
80             return MethodHandles.privateLookupIn(lookupClass, LOOKUP).defineClass(bytes);
81         } catch (IllegalAccessException e) {
82             throw new InternalError(e);




























































83         }
84     }


85 }
