 26 package jdk.jfr.internal;
 27 
 28 import java.io.IOException;
 29 import java.nio.file.Files;
 30 import java.nio.file.Path;
 31 

 32 import jdk.internal.misc.Unsafe;
 33 
 34 import static java.nio.file.LinkOption.*;
 35 
 36 /**
 37  * Options that control Flight Recorder.
 38  *
 39  * Can be set using JFR.configure
 40  *
 41  */
 42 public final class Options {
 43 
 44     private static final long WAIT_INTERVAL = 1000; // ms;
 45 
 46     private static final long MIN_MAX_CHUNKSIZE = 1024 * 1024;
 47 
 48     private static final long DEFAULT_GLOBAL_BUFFER_COUNT = 20;
 49     private static final long DEFAULT_GLOBAL_BUFFER_SIZE = 524288;
 50     private static final long DEFAULT_MEMORY_SIZE = DEFAULT_GLOBAL_BUFFER_COUNT * DEFAULT_GLOBAL_BUFFER_SIZE;
 51     private static long DEFAULT_THREAD_BUFFER_SIZE;
 52     private static final int DEFAULT_STACK_DEPTH = 64;
 53     private static final long DEFAULT_MAX_CHUNK_SIZE = 12 * 1024 * 1024;
 54     private static final Path DEFAULT_DUMP_PATH = null;
 55     private static final boolean DEFAULT_PRESERVE_REPOSITORY = false;
 56 
 57     private static long memorySize;
 58     private static long globalBufferSize;
 59     private static long globalBufferCount;
 60     private static long threadBufferSize;
 61     private static int stackDepth;
 62     private static long maxChunkSize;
 63     private static boolean preserveRepository;
 64 
 65     static {
 66         final long pageSize = Unsafe.getUnsafe().pageSize();
 67         DEFAULT_THREAD_BUFFER_SIZE = pageSize > 8 * 1024 ? pageSize : 8 * 1024;
 68         reset();
 69     }
 70 
 71     public static synchronized void setMaxChunkSize(long max) {
 72         if (max < MIN_MAX_CHUNKSIZE) {
 73             throw new IllegalArgumentException("Max chunk size must be at least " + MIN_MAX_CHUNKSIZE);
 74         }
 75         JVM.setFileNotification(max);
 76         maxChunkSize = max;
 77     }
 78 
 79     public static synchronized long getMaxChunkSize() {
 80         return maxChunkSize;
 81     }
 82 
 83     public static synchronized void setMemorySize(long memSize) {
 84         JVM.setMemorySize(memSize);
 85         memorySize = memSize;
 86     }
 87 
 88     public static synchronized long getMemorySize() {
 89         return memorySize;
 90     }
 91 
 92     public static synchronized void setThreadBufferSize(long threadBufSize) {
 93         JVM.setThreadBufferSize(threadBufSize);
 94         threadBufferSize = threadBufSize;
 95     }
 96 
 97     public static synchronized long getThreadBufferSize() {
 98         return threadBufferSize;
 99     }
100 
101     public static synchronized long getGlobalBufferSize() {
102         return globalBufferSize;
103     }
104 
105     public static synchronized void setGlobalBufferCount(long globalBufCount) {
106         JVM.setGlobalBufferCount(globalBufCount);
107         globalBufferCount = globalBufCount;
108     }
109 
110     public static synchronized long getGlobalBufferCount() {
111         return globalBufferCount;
112     }
113 
114     public static synchronized void setGlobalBufferSize(long globalBufsize) {
115         JVM.setGlobalBufferSize(globalBufsize);
116         globalBufferSize = globalBufsize;
117     }
118 
119     public static synchronized void setDumpPath(Path path) throws IOException {
120         if (path != null) {
121             if (Files.isWritable(path)) {
122                 path = path.toRealPath(NOFOLLOW_LINKS);
123             } else {
124                 throw new IOException("Cannot write JFR emergency dump to " + path.toString());
125             }
126         }
127         JVM.setDumpPath(path == null ? null : path.toString());
128     }
129 
130     public static synchronized Path getDumpPath() {
131         return Path.of(JVM.getDumpPath());
132     }
133 
134     public static synchronized void setStackDepth(Integer stackTraceDepth) {
135         JVM.setStackDepth(stackTraceDepth);
136         stackDepth = stackTraceDepth;
137     }
138 
139     public static synchronized int getStackDepth() {
140         return stackDepth;
141     }
142 
143     public static synchronized void setPreserveRepository(boolean preserve) {
144         preserveRepository = preserve;
145     }
146 
147     public static synchronized boolean getPreserveRepository() {
148         return preserveRepository;
149     }
150 
151     private static synchronized void reset() {
152         setMaxChunkSize(DEFAULT_MAX_CHUNK_SIZE);
153         setMemorySize(DEFAULT_MEMORY_SIZE);
154         setGlobalBufferSize(DEFAULT_GLOBAL_BUFFER_SIZE);
155         setGlobalBufferCount(DEFAULT_GLOBAL_BUFFER_COUNT);
156         try {
157             setDumpPath(DEFAULT_DUMP_PATH);
158         } catch (IOException e) {
159             // Ignore (depends on default value in JVM: it would be NULL)
160         }
161         setStackDepth(DEFAULT_STACK_DEPTH);
162         setThreadBufferSize(DEFAULT_THREAD_BUFFER_SIZE);
163         setPreserveRepository(DEFAULT_PRESERVE_REPOSITORY);
164     }
165 
166     static synchronized long getWaitInterval() {
167         return WAIT_INTERVAL;
168     }
169 
170     static void ensureInitialized() {
171         // trigger clinit which will setup JVM defaults.
172     }
173 
174 
175 }
