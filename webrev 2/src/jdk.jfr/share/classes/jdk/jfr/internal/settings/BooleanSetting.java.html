26 package jdk.jfr.internal.settings;
27 
28 import java.util.Objects;
29 import java.util.Set;
30 
31 import jdk.jfr.SettingControl;
32 import jdk.jfr.internal.PlatformEventType;
33 
34 abstract class BooleanSetting extends SettingControl {
35     private final PlatformEventType eventType;
36     private final String defaultValue;
37     private String value;
38 
39     public BooleanSetting(PlatformEventType eventType, String defaultValue) {
40         this.eventType = Objects.requireNonNull(eventType);
41         this.defaultValue = defaultValue;
42         this.value = defaultValue;
43         if (parse(defaultValue) == null) {
44             throw new InternalError("Only 'true' or 'false' is allowed with class BooleanSetting");
45         }
46     }
47 
48     protected abstract void apply(PlatformEventType eventType, boolean value);
49 
50     @Override
51     public String combine(Set<String> values) {
52         if (values.contains("true")) {
53             return "true";
54         }
55         if (values.contains("false")) {
56             return "false";
57         }
58         return defaultValue;
59     }
60 
61     @Override
62     public void setValue(String value) {
63         Boolean b = parse(value);
64         if (b != null) {
65             apply(eventType, b.booleanValue());
66             this.value = value;
67         }
68     }
69 
70     @Override
71     public String getValue() {
72         return value;
73     }
74 
75     private static Boolean parse(String value) {
76         if ("true".equals(value)) {
77             return Boolean.TRUE;
78         }
79         if ("false".equals(value)) {
80             return Boolean.FALSE;
81         }
82         return null;
83     }
84 }
