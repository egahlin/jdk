454     public static Path getPathInProperty(String prop, String subPath) {
455         String path = System.getProperty(prop);
456         if (path == null) {
457             return null;
458         }
459         File file = subPath == null ? new File(path) : new File(path, subPath);
460         return file.toPath().toAbsolutePath();
