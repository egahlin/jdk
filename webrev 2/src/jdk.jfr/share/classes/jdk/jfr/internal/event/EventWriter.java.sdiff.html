 36 // User code should not be able to get access to an EventWriter instance as it
 37 // would allow it to write arbitrary data into buffers, potentially from
 38 // different threads.
 39 //
 40 // This is prevented in three ways:
 41 //
 42 // 1. For code to access the jdk.jfr.internal.event package
 43 //    at least one event class (for a particular module) must be
 44 //    registered having FlightRecorderPermission("registerEvent").
 45 //
 46 // 2. The EventWriter EventWriterFactory::getEventWriter(long) method can only be linked from
 47 //    the UserEvent::commit() method instrumented by JFR. This is ensured by the JVM.
 48 //    (The EventWriterFactory class is dynamically generated before the first event
 49 //    is instrumented. See EventWriterFactoryRecipe)
 50 //
 51 // 3. Steps 1 and 2 are sufficient to make it fully secure, with or without a Security
 52 //    Manager, but as an additional measure, the method EventWriterFactory::getEventWriter(long)
 53 //    requires the caller to provide a key that is hard to guess. The key is generated
 54 //    into the bytecode of the method invoking getEventWriter(long).
 55 //
 56 public final class EventWriter {
 57 
 58     // Event may not exceed size for a padded integer
 59     private static final long MAX_EVENT_SIZE = (1 << 28) -1;
 60     private static final Unsafe unsafe = Unsafe.getUnsafe();
 61 
 62     // The JVM needs access to these values. Don't remove
 63     private final long threadID;
 64     private long startPosition;
 65     private long currentPosition;
 66     private long maxPosition;
 67     private boolean valid;
 68     private boolean pinVirtualThread;
 69     boolean excluded;
 70 
 71     private PlatformEventType eventType;
 72     private boolean largeSize = false;
 73 








 74     // User code must not be able to instantiate
 75     private EventWriter() {
 76         threadID = 0;
 77     }
 78 
 79     public void putBoolean(boolean i) {
 80         if (isValidForSize(Byte.BYTES)) {
 81             unsafe.putBoolean(null, currentPosition, i);
 82             ++currentPosition;
 83         }
 84     }
 85 
 86     public void putByte(byte i) {
 87         if (isValidForSize(Byte.BYTES)) {
 88             unsafe.putByte(currentPosition, i);
 89             ++currentPosition;
 90         }
 91     }

223                 return false;
224             }

226         return true;
227     }
228 
229     private int usedSize() {
230         return (int) (currentPosition - startPosition);
231     }
232 
233     private void flush() {
234         flush(usedSize(), 0);
235     }
236 
237     private void flush(int usedSize, int requestedSize) {
238         JVM.flush(this, usedSize, requestedSize);
239     }

241     public boolean beginEvent(EventConfiguration configuration, long typeId) {
242         // Malicious code could take the EventConfiguration object from one
243         // event class field and assign it to another. This check makes sure
244         // the event type matches what was added by instrumentation.
245         if (configuration.getId() != typeId) {
246             EventWriterKey.block();
247         }
248         if (excluded) {
249             // thread is excluded from writing events
250             return false;
251         }
252         this.eventType = configuration.getPlatformEventType();
253         reserveEventSizeField();
254         putLong(eventType.getId());
255         return true;
256     }

261             valid = true;
262             return true;
263         }
264         final int eventSize = usedSize();

266             reset();
