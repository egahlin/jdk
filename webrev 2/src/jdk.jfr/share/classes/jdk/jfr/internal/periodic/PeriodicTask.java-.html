 25 package jdk.jfr.internal.periodic;
 26 
 27 import jdk.jfr.internal.LogLevel;
 28 import jdk.jfr.internal.LogTag;
 29 import jdk.jfr.internal.Logger;
 30 
 31 /**
 32  * Base class that holds time related information for a periodic task.
 33  * <p>
 34  * Class hierarchy for periodic tasks:
 35  * <pre>
 36  *               PeriodicTask
 37  *                /        \
 38  *               /          \
 39  *          EventTask    FlushTask
 40  *           /     \
 41  *          /       \
 42  * JVMEventTask   JavaEventTask
 43  *                /         \
 44  *               /           \
 45  *      UserEventTask     JDKEventTask
 46  * </pre>
 47  * <p>
 48  * State modifications should only be done from the periodic task thread.
 49  */
 50 abstract class PeriodicTask {
 51     private final LookupKey lookupKey;
 52     private final String name;
 53     // State only to be modified by the periodic task thread
 54     private long counter;
 55     private long period;
 56     private Batch batch;
 57 
 58     public PeriodicTask(LookupKey lookupKey, String name) {
 59         this.lookupKey = lookupKey;
 60         this.name = name;
 61     }
 62 
 63     public abstract void execute(long timestamp, PeriodicType periodicType);
 64 
 65     public abstract boolean isSchedulable();
 66 
 67     protected abstract long fetchPeriod();
 68 
 69     public final LookupKey getLookupKey() {
 70         return lookupKey;
 71     }
 72 
 73     public final String getName() {
 74         return name;
 75     }
 76 
 77     // Only to be called from periodic task thread
 78     public void setBatch(Batch batch) {
 79         this.batch = batch;
 80     }
 81 
 82     // Only to be called from periodic task thread
 83     public Batch getBatch() {
 84         return batch;
 85     }
 86 
 87     // Only to be called from periodic task thread
 88     public final void tick() {
 89         long increment = batch.getPeriod();
 90         if (period != 0) {
 91             counter = (counter + increment) % period;
 92             if (Logger.shouldLog(LogTag.JFR_PERIODIC, LogLevel.DEBUG)) {
 93                 boolean trace = Logger.shouldLog(LogTag.JFR_PERIODIC, LogLevel.TRACE);
 94                 boolean run = shouldRun();
 95                 if (trace || run) {
 96                     logInterval(trace, run ? "Run" : "Skip");
 97                 }
 98             }
 99         }
100     }
101 
102     // Only to be called from periodic task thread
103     public final boolean shouldRun() {
104         return counter == 0 && period != 0;
105     }
106 
107     // Only to be called from periodic task thread
108     public final void updatePeriod() {
109         long p = fetchPeriod();
110         // Reset counter if new period
111         if (p != period) {
112             counter = 0;
113             period = p;
114             batch = null;
115         }
116     }
117 
118     // Only to be called from periodic task thread
119     public final long getPeriod() {
120         return period;
121     }
122 
123     public final void run(long timestamp, PeriodicType periodicType) {
124         if (periodicType != PeriodicType.INTERVAL) {
125             logChunk(periodicType);
126         }
127         try {
128             execute(timestamp, periodicType);
129         } catch (Throwable e) {
130             // Prevent malicious user to propagate exception callback in the wrong context
131             Logger.log(LogTag.JFR_SYSTEM, LogLevel.WARN, "Exception occurred during execution of " + name);
132         }
133     }
134 
135     private void logChunk(PeriodicType periodicType) {
136         if (Logger.shouldLog(LogTag.JFR_PERIODIC, LogLevel.DEBUG)) {
137             String action = periodicType == PeriodicType.BEGIN_CHUNK ? "beginChunk" : "endChunk";
138             String message = "Run " + action + " " + getName();
139             Logger.log(LogTag.JFR_PERIODIC, LogLevel.DEBUG, message);
140         }
141     }
142 
143     private void logInterval(boolean trace, String action) {
144         String message = action + " periodic " + getName();
145         if (trace) {
146             // Makes the counter run on x/x instead of 0/x which looks strange.
147             long a =  counter == 0 ? period : counter;
148             Logger.log(LogTag.JFR_PERIODIC, LogLevel.TRACE, message + (" " + a + "/" + period));
149         } else {
150             Logger.log(LogTag.JFR_PERIODIC, LogLevel.DEBUG, message);
151         }
152     }
153 }
