31 import jdk.jfr.internal.PlatformEventType;
32 
33 abstract class BooleanSetting extends JDKSettingControl {
34     private final PlatformEventType eventType;
35     private final String defaultValue;
36     private String value;
37 
38     public BooleanSetting(PlatformEventType eventType, String defaultValue) {
39         this.eventType = Objects.requireNonNull(eventType);
40         this.defaultValue = defaultValue;
41         this.value = defaultValue;
42         if (parse(defaultValue) == null) {
43             throw new InternalError("Only 'true' or 'false' is allowed with class BooleanSetting");
44         }
45     }
46 
47     protected abstract void apply(PlatformEventType eventType, boolean value);
48 
49     @Override
50     public String combine(Set<String> values) {
51         if (values.contains("true")) {
52             return "true";
53         }
