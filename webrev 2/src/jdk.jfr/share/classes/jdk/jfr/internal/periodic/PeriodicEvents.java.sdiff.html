 39 /**
 40  * Class that runs and schedules tasks for periodic events.
 41  * <p>
 42  * Events can run at the beginning of a chunk rotation, at the end of a chunk
 43  * rotation or at a periodic interval.
 44  * <p>
 45  * Events with the same period runs in the same order as users added them.
 46  * Temporarily disabling events don't impact the order of execution.
 47  * <p>
 48  * A best effort is made to run events with different periods at the same time.
 49  * For example, an event that runs every two seconds executes half of the time
 50  * with the events that run every second.
 51  */
 52 public final class PeriodicEvents {
 53     private static final TaskRepository taskRepository = new TaskRepository();
 54     private static final BatchManager batchManager = new BatchManager();
 55     private static final FlushTask flushTask = new FlushTask();
 56     private static final AtomicLong settingsIteration = new AtomicLong();
 57 
 58     // State only to be read and modified by periodic task thread
 59     private static long lastTimeMillis;
 60 
 61     public static void addJDKEvent(Class<? extends Event> eventClass, Runnable runnable) {
 62         taskRepository.add(new JDKEventTask(eventClass, runnable));
 63     }
 64 
 65     public static void addJVMEvent(PlatformEventType eventType) {
 66         taskRepository.add(new JVMEventTask(eventType));
 67     }
 68 
 69     public static void addUserEvent(@SuppressWarnings("removal") AccessControlContext acc, Class<? extends Event> eventClass, Runnable runnable) {
 70         taskRepository.add(new UserEventTask(acc, eventClass, runnable));
 71     }
 72 
 73     public static boolean removeEvent(Runnable runnable) {
 74         return taskRepository.removeTask(runnable);
 75     }
 76 
 77     public static void doChunkBegin() {
 78         long timestamp = JVM.counterTime();
 79         for (EventTask task : taskRepository.getTasks()) {
 80             var eventType = task.getEventType();
 81             if (eventType.isEnabled() && eventType.isBeginChunk()) {
 82                 task.run(timestamp, PeriodicType.BEGIN_CHUNK);
 83             }
 84         }
 85     }
 86 
 87     public static void doChunkEnd() {
 88         long timestamp = JVM.counterTime();
 89         for (EventTask task : taskRepository.getTasks()) {
 90             var eventType = task.getEventType();
 91             if (eventType.isEnabled() && eventType.isEndChunk()) {
 92                 task.run(timestamp, PeriodicType.END_CHUNK);
