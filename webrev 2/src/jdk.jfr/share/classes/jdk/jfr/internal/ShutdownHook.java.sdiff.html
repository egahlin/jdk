35 /**
36  * Class responsible for dumping recordings on exit
37  *
38  */
39 final class ShutdownHook implements Runnable {
40     private final PlatformRecorder recorder;
41     Object tlabDummyObject;
42 
43     ShutdownHook(PlatformRecorder recorder) {

44         this.recorder = recorder;
45     }
46 
47     @Override
48     public void run() {
49         // this allocation is done in order to fetch a new TLAB before
50         // starting any "real" operations. In low memory situations,
51         // we would like to take an OOM as early as possible.
52         tlabDummyObject = new Object();
53         recorder.setInShutDown();
54         for (PlatformRecording recording : recorder.getRecordings()) {
55             if (recording.getDumpOnExit() && recording.getState() == RecordingState.RUNNING) {
56                 dump(recording);
57             }
58         }
59         recorder.destroy();
60     }
61 
62     private void dump(PlatformRecording recording) {
63         try {
64             WriteableUserPath dest = recording.getDestination();
65             if (dest == null) {
66                 dest = recording.makeDumpPath();
67                 recording.setDestination(dest);
68             }
69             if (dest != null) {
70                 recording.stop("Dump on exit");
71             }
72         } catch (Exception e) {
73             if (Logger.shouldLog(LogTag.JFR, LogLevel.DEBUG)) {
74                 Logger.log(LogTag.JFR, LogLevel.DEBUG, "Could not dump recording " + recording.getName() + " on exit.");
75             }
76         }
77     }
78 
79     static final class ExceptionHandler implements Thread.UncaughtExceptionHandler {
80         @Override
81         public void uncaughtException(Thread t, Throwable e) {
82             JVM.uncaughtException(t, e);
83         }
84     }
