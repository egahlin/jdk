 26 package jdk.jfr.internal;
 27 




 28 import java.util.HashSet;
 29 import java.util.Objects;
 30 import java.util.Set;
 31 
 32 import jdk.jfr.SettingControl;

 33 import jdk.jfr.internal.settings.PeriodSetting;
 34 import jdk.jfr.internal.settings.StackTraceSetting;
 35 import jdk.jfr.internal.settings.ThresholdSetting;
 36 
 37 final class Control {


 38     private static final int CACHE_SIZE = 5;
 39     private final Set<?>[] cachedUnions = new HashSet<?>[CACHE_SIZE];
 40     private final String[] cachedValues = new String[CACHE_SIZE];
 41     private final SettingControl delegate;
 42     private String defaultValue;
 43     private String lastValue;
 44 
 45     // called by exposed subclass in external API
 46     public Control(SettingControl delegate, String defaultValue) {

 47         this.delegate = delegate;
 48         this.defaultValue = defaultValue;



 49     }
 50 
 51     boolean isType(Class<? extends SettingControl> clazz) {
 52         return delegate.getClass() == clazz;
 53     }
 54 
 55     final void apply(Set<String> values) {
 56         setValue(findCombine(values));
 57     }
 58 
 59     final void setDefault() {
 60         if (defaultValue == null) {
 61             defaultValue = getValue();
 62         }
 63         apply(defaultValue);
 64     }
 65 

 66     public String getValue() {
 67         return delegate.getValue();
















 68     }
 69 
 70     private void apply(String value) {
 71         if (lastValue != null && Objects.equals(value, lastValue)) {
 72             return;
 73         }
 74         setValue(value);
 75     }
 76 

 77     public void setValue(String value) {
 78         try {
 79             delegate.setValue(value);
 80             lastValue = delegate.getValue();
 81         } catch (Throwable t) {
 82             Logger.log(LogTag.JFR_SETTING, LogLevel.WARN, "Exception occurred when setting value \"" + value + "\". " + t.getMessage());
 83             lastValue = null;

















 84         }
 85     }
 86 


 87     public String combine(Set<String> values) {
 88         return delegate.combine(values);















 89     }
 90 
 91     private final String findCombine(Set<String> values) {
 92         if (values.size() == 1) {
 93             return values.iterator().next();
 94         }
 95         for (int i = 0; i < CACHE_SIZE; i++) {
 96             if (Objects.equals(cachedUnions[i], values)) {
 97                 return cachedValues[i];
 98             }
 99         }
100         String result = combine(values);
101         for (int i = 0; i < CACHE_SIZE - 1; i++) {
102             cachedUnions[i + 1] = cachedUnions[i];
103             cachedValues[i + 1] = cachedValues[i];
104         }
105         cachedValues[0] = result;
106         cachedUnions[0] = values;
107         return result;
108     }
109 
110     final String getDefaultValue() {
111         return defaultValue;
112     }
113 
114     final String getLastValue() {
115         return lastValue;
116     }
117 
118     final SettingControl getSettingControl() {
119         return delegate;
120     }
121 
122     boolean isVisible(boolean hasEventHook) {
123         if (isType(ThresholdSetting.class)) {
124             return !hasEventHook;
125         }
126         if (isType(PeriodSetting.class)) {
127             return hasEventHook;
128         }
129         if (isType(StackTraceSetting.class)) {
130             return !hasEventHook;
131         }
132         return true;
133     }
134 }
