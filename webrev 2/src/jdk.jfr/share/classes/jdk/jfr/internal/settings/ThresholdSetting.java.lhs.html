26 package jdk.jfr.internal.settings;
27 
28 import static jdk.jfr.internal.util.ValueParser.MISSING;
29 
30 import java.util.Objects;
31 import java.util.Set;
32 
33 import jdk.jfr.Description;
34 import jdk.jfr.Label;
35 import jdk.jfr.MetadataDefinition;
36 import jdk.jfr.Name;

37 import jdk.jfr.Timespan;
38 import jdk.jfr.internal.PlatformEventType;
39 import jdk.jfr.internal.Type;
40 import jdk.jfr.internal.util.ValueParser;
41 
42 @MetadataDefinition
43 @Label("Threshold")
44 @Name(Type.SETTINGS_PREFIX + "Threshold")
45 @Description("Record event with duration above or equal to threshold")
46 @Timespan
47 public final class ThresholdSetting extends JDKSettingControl {
48     public static final String DEFAULT_VALUE = "0 ns";
49     private static final long typeId = Type.getTypeId(ThresholdSetting.class);
50     private String value = DEFAULT_VALUE;
51     private final PlatformEventType eventType;
52 
53     public ThresholdSetting(PlatformEventType eventType) {
54        this.eventType = Objects.requireNonNull(eventType);
55     }
56 
57     @Override
58     public String combine(Set<String> values) {
59         Long min = null;
60         String text = null;
61         for (String value : values) {
62             long nanos = ValueParser.parseTimespanWithInfinity(value, MISSING);
63             if (nanos != MISSING) {
64                 if (min == null || nanos < min) {
65                     text = value;
66                     min = nanos;
67                 }
68             }
69         }
70         return Objects.requireNonNullElse(text, DEFAULT_VALUE);
71     }
72 
73     @Override
74     public void setValue(String value) {
75         long nanos = ValueParser.parseTimespanWithInfinity(value, MISSING);
76         if (nanos != MISSING) {
77             eventType.setThreshold(nanos);
78             this.value = value;
79         }
80     }
81 
82     @Override
83     public String getValue() {
84         return value;
85     }
86 
87     public static boolean isType(long typeId) {
88         return ThresholdSetting.typeId == typeId;
89     }
90 }
