26 package jdk.jfr.internal.settings;
27 
28 import static jdk.jfr.internal.util.ValueParser.MISSING;
29 
30 import java.util.Objects;
31 import java.util.Set;
32 
33 import jdk.jfr.Description;
34 import jdk.jfr.SettingControl;
35 import jdk.jfr.Label;
36 import jdk.jfr.MetadataDefinition;
37 import jdk.jfr.Name;
38 import jdk.jfr.Timespan;
39 import jdk.jfr.internal.PlatformEventType;
40 import jdk.jfr.internal.Type;
41 import jdk.jfr.internal.util.ValueParser;
42 
43 @MetadataDefinition
44 @Label("Cutoff")
45 @Description("Limit running time of event")
46 @Name(Type.SETTINGS_PREFIX + "Cutoff")
47 @Timespan
48 public final class CutoffSetting extends SettingControl {
49     public static final String DEFAULT_VALUE = ValueParser.INFINITY;
50     private String value = DEFAULT_VALUE;
51     private final PlatformEventType eventType;
52 
53     public CutoffSetting(PlatformEventType eventType) {
54        this.eventType = Objects.requireNonNull(eventType);
55     }
56 
57     @Override
58     public String combine(Set<String> values) {
59         long max = 0;
60         String text = null;
61         for (String value : values) {
62             long nanos = ValueParser.parseTimespanWithInfinity(value, MISSING);
63             if (nanos != MISSING && nanos > max) {
64                 text = value;
65                 max = nanos;
66             }
67         }
68         return Objects.requireNonNullElse(text, DEFAULT_VALUE);
69     }
70 
71     @Override
72     public void setValue(String value) {
73         long nanos = ValueParser.parseTimespanWithInfinity(value, MISSING);
74         if (nanos != MISSING) {
75             eventType.setCutoff(nanos);
76             this.value = value;
77         }
78     }
79 
80     @Override
81     public String getValue() {
82         return value;
83     }
84 
85     public static long parseValueSafe(String value) {
86         if (value == null) {
87             return 0L;
88         }
89         return ValueParser.parseTimespanWithInfinity(value, 0L);
90     }
91 }
