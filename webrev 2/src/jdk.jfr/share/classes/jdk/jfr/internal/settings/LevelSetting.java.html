26 package jdk.jfr.internal.settings;
27 
28 import java.util.Arrays;
29 import java.util.List;
30 import java.util.Objects;
31 import java.util.Set;
32 
33 import jdk.jfr.Label;
34 import jdk.jfr.MetadataDefinition;
35 import jdk.jfr.Name;
36 import jdk.jfr.SettingControl;
37 import jdk.jfr.internal.PlatformEventType;
38 import jdk.jfr.internal.Type;
39 
40 @MetadataDefinition
41 @Label("Level")
42 @Name(Type.SETTINGS_PREFIX + "Level")
43 public final class LevelSetting extends SettingControl {
44     private final PlatformEventType eventType;
45     private final List<String> levels;
46     private String value;
47 
48     public LevelSetting(PlatformEventType eventType, String[] levels) {
49         this.eventType = Objects.requireNonNull(eventType);
50         this.levels = Arrays.asList(Objects.requireNonNull(levels));
51         this.value = levels[0];
52     }
53 
54     @Override
55     public String combine(Set<String> values) {
56         int maxIndex = 0; // index 0 contains the default value
57         for (String value : values) {
58             maxIndex = Math.max(maxIndex, levels.indexOf(value));
59         }
60         return levels.get(maxIndex);
61     }
62 
63     @Override
64     public void setValue(String value) {
65         int index = levels.indexOf(value);
66         if (index != -1) {
67             this.eventType.setLevel(index);
68             this.value = value;
69         }
70     }
71 
72     @Override
73     public String getValue() {
74         return value;
75     }
76 }
