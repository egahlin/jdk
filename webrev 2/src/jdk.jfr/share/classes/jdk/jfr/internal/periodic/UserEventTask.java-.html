25 package jdk.jfr.internal.periodic;
26 
27 import java.security.AccessControlContext;
28 import java.security.AccessController;
29 import java.security.PrivilegedAction;
30 import java.util.Objects;
31 
32 import jdk.internal.event.Event;
33 import jdk.jfr.internal.LogLevel;
34 import jdk.jfr.internal.LogTag;
35 import jdk.jfr.internal.Logger;
36 
37 /**
38  * Class to be used with user-defined events that runs untrusted code.
39  * <p>
40  * This class can be removed once the Security Manager is no longer supported.
41  */
42 final class UserEventTask extends JavaEventTask {
43     @SuppressWarnings("removal")
44     private final AccessControlContext controlContext;
45 
46     public UserEventTask(@SuppressWarnings("removal") AccessControlContext controlContext, Class<? extends Event> eventClass, Runnable runnable) {
47         super(eventClass, runnable);
48         this.controlContext = Objects.requireNonNull(controlContext);
49     }
50 
51     @SuppressWarnings("removal")
52     @Override
53     public void execute(long timestamp, PeriodicType periodicType) {
54         AccessController.doPrivileged((PrivilegedAction<Void>) () -> {
55             execute();
56             return null;
57         }, controlContext);
58     }
59 
60     private void execute() {
61         try {
62             getRunnable().run();
63             if (Logger.shouldLog(LogTag.JFR_EVENT, LogLevel.DEBUG)) {
64                 Logger.log(LogTag.JFR_EVENT, LogLevel.DEBUG, "Executed periodic task for " + getEventType().getLogName());
65             }
66         } catch (Throwable t) {
67             // Prevent malicious user to propagate exception callback in the wrong context
68             Logger.log(LogTag.JFR_EVENT, LogLevel.WARN, "Exception occurred during execution of period task for " + getEventType().getLogName());
69         }
70     }
71 }
