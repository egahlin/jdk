 48 final class Disassemble extends Command {
 49 
 50     @Override
 51     public String getName() {
 52         return "disassemble";
 53     }
 54 
 55     @Override
 56     public List<String> getOptionSyntax() {
 57         List<String> list = new ArrayList<>();
 58         list.add("[--output <directory>]");
 59         list.add("[--max-chunks <chunks>]");
 60         list.add("[--max-size <size>]");
 61         list.add("<file>");
 62         return list;
 63     }
 64 
 65     @Override
 66     public void displayOptionUsage(PrintStream stream) {
 67         stream.println(" --output <directory>    The location to write the disassembled file,");
 68         stream.println("                         by default the current directory");
 69         stream.println("");
 70         stream.println(" --max-chunks <chunks>   Maximum number of chunks per disassembled file,");
 71         stream.println("                         by default 5. The chunk size varies, but is ");
 72         stream.println("                         typically around 15 MB.");
 73         stream.println("");
 74         stream.println(" --max-size <size>       Maximum number of bytes per file.");
 75         stream.println("");
 76         stream.println("  <file>                 Location of the recording file (.jfr)");
 77     }
 78 
 79     @Override
 80     public String getDescription() {
 81         return "Disassemble a recording file into smaller files/chunks";
 82     }
 83 
 84     @Override
 85     public void execute(Deque<String> options) throws UserSyntaxException, UserDataException {
 86         if (options.isEmpty()) {
 87             throw new UserSyntaxException("missing file");
 88         }
 89         Path file = getJFRInputFile(options);
 90         int maxChunks = Integer.MAX_VALUE;
 91         int maxsize = Integer.MAX_VALUE;
 92         String output = System.getProperty("user.dir");
 93         int optionCount = options.size();
 94         while (optionCount > 0) {
 95             if (acceptOption(options, "--output")) {
 96                 output = options.pop();
 97             }
 98             if (acceptOption(options, "--max-size")) {
 99                 String value = options.pop();
100                 try {
101                     maxsize = Integer.parseInt(value);
102                     if (maxsize < 1) {
103                         throw new UserDataException("max size must be at least 1");
104                     }
105                 } catch (NumberFormatException nfe) {
106                     throw new UserDataException("not a valid value for --max-size.");
107                 }
108             }
109             if (acceptOption(options, "--max-chunks")) {
110                 String value = options.pop();
111                 try {
112                     maxChunks = Integer.parseInt(value);
113                     if (maxChunks < 1) {
114                         throw new UserDataException("max chunks must be at least 1.");
115                     }
116                 } catch (NumberFormatException nfe) {
117                     throw new UserDataException("not a valid value for --max-size.");
118                 }
119             }
120             if (optionCount == options.size()) {
121                 // No progress made
122                 throw new UserSyntaxException("unknown option " + options.peek());
123             }
124             optionCount = options.size();
125         }
126         Path outputPath = getDirectory(output);
127 
128         println();
129         println("Examining recording " + file + " ...");
130         List<Long> sizes;
131         if (maxsize != Integer.MAX_VALUE && maxChunks == Integer.MAX_VALUE) {
132             try {
133                 long fileSize = Files.size(file);
134                 if (maxsize >=fileSize) {
135                     println();
136                     println("File size (" + fileSize +") does not exceed max size (" + maxsize + ")");
137                     return;
138                 }
139             } catch (IOException e) {
140                 throw new UserDataException("unexpected i/o error when determining file size" + e.getMessage());
141             }
142         }
143         if (maxsize == Integer.MAX_VALUE && maxChunks == Integer.MAX_VALUE) {
144             maxChunks = 5;
145         }
146 
147         try {
148             sizes = findChunkSizes(file);
149         } catch (IOException e) {
150             throw new UserDataException("unexpected i/o error. " + e.getMessage());
151         }
152         if (maxsize == Integer.MAX_VALUE == sizes.size() <= maxChunks) {
153             throw new UserDataException("number of chunks in recording (" + sizes.size() + ") doesn't exceed max chunks (" + maxChunks + ")");
154         }
155         println();
156         if (sizes.size() > 0) {
157             List<Long> combinedSizes = combineChunkSizes(sizes, maxChunks, maxsize);
158             print("File consists of " + sizes.size() + " chunks. The recording will be split into ");
159             println(combinedSizes.size() + " files");
160             println();
161             splitFile(outputPath, file, combinedSizes);
162         } else {
163             throw new UserDataException("no JFR chunks found in file.");
164         }
165     }
166 
167     private List<Long> findChunkSizes(Path p) throws IOException {
168         try (RecordingInput input = new RecordingInput(p.toFile())) {
169             List<Long> sizes = new ArrayList<>();
170             ChunkHeader ch = new ChunkHeader(input);
171             sizes.add(ch.getSize());
172             while (!ch.isLastChunk()) {
173                 ch = ch.nextHeader();
174                 sizes.add(ch.getSize());
175             }
176             return sizes;
177         }
178     }
179 
180     private List<Long> combineChunkSizes(List<Long> sizes, int maxChunks, long maxSize) {
181         List<Long> reduced = new ArrayList<Long>();
182         int chunks = 1;
183         long fileSize = sizes.getFirst();
184         for (int i = 1; i < sizes.size(); i++) {
185             long size = sizes.get(i);
186             if (fileSize + size > maxSize || chunks == maxChunks) {
187                 reduced.add(fileSize);
188                 chunks = 1;
189                 fileSize = size;
190                 continue;
191             }
192             fileSize += size;
193             chunks++;
194         }
195         if (fileSize != 0) {
196             reduced.add(fileSize);
197         }
198         return reduced;
199     }
200 
201     private void splitFile(Path directory, Path file, List<Long> splitPositions) throws UserDataException {
202         int padAmountZeros = String.valueOf(splitPositions.size() - 1).length();
203         String fileName = file.getFileName().toString();
204         String fileFormatter = fileName.subSequence(0, fileName.length() - 4) + "_%0" + padAmountZeros + "d.jfr";
205         for (int i = 0; i < splitPositions.size(); i++) {
206             String formattedFilename = String.format(fileFormatter, i);
207             try {
208                 Path p = directory.resolve(formattedFilename);
209                 if (Files.exists(p)) {
210                     throw new UserDataException("can't create disassembled file " + p + ", a file with that name already exist");
211                 }
212             } catch (InvalidPathException ipe) {
213                 throw new UserDataException("can't construct path with filename" + formattedFilename);
214             }
215         }
216 
217         try (DataInputStream stream = new DataInputStream(new BufferedInputStream(new FileInputStream(file.toFile())))) {
218             for (int i = 0; i < splitPositions.size(); i++) {
219                 Long l = splitPositions.get(i);
220                 byte[] bytes = readBytes(stream, l.intValue());
221                 String formattedFilename = String.format(fileFormatter, i);
222                 Path p = directory.resolve(formattedFilename);
223                 File splittedFile = p.toFile();
224                 println("Writing " + splittedFile + " ... " + bytes.length);
225                 try (var fos = new FileOutputStream(splittedFile)) {
226                     fos.write(bytes);
227                 }
228             }
229         } catch (IOException ioe) {
230             throw new UserDataException("i/o error writing file " + file);
231         }
232     }
233 
234     private byte[] readBytes(InputStream stream, int count) throws UserDataException, IOException {
235         byte[] data = new byte[count];
236         int totalRead = 0;
237         while (totalRead < data.length) {
238             int read = stream.read(data, totalRead, data.length - totalRead);
239             if (read == -1) {
240                 throw new UserDataException("unexpected end of data");
241             }
242             totalRead += read;
243         }
244         return data;
245     }
