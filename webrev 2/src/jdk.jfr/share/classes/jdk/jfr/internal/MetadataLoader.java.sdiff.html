179     @SuppressWarnings("unchecked")
180     private Class<? extends Annotation> createAnnotationClass(String type) {
181         try {
182             if (!type.startsWith("jdk.jfr.")) {
183                 throw new IllegalStateException("Incorrect type " + type + ". Annotation class must be located in jdk.jfr package.");
184             }
185             Class<?> c = Class.forName(type, true, null);
186             return (Class<? extends Annotation>) c;
187         } catch (ClassNotFoundException cne) {
188             throw new IllegalStateException(cne);
189         }
190     }
