 27 package jdk.jfr.internal.settings;
 28 
 29 import static jdk.jfr.internal.util.TimespanUnit.SECONDS;
 30 import static jdk.jfr.internal.util.TimespanUnit.MILLISECONDS;
 31 
 32 import java.util.Objects;
 33 import java.util.Set;
 34 
 35 import jdk.jfr.Description;
 36 import jdk.jfr.Label;
 37 import jdk.jfr.MetadataDefinition;
 38 import jdk.jfr.Name;
 39 import jdk.jfr.SettingControl;
 40 import jdk.jfr.internal.PlatformEventType;
 41 import jdk.jfr.internal.Throttle;
 42 import jdk.jfr.internal.Type;
 43 import jdk.jfr.internal.util.Rate;
 44 import jdk.jfr.internal.util.TimespanUnit;
 45 import jdk.jfr.internal.util.Utils;
 46 
 47 @MetadataDefinition
 48 @Label("Throttle")
 49 @Description("Throttles the emission rate for an event")
 50 @Name(Type.SETTINGS_PREFIX + "Throttle")
 51 public final class ThrottleSetting extends SettingControl {
 52     public static final String DEFAULT_VALUE = Throttle.DEFAULT;
 53     private final PlatformEventType eventType;
 54     private String value = DEFAULT_VALUE;
 55 
 56     public ThrottleSetting(PlatformEventType eventType) {
 57        this.eventType = Objects.requireNonNull(eventType);
 58     }
 59 
 60     @Override
 61     public String combine(Set<String> values) {
 62         Rate max = null;
 63         String text = null;
 64         for (String value : values) {
 65             Rate rate = Rate.of(value);
 66             if (rate != null) {
 67                 if (max == null || rate.isHigher(max)) {
 68                     text = value;
 69                     max = rate;
 70                 }
 71             }
 72         }
 73         // "off" is default
 74         return Objects.requireNonNullElse(text, DEFAULT_VALUE);
 75     }
 76 
 77     @Override
 78     public void setValue(String value) {
 79         if ("off".equals(value)) {
 80             eventType.setThrottle(-2, 1000);
 81             this.value = value;
 82             return;
 83         }
 84 
 85         Rate rate = Rate.of(value);
 86         if (rate != null) {
 87             long millis = 1000;
 88             long samples = rate.amount();
 89             TimespanUnit unit = rate.unit();
 90             // if unit is more than 1 s, set millis
 91             if (unit.nanos > SECONDS.nanos) {
 92                 millis = unit.nanos / MILLISECONDS.nanos;
 93             }
 94             // if unit is less than 1 s, scale samples
 95             if (unit.nanos < SECONDS.nanos) {
 96                 long perSecond = SECONDS.nanos / unit.nanos;
 97                 samples *= Utils.multiplyOverflow(samples, perSecond, Long.MAX_VALUE);
 98             }
 99             eventType.setThrottle(samples, millis);
100             this.value = value;
101         }
102     }
103 
104     @Override
105     public String getValue() {
106         return value;
107     }
108 }
