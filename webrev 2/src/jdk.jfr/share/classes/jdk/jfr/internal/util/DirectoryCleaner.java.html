25 package jdk.jfr.internal.util;
26 
27 import java.io.IOException;
28 import java.nio.file.FileVisitResult;
29 import java.nio.file.Files;
30 import java.nio.file.Path;
31 import java.nio.file.SimpleFileVisitor;
32 import java.nio.file.attribute.BasicFileAttributes;
33 
34 public final class DirectoryCleaner extends SimpleFileVisitor<Path> {
35 
36     public static void clear(Path path) throws IOException {
37         Files.walkFileTree(path, new DirectoryCleaner());
38     }
39 
40     @Override
41     public FileVisitResult visitFile(Path path, BasicFileAttributes attrs) throws IOException {
42         Files.delete(path);
43         return FileVisitResult.CONTINUE;
44     }
45 
46     @Override
47     public FileVisitResult postVisitDirectory(Path dir, IOException exc) throws IOException {
48         if (exc != null) {
49             throw exc;
50         }
51         Files.delete(dir);
52         return FileVisitResult.CONTINUE;
53     }
54 }
