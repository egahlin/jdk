25 package jdk.jfr.internal;
26 
27 import jdk.jfr.internal.event.EventWriter;
28 
29 // This class is not directly used but renamed to
30 // jdk.jfr.internal.event.EventWriterFactory and loaded dynamically
31 // when the first event class is bytecode instrumented.
32 // See JVMUpcalls and EventWriterKey::ensureEventWriterFactory()
33 public final class EventWriterFactoryRecipe {
34     private static final long KEY = EventWriterKey.getKey();
35 
36     public static EventWriter getEventWriter(long key) {
37         if (key == KEY) {
38             EventWriter ew = JVM.getEventWriter();
39             return ew != null ? ew : JVM.newEventWriter();
40         }
41         EventWriterKey.block();
42         return null; // Can't reach here.
43     }
44 }
