25 package jdk.jfr.internal.periodic;
26 
27 import jdk.internal.event.Event;
28 import jdk.jfr.EventType;
29 import jdk.jfr.internal.MetadataRepository;
30 import jdk.jfr.internal.PlatformEventType;
31 import jdk.jfr.internal.PrivateAccess;
32 /**
33  * Base class for periodic Java events.
34  */
35 abstract class JavaEventTask extends EventTask {
36     private final Runnable runnable;
37 
38     public JavaEventTask(Class<? extends Event> eventClass, Runnable runnable) {
39         super(toPlatformEventType(eventClass), new LookupKey(runnable));
40         this.runnable = runnable;
41         if (getEventType().isJVM()) {
42             throw new InternalError("Must not be a JVM event");
43         }
44     }
45 
46     private static PlatformEventType toPlatformEventType(Class<? extends Event> eventClass) {
47         EventType eventType = MetadataRepository.getInstance().getEventType(eventClass);
48         return PrivateAccess.getInstance().getPlatformEventType(eventType);
49     }
50 
51     protected final Runnable getRunnable() {
52         return runnable;
53     }
54 }
