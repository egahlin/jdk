 27 package jdk.jfr.internal.settings;
 28 
 29 import static jdk.jfr.internal.util.TimespanUnit.SECONDS;
 30 import static jdk.jfr.internal.util.TimespanUnit.MILLISECONDS;
 31 
 32 import java.util.Objects;
 33 import java.util.Set;
 34 
 35 import jdk.jfr.Description;
 36 import jdk.jfr.Label;
 37 import jdk.jfr.MetadataDefinition;
 38 import jdk.jfr.Name;

 39 import jdk.jfr.internal.PlatformEventType;
 40 import jdk.jfr.internal.Throttle;
 41 import jdk.jfr.internal.Type;
 42 import jdk.jfr.internal.util.Rate;
 43 import jdk.jfr.internal.util.TimespanUnit;
 44 import jdk.jfr.internal.util.Utils;
 45 
 46 @MetadataDefinition
 47 @Label("Throttle")
 48 @Description("Throttles the emission rate for an event")
 49 @Name(Type.SETTINGS_PREFIX + "Throttle")
 50 public final class ThrottleSetting extends JDKSettingControl {
 51     public static final String DEFAULT_VALUE = Throttle.DEFAULT;
 52     private final PlatformEventType eventType;
 53     private String value = DEFAULT_VALUE;
 54 
 55     public ThrottleSetting(PlatformEventType eventType) {
 56        this.eventType = Objects.requireNonNull(eventType);
 57     }
 58 
 59     @Override
 60     public String combine(Set<String> values) {
 61         Rate max = null;
 62         String text = null;
 63         for (String value : values) {
 64             Rate rate = Rate.of(value);
 65             if (rate != null) {
 66                 if (max == null || rate.isHigher(max)) {
 67                     text = value;
 68                     max = rate;
 69                 }
 70             }
 71         }
 72         // "off" is default
 73         return Objects.requireNonNullElse(text, DEFAULT_VALUE);
 74     }
 75 
 76     @Override
 77     public void setValue(String value) {
 78         if ("off".equals(value)) {
 79             eventType.setThrottle(-2, 1000);
 80             this.value = value;
 81             return;
 82         }
 83 
 84         Rate rate = Rate.of(value);
 85         if (rate != null) {
 86             long millis = 1000;
 87             long samples = rate.amount();
 88             TimespanUnit unit = rate.unit();
 89             // if unit is more than 1 s, set millis
 90             if (unit.nanos > SECONDS.nanos) {
 91                 millis = unit.nanos / MILLISECONDS.nanos;
 92             }
 93             // if unit is less than 1 s, scale samples
 94             if (unit.nanos < SECONDS.nanos) {
 95                 long perSecond = SECONDS.nanos / unit.nanos;
 96                 samples *= Utils.multiplyOverflow(samples, perSecond, Long.MAX_VALUE);
 97             }
 98             eventType.setThrottle(samples, millis);
 99             this.value = value;
100         }
101     }
102 
103     @Override
104     public String getValue() {
105         return value;
106     }
107 }
