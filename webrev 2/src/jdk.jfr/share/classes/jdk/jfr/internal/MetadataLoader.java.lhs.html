 55     // Caching to reduce allocation pressure and heap usage
 56     private final AnnotationElement RELATIONAL = new AnnotationElement(Relational.class);
 57     private final AnnotationElement ENABLED = new AnnotationElement(Enabled.class, true);
 58     private final AnnotationElement DISABLED = new AnnotationElement(Enabled.class, false);
 59     private final AnnotationElement THRESHOLD = new AnnotationElement(Threshold.class, "0 ns");
 60     private final AnnotationElement STACK_TRACE = new AnnotationElement(StackTrace.class, true);
 61     private final AnnotationElement TRANSITION_TO = new AnnotationElement(TransitionTo.class);
 62     private final AnnotationElement TRANSITION_FROM = new AnnotationElement(TransitionFrom.class);
 63     private final AnnotationElement EXPERIMENTAL = new AnnotationElement(Experimental.class);
 64     private final AnnotationElement UNSIGNED = new AnnotationElement(Unsigned.class);
 65     private final List<Object> SMALL_TEMP_LIST = new ArrayList<>();
 66     private final Type LABEL_TYPE = TypeLibrary.createAnnotationType(Label.class);
 67     private final Type DESCRIPTION_TYPE = TypeLibrary.createAnnotationType(Description.class);
 68     private final Type CATEGORY_TYPE = TypeLibrary.createAnnotationType(Category.class);

 71     // <Event>, <Type> and <Relation>
 72     private static final class TypeElement {
 73         private final List<FieldElement> fields;
 74         private final String name;
 75         private final String label;
 76         private final String description;
 77         private final String category;
 78         private final String period;
 79         private final boolean thread;
 80         private final boolean startTime;
 81         private final boolean stackTrace;
 82         private final boolean cutoff;
 83         private final boolean throttle;
 84         private final String level;
 85         private final boolean isEvent;
 86         private final boolean isRelation;
 87         private final boolean experimental;
 88         private final boolean internal;
 89         private final long id;
 90 
 91         public TypeElement(DataInputStream dis) throws IOException {
 92             int fieldCount = dis.readInt();
 93             fields = new ArrayList<>(fieldCount);
 94             for (int i = 0; i < fieldCount; i++) {
 95                 fields.add(new FieldElement(dis));
 96             }
 97             name = dis.readUTF();
 98             label = dis.readUTF();
 99             description = dis.readUTF();
100             category = dis.readUTF();
101             thread = dis.readBoolean();
102             stackTrace = dis.readBoolean();
103             startTime = dis.readBoolean();
104             period = dis.readUTF();
105             cutoff = dis.readBoolean();
106             throttle = dis.readBoolean();
107             level = dis.readUTF();
108             experimental = dis.readBoolean();
109             internal = dis.readBoolean();
110             id = dis.readLong();
111             isEvent = dis.readBoolean();
112             isRelation = dis.readBoolean();
113         }
114     }

117     private static final class FieldElement {
118         private final String name;
119         private final String label;
120         private final String description;
121         private final String typeName;
122         private final String annotations;
123         private final String transition;
124         private final String relation;
125         private final boolean constantPool;
126         private final boolean array;
127         private final boolean experimental;
128         private final boolean unsigned;
129 
130         public FieldElement(DataInputStream dis) throws IOException {
131             name = dis.readUTF();
132             typeName = dis.readUTF();
133             label = dis.readUTF();
134             description = dis.readUTF();
135             constantPool = dis.readBoolean();
136             array = dis.readBoolean();
137             unsigned = dis.readBoolean();
138             annotations = dis.readUTF();
139             transition = dis.readUTF();
140             relation = dis.readUTF();
141             experimental = dis.readBoolean();
142         }

145     private final List<TypeElement> types;
146     private final Map<String, List<AnnotationElement>> anotationElements = HashMap.newHashMap(16);
147     private final Map<String, AnnotationElement> categories = new HashMap<>();
148 
149     MetadataLoader(DataInputStream dis) throws IOException {
150         SMALL_TEMP_LIST.add(this); // add any object to expand list
151         int typeCount = dis.readInt();
152         types = new ArrayList<>(typeCount);
153         for (int i = 0; i < typeCount; i++) {
154             types.add(new TypeElement(dis));
155         }
156     }

158     private List<AnnotationElement> createAnnotationElements(String annotation) throws InternalError {
159         String[] annotations = annotation.split(",");
160         List<AnnotationElement> annotationElements = new ArrayList<>();
161         for (String a : annotations) {
162             a = a.trim();
163             int leftParenthesis = a.indexOf("(");
164             if (leftParenthesis == -1) {
165                 annotationElements.add(new AnnotationElement(createAnnotationClass(a)));
166             } else {
167                 int rightParenthesis = a.lastIndexOf(")");
168                 if (rightParenthesis == -1) {
169                     throw new InternalError("Expected closing parenthesis for 'XMLContentType'");
170                 }
171                 String value = a.substring(leftParenthesis + 1, rightParenthesis);
172                 String type = a.substring(0, leftParenthesis);
173                 annotationElements.add(new AnnotationElement(createAnnotationClass(type), value));
174             }
175         }
176         return annotationElements;
177     }
178 
179     @SuppressWarnings("unchecked")
180     private Class<? extends Annotation> createAnnotationClass(String type) {
181         try {
182             if (!type.startsWith("jdk.jfr.")) {
183                 throw new IllegalStateException("Incorrect type " + type + ". Annotation class must be located in jdk.jfr package.");
184             }
185             Class<?> c = Class.forName(type, true, null);
186             return (Class<? extends Annotation>) c;
187         } catch (ClassNotFoundException cne) {
188             throw new IllegalStateException(cne);
189         }
190     }

192     public static List<Type> createTypes() throws IOException {
193         try (DataInputStream dis = new DataInputStream(
194                 SecuritySupport.getResourceAsStream("/jdk/jfr/internal/types/metadata.bin"))) {
195             MetadataLoader ml = new MetadataLoader(dis);
196             return ml.buildTypes();
197         } catch (Exception e) {
198             throw new InternalError(e);
199         }
200     }

202     private List<Type> buildTypes() {
203         Map<String, Type> typeMap = buildTypeMap();
204         Map<String, AnnotationElement> relationMap = buildRelationMap(typeMap);
205         addFields(typeMap, relationMap);
206         return new ArrayList<>(typeMap.values());
207     }
208 
209     private Map<String, AnnotationElement> buildRelationMap(Map<String, Type> typeMap) {
210         Map<String, AnnotationElement> relationMap = HashMap.newHashMap(10);
211         for (TypeElement t : types) {
212             if (t.isRelation) {
213                 Type relationType = typeMap.get(t.name);
214                 AnnotationElement ae = PrivateAccess.getInstance().newAnnotation(relationType, Collections.emptyList(), true);
215                 relationMap.put(t.name, ae);
216             }
217         }
218         return relationMap;
219     }
220 
221     private void addFields(Map<String, Type> lookup, Map<String, AnnotationElement> relationMap) {
222         for (TypeElement te : types) {
223             Type type = lookup.get(te.name);
224             if (te.isEvent) {
225                 boolean periodic = !te.period.isEmpty();
226                 TypeLibrary.addImplicitFields(type, periodic, te.startTime && !periodic, te.thread, te.stackTrace && !periodic, te.cutoff);
227             }
228             for (FieldElement f : te.fields) {
229                 Type fieldType = Type.getKnownType(f.typeName);
230                 if (fieldType == null) {
231                     fieldType = Objects.requireNonNull(lookup.get(f.typeName));
232                 }
233                 List<AnnotationElement> aes = new ArrayList<>();
234                 if (f.unsigned) {
235                     aes.add(UNSIGNED);
236                 }
237                 if (!f.annotations.isEmpty()) {
238                     var ae = anotationElements.get(f.annotations);
239                     if (ae == null) {
240                         ae = createAnnotationElements(f.annotations);
241                         anotationElements.put(f.annotations, ae);
242                     }
243                     aes.addAll(ae);
244                 }
245                 if (!f.relation.isEmpty()) {
246                     AnnotationElement t = relationMap.get(f.relation);
247                     aes.add(Objects.requireNonNull(t));
248                 }
249                 if (!f.label.isEmpty()) {
250                     aes.add(newAnnotation(LABEL_TYPE, f.label));
251                 }
252                 if (f.experimental) {
253                     aes.add(EXPERIMENTAL);
254                 }
255                 if (!f.description.isEmpty()) {
256                     aes.add(newAnnotation(DESCRIPTION_TYPE, f.description));
257                 }
258                 if ("from".equals(f.transition)) {
259                     aes.add(TRANSITION_FROM);
260                 }
261                 if ("to".equals(f.transition)) {
262                     aes.add(TRANSITION_TO);
263                 }
264                 if (!"package".equals(f.name) && !"java.lang.Class".equals(te.name)) {
265                     Utils.ensureJavaIdentifier(f.name);
266                 }
267                 type.add(PrivateAccess.getInstance().newValueDescriptor(f.name, fieldType, aes, f.array ? 1 : 0, f.constantPool, null));
268             }
269         }
270     }

272     private AnnotationElement newAnnotation(Type type, Object value) {
273         SMALL_TEMP_LIST.set(0, value);
274         return PrivateAccess.getInstance().newAnnotation(type, SMALL_TEMP_LIST, true);
275     }

277     private Map<String, Type> buildTypeMap() {
278         Map<String, Type> typeMap = HashMap.newHashMap(types.size());
279         Map<String, Type> knownTypeMap = HashMap.newHashMap(16);
280         for (Type kt : Type.getKnownTypes()) {
281             typeMap.put(kt.getName(), kt);
282             knownTypeMap.put(kt.getName(), kt);
283         }
284         for (TypeElement t : types) {
285             List<AnnotationElement> aes = new ArrayList<>();
286             if (!t.category.isEmpty()) {
287                 AnnotationElement cat = categories.get(t.category);
288                 if (cat == null) {
289                     String[] segments = buildCategorySegments(t.category);
290                     cat = newAnnotation(CATEGORY_TYPE, segments);
291                     categories.put(t.category, cat);
292                 }
293                 aes.add(cat);
294             }
295             if (!t.label.isEmpty()) {
296                 aes.add(newAnnotation(LABEL_TYPE, t.label));
297             }
298             if (!t.description.isEmpty()) {
299                 aes.add(newAnnotation(DESCRIPTION_TYPE, t.description));
300             }
301             if (t.isEvent) {
302                 if (!t.period.isEmpty()) {
303                     aes.add(newAnnotation(PERIOD_TYPE, t.period));
304                 } else {
305                     if (t.startTime) {
306                         aes.add(THRESHOLD);
307                     }
308                     if (t.stackTrace) {
309                         aes.add(STACK_TRACE);
310                     }
311                 }
312                 if (!t.level.isEmpty()) {
313                     String[] levels = t.level.split(",");
314                     for (int i = 0; i < levels.length; i++) {
315                         levels[i] = levels[i].strip();
316                     }
317                     aes.add(new AnnotationElement(Level.class, levels));
318                 }
319                 if (t.cutoff) {
320                     aes.add(new AnnotationElement(Cutoff.class, Cutoff.INFINITY));
321                 }
322                 if (t.throttle) {
323                     aes.add(new AnnotationElement(Throttle.class, Throttle.DEFAULT));
324                 }
325             }
326             if (t.experimental) {
327                 aes.add(EXPERIMENTAL);
328             }
329             Type type;
330             if (t.isEvent) {
331                 if (t.internal) {
332                     aes.add(ENABLED);
333                 } else {
334                     aes.add(DISABLED);
335                 }
336                 type = new PlatformEventType(t.name, t.id, false, true);
337             } else {
338                 type = knownTypeMap.get(t.name);
339                 if (type == null) {
340                     if (t.isRelation) {
341                         type = new Type(t.name, Type.SUPER_TYPE_ANNOTATION, t.id);
342                         aes.add(RELATIONAL);
343                     } else {
344                         type = new Type(t.name, null, t.id);
345                     }
346                 }
347             }
348             if (t.internal) {
349                 type.setInternal(true);
350                 // Internal types are hidden by default
351                 type.setVisible(false);
352                 // Internal events are enabled by default
353                 if (type instanceof PlatformEventType pe) {
354                     pe.setEnabled(true);
355                 }
356             }
357             type.setAnnotations(aes);
358             typeMap.put(t.name, type);
359         }
360         return typeMap;
361     }
362 
363     private String[] buildCategorySegments(String category) {
364         String[] segments = category.split(",");
365         for (int i = 0; i < segments.length; i++) {
366             segments[i] = segments[i].trim();
367         }
368         return segments;
369     }
