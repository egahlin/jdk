 36 // User code should not be able to get access to an EventWriter instance as it
 37 // would allow it to write arbitrary data into buffers, potentially from
 38 // different threads.
 39 //
 40 // This is prevented in three ways:
 41 //
 42 // 1. For code to access the jdk.jfr.internal.event package
 43 //    at least one event class (for a particular module) must be
 44 //    registered having FlightRecorderPermission("registerEvent").
 45 //
 46 // 2. The EventWriter EventWriterFactory::getEventWriter(long) method can only be linked from
 47 //    the UserEvent::commit() method instrumented by JFR. This is ensured by the JVM.
 48 //    (The EventWriterFactory class is dynamically generated before the first event
 49 //    is instrumented. See EventWriterFactoryRecipe)
 50 //
 51 // 3. Steps 1 and 2 are sufficient to make it fully secure, with or without a Security
 52 //    Manager, but as an additional measure, the method EventWriterFactory::getEventWriter(long)
 53 //    requires the caller to provide a key that is hard to guess. The key is generated
 54 //    into the bytecode of the method invoking getEventWriter(long).
 55 //
 56 public final class EventWriter {
 57 
 58     // Event may not exceed size for a padded integer
 59     private static final long MAX_EVENT_SIZE = (1 << 28) -1;
 60     private static final Unsafe unsafe = Unsafe.getUnsafe();
 61 
 62     // The JVM needs access to these values. Don't remove
 63     private final long threadID;
 64     private long startPosition;
 65     private long currentPosition;
 66     private long maxPosition;
 67     private boolean valid;
 68     private boolean pinVirtualThread;
 69     boolean excluded;
 70 
 71     private PlatformEventType eventType;
 72     private boolean largeSize = false;
 73 
 74     // User code must not be able to instantiate
 75     private EventWriter() {
 76         threadID = 0;
 77     }
 78 
 79     public void putBoolean(boolean i) {
 80         if (isValidForSize(Byte.BYTES)) {
 81             unsafe.putBoolean(null, currentPosition, i);
 82             ++currentPosition;
 83         }
 84     }
 85 
 86     public void putByte(byte i) {
 87         if (isValidForSize(Byte.BYTES)) {
 88             unsafe.putByte(currentPosition, i);
 89             ++currentPosition;
 90         }
 91     }
 92 
 93     public void putChar(char v) {
 94         if (isValidForSize(Character.BYTES + 1)) {
 95             putUncheckedLong(v);
 96         }
 97     }
 98 
 99     private void putUncheckedChar(char v) {
100         putUncheckedLong(v);
101     }
102 
103     public void putShort(short v) {
104         if (isValidForSize(Short.BYTES + 1)) {
105             putUncheckedLong(v & 0xFFFF);
106         }
107     }
108 
109     public void putInt(int v) {
110         if (isValidForSize(Integer.BYTES + 1)) {
111             putUncheckedLong(v & 0x00000000ffffffffL);
112         }
113     }
114 
115     private void putUncheckedInt(int v) {
116         putUncheckedLong(v & 0x00000000ffffffffL);
117     }
118 
119     public void putFloat(float i) {
120         if (isValidForSize(Float.BYTES)) {
121             currentPosition += Bits.putFloat(currentPosition, i);
122         }
123     }
124 
125     public void putLong(long v) {
126         if (isValidForSize(Long.BYTES + 1)) {
127             putUncheckedLong(v);
128         }
129     }
130 
131     public void putDouble(double i) {
132         if (isValidForSize(Double.BYTES)) {
133             currentPosition += Bits.putDouble(currentPosition, i);
134         }
135     }
136 
137     public void putString(String s) {
138         if (s == null) {
139             putByte(StringParser.Encoding.NULL.byteValue());
140             return;
141         }
142         int length = s.length();
143         if (length == 0) {
144             putByte(StringParser.Encoding.EMPTY_STRING.byteValue());
145             return;
146         }
147         if (length > StringPool.MIN_LIMIT && length < StringPool.MAX_LIMIT) {
148             long l = StringPool.addString(s, pinVirtualThread);
149             if (l > 0) {
150                 putByte(StringParser.Encoding.CONSTANT_POOL.byteValue());
151                 putLong(l);
152                 return;
153             }
154         }
155         putStringValue(s);
156         return;
157     }
158 
159     private void putStringValue(String s) {
160         int length = s.length();
161         if (isValidForSize(1 + 5 + 3 * length)) {
162             putUncheckedByte(StringParser.Encoding.CHAR_ARRAY.byteValue()); // 1 byte
163             putUncheckedInt(length); // max 5 bytes
164             for (int i = 0; i < length; i++) {
165                 putUncheckedChar(s.charAt(i)); // max 3 bytes
166             }
167         }
168     }
169 
170     public void putEventThread() {
171         putLong(threadID);
172     }
173 
174     public void putThread(Thread athread) {
175         if (athread == null) {
176             putLong(0L);
177         } else {
178             putLong(JVM.getThreadId(athread));
179         }
180     }
181 
182     public void putClass(Class<?> aClass) {
183         if (aClass == null) {
184             putLong(0L);
185         } else {
186             putLong(JVM.getClassId(aClass));
187         }
188     }
189 
190     public void putStackTrace() {
191         if (eventType.getStackTraceEnabled()) {
192             putLong(JVM.getStackTraceId(eventType.getStackTraceOffset(), eventType.getStackFilterId()));
193         } else {
194             putLong(0L);
195         }
196     }
197 
198     private void reserveEventSizeField() {
199         this.largeSize = eventType.isLargeSize();
200         if (largeSize) {
201             if (isValidForSize(Integer.BYTES)) {
202                 currentPosition +=  Integer.BYTES;
203             }
204         } else {
205             if (isValidForSize(1)) {
206                 currentPosition += 1;
207             }
208         }
209     }
210 
211     public void reset() {
212         currentPosition = startPosition;
213     }
214 
215     private boolean isValidForSize(int requestedSize) {
216         if (!valid) {
217             return false;
218         }
219         if (currentPosition + requestedSize > maxPosition) {
220             flush(usedSize(), requestedSize);
221             // retry
222             if (!valid) {
223                 return false;
224             }
225         }
226         return true;
227     }
228 
229     private int usedSize() {
230         return (int) (currentPosition - startPosition);
231     }
232 
233     private void flush() {
234         flush(usedSize(), 0);
235     }
236 
237     private void flush(int usedSize, int requestedSize) {
238         JVM.flush(this, usedSize, requestedSize);
239     }
240 
241     public boolean beginEvent(EventConfiguration configuration, long typeId) {
242         // Malicious code could take the EventConfiguration object from one
243         // event class field and assign it to another. This check makes sure
244         // the event type matches what was added by instrumentation.
245         if (configuration.getId() != typeId) {
246             EventWriterKey.block();
247         }
248         if (excluded) {
249             // thread is excluded from writing events
250             return false;
251         }
252         this.eventType = configuration.getPlatformEventType();
253         reserveEventSizeField();
254         putLong(eventType.getId());
255         return true;
256     }
257 
258     public boolean endEvent() {
259         if (!valid) {
260             reset();
261             valid = true;
262             return true;
263         }
264         final int eventSize = usedSize();
265         if (eventSize > MAX_EVENT_SIZE) {
266             reset();
267             return true;
268         }
269         if (largeSize) {
270             Bits.putInt(startPosition, makePaddedInt(eventSize));
271         } else {
272             if (eventSize < 128) {
273                 unsafe.putByte(startPosition, (byte) eventSize);
274             } else {
275                 eventType.setLargeSize();
276                 reset();
277                 // returning false will trigger restart of the
278                 // event write attempt
279                 return false;
280             }
281         }
282         long nextPosition = JVM.commit(currentPosition);
283         if (nextPosition == currentPosition) {
284             // Successful commit. Update the writer start position.
285             startPosition = nextPosition;
286             return true;
287         }
288         // If nextPosition == 0, the event was committed, the underlying buffer lease
289         // returned and new writer positions updated. Nothing to do.
290         if (nextPosition == 0) {
291             return true;
292         }
293         // The commit was aborted because of an interleaving epoch shift.
294         // The nextPosition returned is the current start position.
295         // Reset the writer and return false to restart the write attempt.
296         currentPosition = nextPosition;
297         return false;
298     }

300     private EventWriter(long startPos, long maxPos, long threadID, boolean valid, boolean pinVirtualThread, boolean excluded) {
301         startPosition = currentPosition = startPos;
302         maxPosition = maxPos;
303         this.threadID = threadID;
304         this.valid = valid;
305         this.pinVirtualThread = pinVirtualThread;
306         this.excluded = excluded;
307     }

309     private static int makePaddedInt(int v) {
310         // bit  0-6 + pad => bit 24 - 31
311         long b1 = (((v >>> 0) & 0x7F) | 0x80) << 24;
312 
313         // bit  7-13 + pad => bit 16 - 23
314         long b2 = (((v >>> 7) & 0x7F) | 0x80) << 16;
315 
316         // bit 14-20 + pad => bit  8 - 15
317         long b3 = (((v >>> 14) & 0x7F) | 0x80) << 8;
318 
319         // bit 21-28       => bit  0 -  7
320         long b4 = (((v >>> 21) & 0x7F)) << 0;
321 
322         return (int) (b1 + b2 + b3 + b4);
323     }
324 
325     private void putUncheckedLong(long v) {
326         if ((v & ~0x7FL) == 0L) {
327             putUncheckedByte((byte) v); // 0-6
328             return;
329         }
330         putUncheckedByte((byte) (v | 0x80L)); // 0-6
331         v >>>= 7;
332         if ((v & ~0x7FL) == 0L) {
333             putUncheckedByte((byte) v); // 7-13
334             return;
335         }
336         putUncheckedByte((byte) (v | 0x80L)); // 7-13
337         v >>>= 7;
338         if ((v & ~0x7FL) == 0L) {
339             putUncheckedByte((byte) v); // 14-20
340             return;
341         }
342         putUncheckedByte((byte) (v | 0x80L)); // 14-20
343         v >>>= 7;
344         if ((v & ~0x7FL) == 0L) {
345             putUncheckedByte((byte) v); // 21-27
346             return;
347         }
348         putUncheckedByte((byte) (v | 0x80L)); // 21-27
349         v >>>= 7;
350         if ((v & ~0x7FL) == 0L) {
351             putUncheckedByte((byte) v); // 28-34
352             return;
353         }
354         putUncheckedByte((byte) (v | 0x80L)); // 28-34
355         v >>>= 7;
356         if ((v & ~0x7FL) == 0L) {
357             putUncheckedByte((byte) v); // 35-41
358             return;
359         }
360         putUncheckedByte((byte) (v | 0x80L)); // 35-41
361         v >>>= 7;
362         if ((v & ~0x7FL) == 0L) {
363             putUncheckedByte((byte) v); // 42-48
364             return;
365         }
366         putUncheckedByte((byte) (v | 0x80L)); // 42-48
367         v >>>= 7;
368 
369         if ((v & ~0x7FL) == 0L) {
370             putUncheckedByte((byte) v); // 49-55
371             return;
372         }
373         putUncheckedByte((byte) (v | 0x80L)); // 49-55
374         putUncheckedByte((byte) (v >>> 7)); // 56-63, last byte as is.
375     }
376 
377     private void putUncheckedByte(byte i) {
378         unsafe.putByte(currentPosition, i);
379         ++currentPosition;
380     }
