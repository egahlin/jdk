26 package jdk.jfr.internal.settings;

28 import static jdk.jfr.internal.util.ValueParser.MISSING;
29 
30 import java.util.Objects;
31 import java.util.Set;
32 
33 import jdk.jfr.Description;

34 import jdk.jfr.Label;
35 import jdk.jfr.MetadataDefinition;
36 import jdk.jfr.Name;
37 import jdk.jfr.Timespan;
38 import jdk.jfr.internal.PlatformEventType;
39 import jdk.jfr.internal.Type;
40 import jdk.jfr.internal.util.ValueParser;
41 
42 @MetadataDefinition
43 @Label("Cutoff")
44 @Description("Limit running time of event")
45 @Name(Type.SETTINGS_PREFIX + "Cutoff")
46 @Timespan
47 public final class CutoffSetting extends JDKSettingControl {
48     public static final String DEFAULT_VALUE = ValueParser.INFINITY;
49     private String value = DEFAULT_VALUE;
50     private final PlatformEventType eventType;
51 
52     public CutoffSetting(PlatformEventType eventType) {
53        this.eventType = Objects.requireNonNull(eventType);
54     }
55 
56     @Override
57     public String combine(Set<String> values) {
58         long max = 0;
59         String text = null;
60         for (String value : values) {
61             long nanos = ValueParser.parseTimespanWithInfinity(value, MISSING);
62             if (nanos != MISSING && nanos > max) {
63                 text = value;
64                 max = nanos;
65             }
66         }
67         return Objects.requireNonNullElse(text, DEFAULT_VALUE);
