 25 package jdk.jfr.internal.periodic;
 26 
 27 import java.security.AccessControlContext;
 28 import java.util.ArrayList;
 29 import java.util.List;
 30 import java.util.concurrent.atomic.AtomicLong;
 31 
 32 import jdk.internal.event.Event;
 33 import jdk.jfr.internal.JVM;
 34 import jdk.jfr.internal.LogLevel;
 35 import jdk.jfr.internal.LogTag;
 36 import jdk.jfr.internal.Logger;
 37 import jdk.jfr.internal.PlatformEventType;
 38 
 39 /**
 40  * Class that runs and schedules tasks for periodic events.
 41  * <p>
 42  * Events can run at the beginning of a chunk rotation, at the end of a chunk
 43  * rotation or at a periodic interval.
 44  * <p>
 45  * Events with the same period runs in the same order as users added them.
 46  * Temporarily disabling events don't impact the order of execution.
 47  * <p>
 48  * A best effort is made to run events with different periods at the same time.
 49  * For example, an event that runs every two seconds executes half of the time
 50  * with the events that run every second.
 51  */
 52 public final class PeriodicEvents {
 53     private static final TaskRepository taskRepository = new TaskRepository();
 54     private static final BatchManager batchManager = new BatchManager();
 55     private static final FlushTask flushTask = new FlushTask();
 56     private static final AtomicLong settingsIteration = new AtomicLong();
 57 
 58     // State only to be read and modified by periodic task thread
 59     private static long lastTimeMillis;
 60 
 61     public static void addJDKEvent(Class<? extends Event> eventClass, Runnable runnable) {
 62         taskRepository.add(new JDKEventTask(eventClass, runnable));
 63     }
 64 
 65     public static void addJVMEvent(PlatformEventType eventType) {
 66         taskRepository.add(new JVMEventTask(eventType));
 67     }
 68 
 69     public static void addUserEvent(@SuppressWarnings("removal") AccessControlContext acc, Class<? extends Event> eventClass, Runnable runnable) {
 70         taskRepository.add(new UserEventTask(acc, eventClass, runnable));
 71     }
 72 
 73     public static boolean removeEvent(Runnable runnable) {
 74         return taskRepository.removeTask(runnable);
 75     }
 76 
 77     public static void doChunkBegin() {
 78         long timestamp = JVM.counterTime();
 79         for (EventTask task : taskRepository.getTasks()) {
 80             var eventType = task.getEventType();
 81             if (eventType.isEnabled() && eventType.isBeginChunk()) {
 82                 task.run(timestamp, PeriodicType.BEGIN_CHUNK);
 83             }
 84         }
 85     }
 86 
 87     public static void doChunkEnd() {
 88         long timestamp = JVM.counterTime();
 89         for (EventTask task : taskRepository.getTasks()) {
 90             var eventType = task.getEventType();
 91             if (eventType.isEnabled() && eventType.isEndChunk()) {
 92                 task.run(timestamp, PeriodicType.END_CHUNK);
 93             }
 94         }
 95     }
 96 
 97     // Only to be called from periodic task thread
 98     public static long doPeriodic() {
 99         long eventTimestamp = JVM.counterTime();
100         // Code copied from prior native implementation
101         long last = lastTimeMillis;
102         // The interval for periodic events is typically at least 1 s, so
103         // System.currentTimeMillis() is sufficient. JVM.counterTime() lacks
104         // unit and has in the past been more unreliable.
105         long now = System.currentTimeMillis();
106         long min = 0;
107         long delta = 0;
108 
109         if (last == 0) {
110             last = now;
111         }
112 
113         // time from then to now
114         delta = now - last;
115         if (delta < 0) {
116             // to handle time adjustments
117             // for example Daylight Savings
118             lastTimeMillis = now;
119             return 0;
120         }
121         long iteration = settingsIteration.get();
122         if (iteration > batchManager.getIteration()) {
123             List<PeriodicTask> tasks = new ArrayList<>();
124             tasks.addAll(taskRepository.getTasks());
125             tasks.add(flushTask);
126             batchManager.refresh(iteration, tasks);
127 
128         }
129         boolean flush = false;
130         Logger.log(LogTag.JFR_PERIODIC, LogLevel.DEBUG,"Periodic work started");
131         for (Batch batch : batchManager.getBatches()) {
132             long left = 0;
133             long r_period = batch.getPeriod();
134             long r_delta = batch.getDelta();
135 
136             // add time elapsed.
137             r_delta += delta;
138 
139             // above threshold?
140             if (r_delta >= r_period) {
141                 // Bug 9000556 - don't try to compensate
142                 // for wait > period
143                 r_delta = 0;
144                 for (PeriodicTask task : batch.getTasks()) {
145                     task.tick();
146                     if (task.shouldRun()) {
147                         if (task instanceof FlushTask) {
148                             flush = true;
149                         } else {
150                             task.run(eventTimestamp, PeriodicType.INTERVAL);
151                         }
152                     }
153                 }
154             }
155 
156             // calculate time left
157             left = (r_period - r_delta);
158 
159             /*
160              * nothing outside checks that a period is >= 0, so left can end up negative
161              * here. ex. (r_period =(-1)) - (r_delta = 0) if it is, handle it.
162              */
163             if (left < 0) {
164                 left = 0;
165             }
166 
167             // assign delta back
168             batch.setDelta(r_delta);
169 
170             if (min == 0 || left < min) {
171                 min = left;
172             }
173         }
174         if (flush) {
175             flushTask.run(eventTimestamp, PeriodicType.INTERVAL);
176         }
177         Logger.log(LogTag.JFR_PERIODIC, LogLevel.DEBUG,"Periodic work ended");
178         lastTimeMillis = now;
179         return min;
180     }
181 
182     /**
183      * Marks that a change has happened to a periodic event.
184      * <p>
185      * This method should be invoked if a periodic event has:
186      * <ul>
187      * <li>been added</li>
188      * <li>been removed</li>
189      * <li>been enabled
190      * <li>been disabled</li>
191      * <li>changed period</li>
192      * </ul>
193      * <p>
194      * The periodic task thread will poll the changed state at least once every
195      * second to see if a change has occurred. if that's the case, it will refresh
196      * periodic tasks that need to be run.
197      */
198     public static void setChanged() {
199         settingsIteration.incrementAndGet();
200     }
201 
202     public static void setFlushInterval(long millis) {
203         flushTask.setInterval(millis);
204     }
205 }
