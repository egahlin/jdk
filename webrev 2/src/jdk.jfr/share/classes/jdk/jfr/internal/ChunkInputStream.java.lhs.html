 26 package jdk.jfr.internal;
 27 
 28 import java.io.BufferedInputStream;
 29 import java.io.IOException;
 30 import java.io.InputStream;

 31 import java.util.ArrayList;
 32 import java.util.Iterator;
 33 import java.util.List;
 34 import java.util.Objects;
 35 
 36 final class ChunkInputStream extends InputStream {
 37     private final Iterator<RepositoryChunk> chunks;
 38     private long unstreamedSize = 0;
 39     private RepositoryChunk currentChunk;
 40     private InputStream stream;
 41 
 42     ChunkInputStream(List<RepositoryChunk> chunks) throws IOException {
 43         List<RepositoryChunk> l = new ArrayList<>(chunks.size());
 44         for (RepositoryChunk c : chunks) {
 45             c.use(); // keep alive while we're reading.
 46             l.add(c);
 47             unstreamedSize += c.getSize();
 48         }
 49 
 50         this.chunks = l.iterator();
 51         nextStream();
 52     }
 53 
 54     @Override
 55     public int available() throws IOException {
 56         long total = unstreamedSize;
 57         if (stream != null) {
 58             total += stream.available();
 59         }
 60         return total <= Integer.MAX_VALUE ? (int) total : Integer.MAX_VALUE;
 61     }
 62 
 63     private boolean nextStream() throws IOException {
 64         if (!nextChunk()) {
 65             return false;
 66         }
 67 
 68         stream = new BufferedInputStream(SecuritySupport.newFileInputStream(currentChunk.getFile()));
 69         unstreamedSize -= currentChunk.getSize();
 70         return true;
 71     }
 72 
 73     private boolean nextChunk() {
 74         if (!chunks.hasNext()) {
 75             return false;
 76         }
 77         currentChunk = chunks.next();
 78         return true;
 79     }
 80 
 81     @Override
 82     public int read() throws IOException {
 83         while (true) {
 84             if (stream != null) {
 85                 int r = stream.read();
 86                 if (r != -1) {
 87                     return r;
 88                 }
 89                 closeStream();
 90             }
 91             if (!nextStream()) {
 92                 return -1;
 93             }
 94         }
 95     }
 96 
 97     @Override
 98     public int read(byte[] buf, int off, int len) throws IOException {
 99         Objects.checkFromIndexSize(off, len, buf.length);
100         if (len == 0) {
101             return 0;
102         }
103 
104         int totalRead = 0;
105         while (len > 0) {
106             if (stream == null) {
107                 closeChunk();
108                 if (!nextStream()) {
109                     return totalRead > 0 ? totalRead : -1;
110                 }
111             }
112             int read = stream.read(buf, off, len);
113             if (read > -1) {
114                 totalRead += read;
115                 len -= read;
116                 if (len == 0) {
117                     return totalRead;
118                 }
119                 off += read;
120             } else {
121                 closeStream();
122             }
123         }
124         return totalRead;
125     }
126 
127     private void closeStream() throws IOException {
128         if (stream != null) {
129             stream.close();
130             stream = null;
131         }
132         closeChunk();
133     }
134 
135     private void closeChunk() {
136         if (currentChunk != null) {
137             currentChunk.release();
138             currentChunk = null;
139         }
140     }
141 
142     @Override
143     public void close() throws IOException {
144         closeStream();
145         while (currentChunk != null) {
146             closeChunk();
147             if (!nextChunk()) {
148                 return;
149             }
150         }
151     }
152 }
