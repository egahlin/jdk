25 package jdk.jfr.internal.periodic;
26 
27 /**
28  * Lookup key that can safely be used in a {@code Map}.
29  * <p>
30  * {@code Runnable} objects can't be used with {@code LinkedHashMap} as it
31  * invokes {@code hashCode} and {@code equals}, for example when resizing the
32  * {@code Map}, possibly in a non-secure context.
33  * <p>
34  * {@code IdentityHashMap} can't be used as it will not preserve order.
35  */
36 final class LookupKey {
37     private final Object object;
38 
39     public LookupKey(Object object) {
40         this.object = object;
41     }
42 
43     public int hashCode() {
44         return System.identityHashCode(object);
45     }
46 
47     public boolean equals(Object that) {
48         if (that instanceof LookupKey lookupKey) {
49             return lookupKey.object == object;
50         }
51         return false;
52     }
53 }
