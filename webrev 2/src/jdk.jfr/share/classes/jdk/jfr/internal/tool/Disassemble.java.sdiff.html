168     private List<Long> findChunkSizes(Path p) throws IOException {
169         try (RecordingInput input = new RecordingInput(p.toFile(), FileAccess.UNPRIVILEGED)) {
170             List<Long> sizes = new ArrayList<>();
171             ChunkHeader ch = new ChunkHeader(input);
172             sizes.add(ch.getSize());
173             while (!ch.isLastChunk()) {
174                 ch = ch.nextHeader();
175                 sizes.add(ch.getSize());
176             }
177             return sizes;
178         }
179     }
