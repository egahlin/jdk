26 package jdk.jfr.internal;
27 
28 import java.io.IOException;


29 import java.util.ArrayList;
30 import java.util.LinkedHashSet;
31 import java.util.SequencedSet;
32 
33 import jdk.jfr.internal.SecuritySupport.SafePath;
34 
35 // This class keeps track of files that can't be deleted
36 // so they can at a later staged be removed.
37 final class FilePurger {
38 
39     private static final SequencedSet<SafePath> paths = new LinkedHashSet<>();
40 
41     public static synchronized void add(SafePath p) {
42         paths.add(p);
43         if (paths.size() > 1000) {
44             removeOldest();
45         }
46     }
47 
48     public static synchronized void purge() {
49         if (paths.isEmpty()) {
50             return;
51         }
52 
53         for (SafePath p : new ArrayList<>(paths)) {
54             if (delete(p)) {
55                 paths.remove(p);
56             }
57         }
58     }
59 
60     private static void removeOldest() {
61         paths.removeFirst();
62     }
63 
64     private static boolean delete(SafePath p) {
65         try {
66             if (!SecuritySupport.exists(p)) {
67                 return true;
68             }
69         } catch (IOException e) {
70             // ignore
71         }
72         try {
73             SecuritySupport.delete(p);
74             return true;
75         } catch (IOException e) {
76             return false;
77         }
78     }
79 }
