 26 package jdk.jfr.internal;
 27 
 28 import java.io.BufferedInputStream;
 29 import java.io.IOException;
 30 import java.io.InputStream;
 31 import java.nio.file.Files;
 32 import java.util.ArrayList;
 33 import java.util.Iterator;
 34 import java.util.List;
 35 import java.util.Objects;
 36 
 37 final class ChunkInputStream extends InputStream {
 38     private final Iterator<RepositoryChunk> chunks;
 39     private long unstreamedSize = 0;
 40     private RepositoryChunk currentChunk;
 41     private InputStream stream;
 42 
 43     ChunkInputStream(List<RepositoryChunk> chunks) throws IOException {
 44         List<RepositoryChunk> l = new ArrayList<>(chunks.size());
 45         for (RepositoryChunk c : chunks) {
 46             c.use(); // keep alive while we're reading.
 47             l.add(c);
 48             unstreamedSize += c.getSize();
 49         }
 50 
 51         this.chunks = l.iterator();
 52         nextStream();
 53     }
 54 
 55     @Override
 56     public int available() throws IOException {
 57         long total = unstreamedSize;
 58         if (stream != null) {
 59             total += stream.available();
 60         }
 61         return total <= Integer.MAX_VALUE ? (int) total : Integer.MAX_VALUE;
 62     }
 63 
 64     private boolean nextStream() throws IOException {
 65         if (!nextChunk()) {
 66             return false;
 67         }
 68 
 69         stream = new BufferedInputStream(Files.newInputStream(currentChunk.getFile()));
 70         unstreamedSize -= currentChunk.getSize();
 71         return true;
 72     }
 73 
 74     private boolean nextChunk() {
 75         if (!chunks.hasNext()) {
 76             return false;
 77         }
 78         currentChunk = chunks.next();
 79         return true;
 80     }
 81 
 82     @Override
 83     public int read() throws IOException {
 84         while (true) {
 85             if (stream != null) {
 86                 int r = stream.read();
 87                 if (r != -1) {
 88                     return r;
 89                 }
 90                 closeStream();
 91             }
 92             if (!nextStream()) {
 93                 return -1;
 94             }
 95         }
 96     }
 97 
 98     @Override
 99     public int read(byte[] buf, int off, int len) throws IOException {
100         Objects.checkFromIndexSize(off, len, buf.length);
101         if (len == 0) {
102             return 0;
103         }
104 
105         int totalRead = 0;
106         while (len > 0) {
107             if (stream == null) {
108                 closeChunk();
109                 if (!nextStream()) {
110                     return totalRead > 0 ? totalRead : -1;
111                 }
112             }
113             int read = stream.read(buf, off, len);
114             if (read > -1) {
115                 totalRead += read;
116                 len -= read;
117                 if (len == 0) {
118                     return totalRead;
119                 }
120                 off += read;
121             } else {
122                 closeStream();
123             }
124         }
125         return totalRead;
126     }
127 
128     private void closeStream() throws IOException {
129         if (stream != null) {
130             stream.close();
131             stream = null;
132         }
133         closeChunk();
134     }
135 
136     private void closeChunk() {
137         if (currentChunk != null) {
138             currentChunk.release();
139             currentChunk = null;
140         }
141     }
142 
143     @Override
144     public void close() throws IOException {
145         closeStream();
146         while (currentChunk != null) {
147             closeChunk();
148             if (!nextChunk()) {
149                 return;
150             }
151         }
152     }
153 }
