 51 /**
 52  * A recording file.
 53  * <p>
 54  * The following example shows how read and print all events in a recording file.
 55  *
 56  * {@snippet class="Snippets" region="RecordingFileOverview"}
 57  *
 58  * @since 9
 59  */
 60 public final class RecordingFile implements Closeable {
 61 
 62     private final ParserState parserState = new ParserState();
 63     private final ChunkWriter chunkWriter;
 64     private boolean isLastEventInChunk;
 65     private final File file;
 66     private RecordingInput input;
 67     private ChunkParser chunkParser;
 68     private RecordedEvent nextEvent;
 69     private boolean eof;
 70 
 71     /**
 72      * Creates a recording file.
 73      * <p>
 74      * Only recording files from trusted sources should be used.
 75      *
 76      * @param file the path of the file to open, not {@code null}
 77      * @throws IOException if it's not a valid recording file, or an I/O error
 78      *         occurred
 79      * @throws NoSuchFileException if the {@code file} can't be located
 80      */
 81     public RecordingFile(Path file) throws IOException {
 82         Objects.requireNonNull(file, "file");
 83         this.file = file.toFile();
 84         this.input = new RecordingInput(this.file, FileAccess.UNPRIVILEGED);
 85         this.chunkWriter = null;
 86         findNext();
 87     }
 88 
 89     // Only used by RecordingFile::write(Path, Predicate<RecordedEvent>)
 90     private RecordingFile(ChunkWriter chunkWriter) throws IOException {
 91         this.file = null; // not used
 92         this.input = chunkWriter.getInput();
 93         this.chunkWriter = chunkWriter;
 94         findNext();
 95     }
 96 
 97     /**
 98      * Reads the next event in the recording.
 99      *
100      * @return the next event, not {@code null}
101      *
102      * @throws EOFException if no more events exist in the recording file
103      * @throws IOException if an I/O error occurs
104      *
105      * @see #hasMoreEvents()
106      */
107     public RecordedEvent readEvent() throws IOException {
108         if (eof) {
109             ensureOpen();
110             throw new EOFException();
111         }
112         isLastEventInChunk = false;
113         RecordedEvent event = nextEvent;
114         nextEvent = chunkParser.readEvent();
115         while (nextEvent == ChunkParser.FLUSH_MARKER) {
116             nextEvent = chunkParser.readEvent();
117         }
118         if (nextEvent == null) {
119             isLastEventInChunk = true;
120             findNext();
121         }
122         return event;
123     }
124 
125     /**
126      * Returns {@code true} if unread events exist in the recording file,
127      * {@code false} otherwise.
128      *
129      * @return {@code true} if unread events exist in the recording, {@code false}
130      *         otherwise.
131      */
132     public boolean hasMoreEvents() {
133         return !eof;
134     }
135 
136     /**
137      * Returns a list of all event types in this recording.
138      *
139      * @return a list of event types, not {@code null}
140      * @throws IOException if an I/O error occurred while reading from the file
141      *
142      * @see #hasMoreEvents()
143      */
144     public List<EventType> readEventTypes() throws IOException {
145         ensureOpen();
146         MetadataDescriptor previous = null;
147         List<EventType> types = new ArrayList<>();
148         HashSet<Long> foundIds = new HashSet<>();
149         try (RecordingInput ri = new RecordingInput(file, FileAccess.UNPRIVILEGED)) {
150             ChunkHeader ch = new ChunkHeader(ri);
151             aggregateEventTypeForChunk(ch, null, types, foundIds);
152             while (!ch.isLastChunk()) {
153                 ch = ch.nextHeader();
154                 previous = aggregateEventTypeForChunk(ch, previous, types, foundIds);
155             }
156         }
157         return types;
158     }
159 
160     List<Type> readTypes() throws IOException  {
161         ensureOpen();
162         MetadataDescriptor previous = null;
163         List<Type> types = new ArrayList<>(200);
164         HashSet<Long> foundIds = HashSet.newHashSet(types.size());
165         try (RecordingInput ri = new RecordingInput(file, FileAccess.UNPRIVILEGED)) {
166             ChunkHeader ch = new ChunkHeader(ri);
167             ch.awaitFinished();
168             aggregateTypeForChunk(ch, null, types, foundIds);
169             while (!ch.isLastChunk()) {
170                 ch = ch.nextHeader();
171                 previous = aggregateTypeForChunk(ch, previous, types, foundIds);
172             }
173         }
174         return types;
175     }
176 
177     private MetadataDescriptor aggregateTypeForChunk(ChunkHeader ch, MetadataDescriptor previous, List<Type> types, HashSet<Long> foundIds) throws IOException {
178         MetadataDescriptor m = ch.readMetadata(previous);
179         for (Type t : m.getTypes()) {
180             if (!foundIds.contains(t.getId())) {
181                 types.add(t);
182                 foundIds.add(t.getId());
183             }
184         }
185         return m;
186     }
187 
188     private static MetadataDescriptor aggregateEventTypeForChunk(ChunkHeader ch,  MetadataDescriptor previous, List<EventType> types, HashSet<Long> foundIds) throws IOException {
189         MetadataDescriptor m = ch.readMetadata(previous);
190         for (EventType t : m.getEventTypes()) {
191             if (!foundIds.contains(t.getId())) {
192                 types.add(t);
193                 foundIds.add(t.getId());
194             }
195         }
196         return m;
197     }
198 
199     /**
200      * Closes this recording file and releases any system resources that are
201      * associated with it.
202      *
203      * @throws IOException if an I/O error occurred
204      */
205     @Override
206     public void close() throws IOException {
207         if (input != null) {
208             eof = true;
209             input.close();
210             chunkParser = null;
211             input = null;
212             nextEvent = null;
213         }
214     }
215 
216     /**
217      * Filter out events and write them to a new file.
218      *
219      * @param destination path where the new file should be written, not
220      *                    {@code null}
221      *
222      * @param filter      filter that determines if an event should be included, not
223      *                    {@code null}
224      * @throws IOException       if an I/O error occurred, it's not a Flight
225      *                           Recorder file or a version of a JFR file that can't
226      *                           be parsed
227      *
228      * @since 19
229      */
230     public void write(Path destination, Predicate<RecordedEvent> filter) throws IOException {
231         Objects.requireNonNull(destination, "destination");
232         Objects.requireNonNull(filter, "filter");
233         try (ChunkWriter cw = new ChunkWriter(file.toPath(), destination, filter)) {
234             try (RecordingFile rf = new RecordingFile(cw)) {
235                 while (rf.hasMoreEvents()) {
236                     rf.readEvent();
237                 }
238             }
239         }
240     }
241 
242     /**
243      * Returns a list of all events in a file.
244      * <p>
245      * This method is intended for simple cases where it's convenient to read all
246      * events in a single operation. It isn't intended for reading large files.
247      * <p>
248      * Only recording files from trusted sources should be used.
249      *
250      * @param path the path to the file, not {@code null}
251      *
252      * @return the events from the file as a {@code List} object; whether the
253      *         {@code List} is modifiable or not is implementation dependent and
254      *         therefore not specified, not {@code null}
255      *
256      * @throws IOException if an I/O error occurred, it's not a Flight Recorder
257      *         file or a version of a JFR file that can't be parsed
258      */
259     public static List<RecordedEvent> readAllEvents(Path path) throws IOException {
260         Objects.requireNonNull(path, "path");
261         try (RecordingFile r = new RecordingFile(path)) {
262             List<RecordedEvent> list = new ArrayList<>();
263             while (r.hasMoreEvents()) {
264                 list.add(r.readEvent());
265             }
266             return list;
267         }
268     }
269 
270     // package protected
271     File getFile() {
272         return file;
273     }
274 
275     // package protected
276     boolean isLastEventInChunk() {
277         return isLastEventInChunk;
278     }
279 
280     // either sets next to an event or sets eof to true
281     private void findNext() throws IOException {
282         while (nextEvent == null) {
283             if (chunkParser == null) {
284                 chunkParser = createChunkParser();
285             } else if (!chunkParser.isLastChunk()) {
286                 chunkParser = nextChunkParser();
287             } else {
288                 endChunkParser();
289                 eof = true;
290                 return;
291             }
292             nextEvent = chunkParser.readEvent();
293             while (nextEvent == ChunkParser.FLUSH_MARKER) {
294                 nextEvent = chunkParser.readEvent();
295             }
296         }
297     }
298 
299     private ChunkParser createChunkParser() throws IOException {
300         if (chunkWriter != null) {
301             boolean reuse = true;
302             boolean ordered = false;
303             ParserConfiguration pc = new ParserConfiguration(0, Long.MAX_VALUE, reuse, ordered, ParserFilter.ACCEPT_ALL, chunkWriter);
304             ChunkParser chunkParser = new ChunkParser(chunkWriter.getInput(), pc, new ParserState());
305             chunkWriter.beginChunk(chunkParser.getHeader());
306             return chunkParser;
307         } else {
308             return new ChunkParser(input, parserState);
309         }
310     }
311 
312     private void endChunkParser() throws IOException {
313         if (chunkWriter != null) {
314             chunkWriter.endChunk(chunkParser.getHeader());
315         }
316     }
317 
318     private ChunkParser nextChunkParser() throws IOException {
319         if (chunkWriter != null) {
320             chunkWriter.endChunk(chunkParser.getHeader());
321         }
322         ChunkParser next = chunkParser.nextChunkParser();
323         if (chunkWriter != null) {
324             chunkWriter.beginChunk(next.getHeader());
325         }
326         return next;
327     }
328 
329     private void ensureOpen() throws IOException {
330         if (input == null) {
331             throw new IOException("Stream Closed");
332         }
333     }
