 25 package jdk.management.jfr;
 26 
 27 import java.io.IOException;


 28 import java.time.DateTimeException;
 29 import java.time.Duration;
 30 import java.time.Instant;
 31 import java.time.format.DateTimeParseException;
 32 import java.util.List;
 33 import java.util.function.Function;
 34 import java.util.stream.Collectors;
 35 
 36 import javax.management.MalformedObjectNameException;
 37 import javax.management.ObjectName;
 38 
 39 import jdk.jfr.Recording;
 40 import jdk.jfr.internal.management.ManagementSupport;
 41 
 42 final class MBeanUtils {
 43 



 44     static ObjectName createObjectName() {
 45         try {
 46             return new ObjectName(FlightRecorderMXBean.MXBEAN_NAME);
 47         } catch (MalformedObjectNameException mne) {
 48             throw new Error("Can't happen", mne);
 49         }
 50     }
 51 
















 52     static <T, R> List<R> transformList(List<T> source, Function<T, R> function) {
 53         return source.stream().map(function).collect(Collectors.toList());
 54     }
 55 
 56     static boolean booleanValue(String s) {
 57         if ("true".equals(s)) {
 58             return true;
 59         }
 60         if ("false".equals(s)) {
 61             return false;
 62         }
 63         throw new IllegalArgumentException("Value must be true or false.");
 64     }
 65 
 66     static Duration duration(String s) throws NumberFormatException {
 67         if (s == null) {
 68             return null;
 69         }
 70         long l = ManagementSupport.parseTimespan(s);
 71         if (l == 0) {
 72             return null;
 73         }
 74         return Duration.ofNanos(l);
 75     }
 76 
 77     public static Instant parseTimestamp(String s, Instant defaultValue) {
 78         if (s == null) {
 79             return defaultValue;
 80         }
 81         try {
 82             return Instant.parse(s);
 83         } catch(DateTimeParseException e ) {
 84             // OK, try with milliseconds since epoch
 85             // before giving up.
 86         }
 87         try {
 88             return Instant.ofEpochMilli(Long.parseLong(s));
 89         } catch (NumberFormatException | DateTimeException nfr) {
 90             throw new IllegalArgumentException("Not a valid timestamp " + s);
 91         }
 92     }
 93 
 94     static Long size(String s) throws NumberFormatException {
 95         long size = Long.parseLong(s);
 96         if (size < 0) {
 97             throw new IllegalArgumentException("Negative size not allowed");
 98         }
 99         return size;
100     }
101 
102     public static int parseBlockSize(String string, int defaultSize) {
103         if (string == null) {
104             return defaultSize;
105         }
106         int size = Integer.parseInt(string);
107         if (size <1)  {
108             throw new IllegalArgumentException("Block size must be at least 1 byte");
109         }
110         return size;
111     }
112 
113     public static String destination(Recording recording, String destination) throws IllegalArgumentException{
114         try {
115             ManagementSupport.checkSetDestination(recording, destination);
116             return destination;
117         }catch(IOException e){
118             IllegalArgumentException iae = new IllegalArgumentException("Not a valid destination " + destination);
119             iae.addSuppressed(e);
120             throw iae;
121         }
122     }
123 }
