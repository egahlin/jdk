 25 package jdk.management.jfr;
 26 
 27 import java.io.IOException;
 28 import java.lang.management.ManagementPermission;
 29 import java.security.Permission;
 30 import java.time.DateTimeException;
 31 import java.time.Duration;
 32 import java.time.Instant;
 33 import java.time.format.DateTimeParseException;
 34 import java.util.List;
 35 import java.util.function.Function;
 36 import java.util.stream.Collectors;
 37 
 38 import javax.management.MalformedObjectNameException;
 39 import javax.management.ObjectName;
 40 
 41 import jdk.jfr.Recording;
 42 import jdk.jfr.internal.management.ManagementSupport;
 43 
 44 final class MBeanUtils {
 45 
 46     private static final Permission monitor = new ManagementPermission("monitor");
 47     private static final Permission control = new ManagementPermission("control");
 48 
 49     static ObjectName createObjectName() {
 50         try {
 51             return new ObjectName(FlightRecorderMXBean.MXBEAN_NAME);
 52         } catch (MalformedObjectNameException mne) {
 53             throw new Error("Can't happen", mne);
 54         }
 55     }
 56 
 57     static void checkControl() {
 58         @SuppressWarnings("removal")
 59         SecurityManager secManager = System.getSecurityManager();
 60         if (secManager != null) {
 61             secManager.checkPermission(control);
 62         }
 63     }
 64 
 65     static void checkMonitor() {
 66         @SuppressWarnings("removal")
 67         SecurityManager secManager = System.getSecurityManager();
 68         if (secManager != null) {
 69             secManager.checkPermission(monitor);
 70         }
 71     }
 72 
 73     static <T, R> List<R> transformList(List<T> source, Function<T, R> function) {
 74         return source.stream().map(function).collect(Collectors.toList());
 75     }
 76 
 77     static boolean booleanValue(String s) {
 78         if ("true".equals(s)) {
 79             return true;
 80         }
 81         if ("false".equals(s)) {
 82             return false;
 83         }
 84         throw new IllegalArgumentException("Value must be true or false.");
 85     }
 86 
 87     static Duration duration(String s) throws NumberFormatException {
 88         if (s == null) {
 89             return null;
 90         }
 91         long l = ManagementSupport.parseTimespan(s);
 92         if (l == 0) {
 93             return null;
 94         }
 95         return Duration.ofNanos(l);
 96     }
 97 
 98     public static Instant parseTimestamp(String s, Instant defaultValue) {
 99         if (s == null) {
100             return defaultValue;
101         }
102         try {
103             return Instant.parse(s);
104         } catch(DateTimeParseException e ) {
105             // OK, try with milliseconds since epoch
106             // before giving up.
107         }
108         try {
109             return Instant.ofEpochMilli(Long.parseLong(s));
110         } catch (NumberFormatException | DateTimeException nfr) {
111             throw new IllegalArgumentException("Not a valid timestamp " + s);
112         }
113     }
114 
115     static Long size(String s) throws NumberFormatException {
116         long size = Long.parseLong(s);
117         if (size < 0) {
118             throw new IllegalArgumentException("Negative size not allowed");
119         }
120         return size;
121     }
122 
123     public static int parseBlockSize(String string, int defaultSize) {
124         if (string == null) {
125             return defaultSize;
126         }
127         int size = Integer.parseInt(string);
128         if (size <1)  {
129             throw new IllegalArgumentException("Block size must be at least 1 byte");
130         }
131         return size;
132     }
133 
134     public static String destination(Recording recording, String destination) throws IllegalArgumentException{
135         try {
136             ManagementSupport.checkSetDestination(recording, destination);
137             return destination;
138         }catch(IOException e){
139             IllegalArgumentException iae = new IllegalArgumentException("Not a valid destination " + destination);
140             iae.addSuppressed(e);
141             throw iae;
142         }
143     }
144 }
