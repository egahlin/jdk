 26 import java.io.FileWriter;

 32 import jdk.jfr.Event;

 35 import jdk.jfr.consumer.RecordedEvent;
 36 import jdk.jfr.consumer.RecordingFile;
 37 import jdk.jfr.internal.Repository;

 42 /**
 43  * @test
 44  * @summary Test jfr reconstruct
 45  * @key jfr
 46  * @requires vm.hasJFR
 47  * @library /test/lib /test/jdk
 48  * @modules jdk.jfr/jdk.jfr.internal
 49  * @run main/othervm jdk.jfr.tool.TestAssemble
 50  */
 51 public class TestAssemble {
 52 
 53     @Name("Correlation")
 54     static class CorrelationEvent extends Event {
 55         int id;
 56     }
 57     private static int RECORDING_COUNT = 5;
 58 

 63         for (int i = 0; i < RECORDING_COUNT; i++) {
 64             Recording r = new Recording();
 65             r.setToDisk(true);
 66             r.start();
 67             CorrelationEvent ce = new CorrelationEvent();
 68             ce.id = i;
 69             ce.commit();
 70             r.stop();
 71             recordings[i] = r;
 72         }
 73         Path dir = Paths.get("reconstruction-parts");
 74         Files.createDirectories(dir);
 75 
 76         long expectedCount = 0;
 77         for (int i = 0; i < RECORDING_COUNT; i++) {
 78             Path tmp = dir.resolve("chunk-part-" + i + ".jfr");
 79             recordings[i].dump(tmp);
 80             expectedCount += countEventInRecording(tmp);
 81         }
 82 
 83         SafePath repository = Repository.getRepository().getRepositoryPath();
 84         Path destinationPath = Paths.get("reconstructed.jfr");
 85 
 86         String directory = repository.toString();
 87         String destination = destinationPath.toAbsolutePath().toString();
 88 
 89         // Test failure
 90         OutputAnalyzer output = ExecuteHelper.jfr("assemble");
 91         output.shouldContain("too few arguments");
 92 
 93         output = ExecuteHelper.jfr("assemble", directory);
 94         output.shouldContain("too few arguments");
 95 
 96         output = ExecuteHelper.jfr("assemble", "not-a-directory", destination);
 97         output.shouldContain("directory does not exist, not-a-directory");
 98 
 99         output = ExecuteHelper.jfr("assemble", directory, "not-a-destination");
100         output.shouldContain("filename must end with '.jfr'");
101 
102         output = ExecuteHelper.jfr("assemble", "--wrongOption", directory, destination);
103         output.shouldContain("too many arguments");
