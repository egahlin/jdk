 24 package jdk.jfr.tool;
 25 
 26 import java.io.FileWriter;
 27 import java.io.IOException;
 28 import java.nio.file.Files;
 29 import java.nio.file.Path;
 30 import java.nio.file.Paths;
 31 
 32 import jdk.jfr.Event;
 33 import jdk.jfr.Name;
 34 import jdk.jfr.Recording;
 35 import jdk.jfr.consumer.RecordedEvent;
 36 import jdk.jfr.consumer.RecordingFile;
 37 import jdk.jfr.internal.Repository;
 38 import jdk.jfr.internal.SecuritySupport.SafePath;
 39 import jdk.test.lib.Asserts;
 40 import jdk.test.lib.process.OutputAnalyzer;
 41 
 42 /**
 43  * @test
 44  * @summary Test jfr reconstruct
 45  * @key jfr
 46  * @requires vm.hasJFR
 47  * @library /test/lib /test/jdk
 48  * @modules jdk.jfr/jdk.jfr.internal
 49  * @run main/othervm jdk.jfr.tool.TestAssemble
 50  */
 51 public class TestAssemble {
 52 
 53     @Name("Correlation")
 54     static class CorrelationEvent extends Event {
 55         int id;
 56     }
 57     private static int RECORDING_COUNT = 5;
 58 
 59     @SuppressWarnings("resource")
 60     public static void main(String[] args) throws Throwable {
 61         // Create some disk recordings
 62         Recording[] recordings = new Recording[5];
 63         for (int i = 0; i < RECORDING_COUNT; i++) {
 64             Recording r = new Recording();
 65             r.setToDisk(true);
 66             r.start();
 67             CorrelationEvent ce = new CorrelationEvent();
 68             ce.id = i;
 69             ce.commit();
 70             r.stop();
 71             recordings[i] = r;
 72         }
 73         Path dir = Paths.get("reconstruction-parts");
 74         Files.createDirectories(dir);
 75 
 76         long expectedCount = 0;
 77         for (int i = 0; i < RECORDING_COUNT; i++) {
 78             Path tmp = dir.resolve("chunk-part-" + i + ".jfr");
 79             recordings[i].dump(tmp);
 80             expectedCount += countEventInRecording(tmp);
 81         }
 82 
 83         SafePath repository = Repository.getRepository().getRepositoryPath();
 84         Path destinationPath = Paths.get("reconstructed.jfr");
 85 
 86         String directory = repository.toString();
 87         String destination = destinationPath.toAbsolutePath().toString();
 88 
 89         // Test failure
 90         OutputAnalyzer output = ExecuteHelper.jfr("assemble");
 91         output.shouldContain("too few arguments");
 92 
 93         output = ExecuteHelper.jfr("assemble", directory);
 94         output.shouldContain("too few arguments");
 95 
 96         output = ExecuteHelper.jfr("assemble", "not-a-directory", destination);
 97         output.shouldContain("directory does not exist, not-a-directory");
 98 
 99         output = ExecuteHelper.jfr("assemble", directory, "not-a-destination");
100         output.shouldContain("filename must end with '.jfr'");
101 
102         output = ExecuteHelper.jfr("assemble", "--wrongOption", directory, destination);
103         output.shouldContain("too many arguments");
104 
105         FileWriter fw = new FileWriter(destination);
106         fw.write('d');
107         fw.close();
108         output = ExecuteHelper.jfr("assemble", directory, destination);
109         output.shouldContain("already exists");
110         Files.delete(destinationPath);
111 
112         // test success
113         output = ExecuteHelper.jfr("assemble", directory, destination);
114         System.out.println(output.getOutput());
115         output.shouldContain("Finished.");
116 
117         long reconstructedCount = countEventInRecording(destinationPath);
118         Asserts.assertEquals(expectedCount, reconstructedCount);
119         // Cleanup
120         for (int i = 0; i < RECORDING_COUNT; i++) {
121             recordings[i].close();
122         }
123     }
124 
125     private static long countEventInRecording(Path file) throws IOException {
126         Integer lastId = -1;
127         try (RecordingFile rf = new RecordingFile(file)) {
128             long count = 0;
129             while (rf.hasMoreEvents()) {
130                 RecordedEvent re = rf.readEvent();
131                 if (re.getEventType().getName().equals("Correlation")) {
132                     Integer id = re.getValue("id");
133                     if (id < lastId) {
134                         Asserts.fail("Expected chunk number to increase");
135                     }
136                     lastId = id;
137                 }
138                 count++;
139             }
140             return count;
141         }
142     }
143 }
