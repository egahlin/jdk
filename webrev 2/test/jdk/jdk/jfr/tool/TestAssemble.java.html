 24 package jdk.jfr.tool;
 25 
 26 import java.io.FileWriter;
 27 import java.io.IOException;
 28 import java.nio.file.Files;
 29 import java.nio.file.Path;
 30 import java.nio.file.Paths;
 31 
 32 import jdk.jfr.Event;
 33 import jdk.jfr.Name;
 34 import jdk.jfr.Recording;
 35 import jdk.jfr.consumer.RecordedEvent;
 36 import jdk.jfr.consumer.RecordingFile;
 37 import jdk.jfr.internal.Repository;
 38 import jdk.test.lib.Asserts;
 39 import jdk.test.lib.process.OutputAnalyzer;
 40 
 41 /**
 42  * @test
 43  * @summary Test jfr reconstruct
 44  * @key jfr
 45  * @requires vm.hasJFR
 46  * @library /test/lib /test/jdk
 47  * @modules jdk.jfr/jdk.jfr.internal
 48  * @run main/othervm jdk.jfr.tool.TestAssemble
 49  */
 50 public class TestAssemble {
 51 
 52     @Name("Correlation")
 53     static class CorrelationEvent extends Event {
 54         int id;
 55     }
 56     private static int RECORDING_COUNT = 5;
 57 
 58     @SuppressWarnings("resource")
 59     public static void main(String[] args) throws Throwable {
 60         // Create some disk recordings
 61         Recording[] recordings = new Recording[5];
 62         for (int i = 0; i < RECORDING_COUNT; i++) {
 63             Recording r = new Recording();
 64             r.setToDisk(true);
 65             r.start();
 66             CorrelationEvent ce = new CorrelationEvent();
 67             ce.id = i;
 68             ce.commit();
 69             r.stop();
 70             recordings[i] = r;
 71         }
 72         Path dir = Paths.get("reconstruction-parts");
 73         Files.createDirectories(dir);
 74 
 75         long expectedCount = 0;
 76         for (int i = 0; i < RECORDING_COUNT; i++) {
 77             Path tmp = dir.resolve("chunk-part-" + i + ".jfr");
 78             recordings[i].dump(tmp);
 79             expectedCount += countEventInRecording(tmp);
 80         }
 81 
 82         Path repository = Repository.getRepository().getRepositoryPath();
 83         Path destinationPath = Paths.get("reconstructed.jfr");
 84 
 85         String directory = repository.toString();
 86         String destination = destinationPath.toAbsolutePath().toString();
 87 
 88         // Test failure
 89         OutputAnalyzer output = ExecuteHelper.jfr("assemble");
 90         output.shouldContain("too few arguments");
 91 
 92         output = ExecuteHelper.jfr("assemble", directory);
 93         output.shouldContain("too few arguments");
 94 
 95         output = ExecuteHelper.jfr("assemble", "not-a-directory", destination);
 96         output.shouldContain("directory does not exist, not-a-directory");
 97 
 98         output = ExecuteHelper.jfr("assemble", directory, "not-a-destination");
 99         output.shouldContain("filename must end with '.jfr'");
100 
101         output = ExecuteHelper.jfr("assemble", "--wrongOption", directory, destination);
102         output.shouldContain("too many arguments");
103 
104         FileWriter fw = new FileWriter(destination);
105         fw.write('d');
106         fw.close();
107         output = ExecuteHelper.jfr("assemble", directory, destination);
108         output.shouldContain("already exists");
109         Files.delete(destinationPath);
110 
111         // test success
112         output = ExecuteHelper.jfr("assemble", directory, destination);
113         System.out.println(output.getOutput());
114         output.shouldContain("Finished.");
115 
116         long reconstructedCount = countEventInRecording(destinationPath);
117         Asserts.assertEquals(expectedCount, reconstructedCount);
118         // Cleanup
119         for (int i = 0; i < RECORDING_COUNT; i++) {
120             recordings[i].close();
121         }
122     }
123 
124     private static long countEventInRecording(Path file) throws IOException {
125         Integer lastId = -1;
126         try (RecordingFile rf = new RecordingFile(file)) {
127             long count = 0;
128             while (rf.hasMoreEvents()) {
129                 RecordedEvent re = rf.readEvent();
130                 if (re.getEventType().getName().equals("Correlation")) {
131                     Integer id = re.getValue("id");
132                     if (id < lastId) {
133                         Asserts.fail("Expected chunk number to increase");
134                     }
135                     lastId = id;
136                 }
137                 count++;
138             }
139             return count;
140         }
141     }
142 }
