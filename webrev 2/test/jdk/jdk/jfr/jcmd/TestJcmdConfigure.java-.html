 24 package jdk.jfr.jcmd;
 25 
 26 import java.io.File;
 27 import java.nio.file.Files;
 28 import java.nio.file.Paths;
 29 import java.util.ArrayList;
 30 import java.util.List;
 31 
 32 import jdk.jfr.internal.Repository;
 33 import jdk.jfr.internal.SecuritySupport.SafePath;
 34 import jdk.jfr.internal.Options;
 35 import jdk.test.lib.Asserts;
 36 import jdk.test.lib.Utils;
 37 
 38 /**
 39  * @test
 40  * @summary The test verifies JFR.configure command
 41  * @key jfr
 42  * @requires vm.hasJFR
 43  * @library /test/lib /test/jdk
 44  * @modules jdk.jfr/jdk.jfr.internal
 45  * @run main/othervm -Xlog:jfr=info:file=jfr_info.txt jdk.jfr.jcmd.TestJcmdConfigure
 46  */
 47 public class TestJcmdConfigure {
 48 
 49     private static final String DUMPPATH = "dumppath";
 50     private static final String STACK_DEPTH = "stackdepth";
 51     private static final String GLOBAL_BUFFER_COUNT = "globalbuffercount";
 52     private static final String GLOBAL_BUFFER_SIZE = "globalbuffersize";
 53     private static final String THREAD_BUFFER_SIZE = "thread_buffer_size";
 54     private static final String MAX_CHUNK_SIZE = "maxchunksize";
 55     private static final String UNSUPPORTED_OPTION = "unsupportedoption";
 56 
 57     private static final String REPOSITORYPATH_1 = "." + File.pathSeparator + "repo1";
 58     private static final String REPOSITORYPATH_2 = "." + File.pathSeparator + "repo2";
 59 
 60     private static final String REPOSITORYPATH_SETTING_1 = "repositorypath="+REPOSITORYPATH_1;
 61     private static final String REPOSITORYPATH_SETTING_2 = "repositorypath="+REPOSITORYPATH_2;
 62     private static final String JFR_UNIFIED_LOG_FILE = "jfr_info.txt";
 63 
 64     public static void main(String[] args) throws Exception {
 65         //
 66         // Simple sanity tests against what is available in Java,
 67         // before Flight Recorder is loaded. To do:
 68         //
 69         // - set values when JFR is running, check for errors.
 70         // - validate against output from JFR.configure
 71         // - where feasible, check if they are respected
 72         //
 73 
 74         String dumpPath = Utils.createTempDirectory("dump-path-").toAbsolutePath().toString();
 75 
 76         test(DUMPPATH, dumpPath);
 77         test(STACK_DEPTH, 15);
 78         test(GLOBAL_BUFFER_COUNT, 7);
 79         test(GLOBAL_BUFFER_SIZE, 6);
 80         test(THREAD_BUFFER_SIZE, 5);
 81         test(MAX_CHUNK_SIZE, 14 * 1000 * 1000);
 82         testNegative(UNSUPPORTED_OPTION, 100000);
 83         testNegative(MAX_CHUNK_SIZE, -500);
 84 
 85         testRepository();
 86 
 87         if (!testExceptions.isEmpty()) {
 88             for (Exception e : testExceptions) {
 89                 System.out.println("Error: " + e.getMessage());
 90             }
 91             throw testExceptions.getFirst();
 92         }
 93     }
 94 
 95     private static List<Exception> testExceptions = new ArrayList<>();
 96 
 97     private static void test(String configName, Object value) {
 98         JcmdHelper.jcmd("JFR.configure", configName + "=" + value);
 99         Object actualValue = getOption(configName);
100         System.out.format("Test param='%s', expected='%s', actual='%s'%n", configName, value, actualValue);
101         try {
102             // Need convert to string to compare Integer and Long
103             Asserts.assertEquals(value.toString(), actualValue.toString(), "Wrong JFR.configure " + configName);
104         } catch (Exception e) {
105             testExceptions.add(e);
106         }
107     }
108 
109     private static void testNegative(String configName, Object value) {
110         try {
111             JcmdHelper.jcmd(1, "JFR.configure", configName + "=" + value);
112         } catch(Exception e) {
113             testExceptions.add(e);
114         }
115     }
116 
117     private static Object getOption(String name) {
118         switch (name) {
119             case DUMPPATH: return Options.getDumpPath().toString();
120             case STACK_DEPTH: return Options.getStackDepth();
121             case GLOBAL_BUFFER_COUNT: return Options.getGlobalBufferCount();
122             case GLOBAL_BUFFER_SIZE: return Options.getGlobalBufferSize();
123             case THREAD_BUFFER_SIZE: return Options.getThreadBufferSize();
124             case MAX_CHUNK_SIZE: return Options.getMaxChunkSize();
125             default: throw new RuntimeException("Unknown option " + name);
126         }
127     }
128 
129     private static void testRepository(){
130         final String findWhat = "[info][jfr] Same base repository path " + REPOSITORYPATH_1 + " is set";
131 
132         try {
133             JcmdHelper.jcmd("JFR.configure", REPOSITORYPATH_SETTING_1);
134             SafePath initialPath = Repository.getRepository().getRepositoryPath();
135 
136             JcmdHelper.jcmd("JFR.configure", REPOSITORYPATH_SETTING_1);
137             SafePath samePath = Repository.getRepository().getRepositoryPath();
138             Asserts.assertTrue(samePath.equals(initialPath));
139 
140             List<String> lines = Files.readAllLines(Paths.get(JFR_UNIFIED_LOG_FILE));
141             Asserts.assertTrue(lines.stream().anyMatch(l->l.contains(findWhat)));
142 
143             JcmdHelper.jcmd("JFR.configure", REPOSITORYPATH_SETTING_2);
144             SafePath changedPath = Repository.getRepository().getRepositoryPath();
145 
146             Asserts.assertFalse(changedPath.equals(initialPath));
147 
148         } catch(Exception e) {
149             testExceptions.add(e);
150         }
151     }
152 }
